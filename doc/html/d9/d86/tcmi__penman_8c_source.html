131 /**
132  * @return 1, if there is a duplicate address for some of the peers
133  */
134 static int check_duplicates(const char* address, int address_len) {
135  int ret = 0;
136  struct tcmi_slot *slot;
137  tcmi_slot_node_t* node;
138  struct tcmi_migman *migman = NULL;
139  tcmi_slotvec_lock(TCMI_MAN(&self)->mig_mans);
140 
141  tcmi_slotvec_for_each_used_slot(slot, TCMI_MAN(&self)->mig_mans) {
142  tcmi_slot_for_each(node, slot) {
143  migman = tcmi_slot_entry(node, struct tcmi_migman, node);
144 
145  if ( kkc_sock_is_address_equal_to(tcmi_migman_sock(migman),address,address_len, 1) ) {
146  ret = 1;
147  goto found;
148  }
149  };
150  };
151 
152 found:
153  tcmi_slotvec_unlock(TCMI_MAN(&self)->mig_mans);
154  return ret;
155 }

157 /**
158  * <<private>> Creates a new connection and instantiates a new
159  * migration manager that will be responsible for handling it.
160  *
161  * @param *obj - pointer to an object - NULL is expected as
162  * TCMI PEN manager is a singleton class.
163  * @param *str - connection string - It is architecture dependent
164  * and passed further on tcmi_listen class. "@" char is used to separate address of the remote node and optional authentication data
165  * @return 0 upon success
166  */
167 static int tcmi_penman_connect(void *obj, void *str)
168 {
169  int err, separation_index, auth_data_size = 0;
170  char *connect_str = (char *) str;
171  char *auth_data = NULL;
172  char *separation_char = NULL;
173  struct tcmi_migman *migman;
174  struct tcmi_slot *slot;
175  struct kkc_sock *sock;
176 
177  mdbg(INFO3, "Connection request with param: %s", connect_str);
178 
179  // @ serves as a special separator of "node address" from "authentication data
180  if ( (separation_char=strchr(connect_str, '@')) != NULL ) {
181  separation_index = separation_char - connect_str;
182  // TODO: Here we use strlen on auth data length calculation, which means we loose a capability of passing pure binary data.. improve it if possible
183  auth_data_size = strlen(connect_str) - separation_index - 1;
184  // We separate the string by zero so that it is internally threated as a short string for kkc soct
185  connect_str[separation_index] = '\0';
186  auth_data = connect_str + separation_index + 1;
187  }
188 
189  if ( check_duplicates(connect_str, strlen(connect_str)) ) {
190  mdbg(ERR3, "Duplicate peer, already connected:'%s' ", connect_str);
191  goto exit0;
192  }
193 
194 
195  /* check if there is incoming connection */
196  if (kkc_connect(&sock, connect_str)) {
197  mdbg(ERR3, "Failed connecting to '%s' ", connect_str);
198  goto exit0;
199  }
200  mdbg(INFO3, "Connected to local: '%s', remote: '%s', auth_data_size: %d",
201  kkc_sock_getsockname2(sock), kkc_sock_getpeername2(sock), auth_data_size);
202  /* Reserve an empty slot */
203  if (!(slot = tcmi_man_reserve_migmanslot(TCMI_MAN(&self)))) {
204  mdbg(ERR3, "Failed allocating an empty slot for PEN migman");
205  goto exit1;
206  }
207 
208  /* Accept the first incoming connection */
209  if (!(migman = tcmi_penmigman_new(sock,
210  tcmi_man_id(TCMI_MAN(&self)),
211  slot,
212  tcmi_man_nodes_dir(TCMI_MAN(&self)),
213  tcmi_man_migproc_dir(TCMI_MAN(&self)),
214  "%d", tcmi_slot_index(slot)))) {
215  minfo(ERR1, "Failed to instantiate the PEN migration manager");
216  goto exit2;
217  }
218 
219  if ((err = tcmi_penmigman_auth_ccn(TCMI_PENMIGMAN(migman), auth_data_size, auth_data)) < 0) {
220  minfo(ERR1, "Failed to authenticate the CCN! %d", err);
221  goto exit3;
222  }
223  tcmi_slot_insert(slot, tcmi_migman_node(migman));
224 
225  kkc_sock_put(sock);
226 
227  return 0;
228 /* error handling */
229  exit3:
230  tcmi_migman_put(migman);
231  exit2:
232  tcmi_man_release_migmanslot(TCMI_MAN(&self), slot);
233  exit1:
234  kkc_sock_put(sock);
235  exit0:
236  return -EINVAL;
237 }

250  * Performs non-preemtive migration back from PEN.
251  * Operates on "current" task
252  */
253 int tcmi_penman_migrateback_npm(struct tcmi_man *self, struct pt_regs* regs, struct tcmi_npm_params* npm_params) {
254  struct tcmi_task* task = current->tcmi.tcmi_task;
255 
256  BUG_ON(current->tcmi.tcmi_task == NULL);
257  BUG_ON(current->tcmi.task_type != guest);
258 
259  // Submits task specific method for performing npm migback
260  tcmi_task_submit_method(task, tcmi_task_migrateback_npm, npm_params, sizeof(*npm_params));
261  // And enter directly to migmide as we are already in the task context
262  tcmi_taskhelper_do_mig_mode(regs);
263 
264  return 0;
265 }
