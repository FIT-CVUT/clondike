36 /** @defgroup tcmi_p_emigrate_msg_class tcmi_p_emigrate_msg class
37  *
38  * @ingroup tcmi_msg_class
39  *
40  * This class represents a message for preemptive/non-preemtive process migration
41  * via a physical checkpoint image. The migrating process sends this
42  * message to the PEN, that is responsible for creation of a guest
43  * process and finishing the migration. Should the migration fail, an
44  * error is generated and sent back as a reply. This is a migration
45  * control message that will be received by the migration manager and
46  * passed onto the newly created guest. The response to this message
47  * is however a process control message (\link
48  * tcmi_guest_started_procmsg_class TCMI_GUEST_STARTED_PROCMSG
49  * \endlink) that will be directly routed to the sending task.
50  *
51  * The sending task specifies it's PID, so that the receiving guest can use
52  * it for further communication via process control connection. Also,
53  * the checkpoint file name is specified as part of the message.
54  *
55  * @{
56  */
57 
58 /** Compound structure, inherits from tcmi_msg_class */
59 struct tcmi_p_emigrate_msg {
60  /** parent class instance. */
61  struct tcmi_msg super;
62  /** groups pid and size, so they can be sent/received at once. */
63  struct {
64  /** remote PID, used for further communication. */
65  pid_t reply_pid;
66  /** size of the checkpoint name in bytes (including
67  trailing zero) */
68  int32_t size;
69  /** size of the exec name in bytes (including
70  trailing zero) */
71  int32_t exec_name_size;
72  /** effective user id of a process as seen by the node where it is running */
73  int16_t euid;
74  /** effective group id of a process as seen by the node where it is running */
75  int16_t egid;
76  /** fsuid of the process on the core node.. may be needed for DFS mount before the checkpoint is read */
77  int16_t fsuid;
78  /** fsgid of the process on the core node.. may be needed for DFS mount before the checkpoint is read */
79  int16_t fsgid;
80  } pid_and_size __attribute__((__packed__));
81 
82  /** name of the executable of the process. */
83  char *exec_name;
84  /** name of the checkpoint file. */
85  char *ckpt_name;
86 };
