50 /**
51  * <<private>> Helper method that handles both PPM and NPM checkpoint creation
52  *
53  * This consists of:
54  * - creating a new checkpoint instance
55  * - writing a checkpoint header - might fail if the process can't be
56  * checkpointed (e.g. has open files or memory areas that are not
57  * supported)
58  * - writing open files
59  * - writing memory descriptor
60  * - writing memory areas along with pages
61  * - writing process state (registers)
62  * - writing signal handlers
63  *
64  * @param *file - file where the checkpoint is to be stored
65  * @param *regs - registers of the checkpointed process
66  * @param heavy - if set a full checkpoint of all process pages
67  * is made, all memory mapped files will be then restored
68  * from the checkpoint image. Setting this options yields
69  * a bigger checkpoint image size..
70  * @param npm_params - Non-preemptive checkpoint params, or null if we are doing PPM checkpoint
71  *
72  * @return 0 upon success
73  */
74 static int tcmi_ckptcom_checkpoint(struct file *file, struct pt_regs *regs,
75  int heavy, struct tcmi_npm_params* npm_params)
76 {
77  struct tcmi_ckpt *ckpt;
78  int is_npm = npm_params != NULL;
79  u64 beg_time, end_time;
80 
81  beg_time = cpu_clock(smp_processor_id());
82 
83  mdbg(INFO3, "Start checkpointing. Is_npm: %d", is_npm);
84  if ( !regs ) {
85  mdbg(ERR3, "Failed to create a checkpoint file. No regs provided");
86  goto exit0;
87  }
88 
89  ckpt = tcmi_ckpt_new(file);
90  if ( IS_ERR(ckpt) ) {
91  mdbg(ERR3, "Failed to create a checkpoint file. Err: %ld", PTR_ERR(ckpt));
92  goto exit0;
93  }
94  if ( ckpt == NULL ) {
95  mdbg(ERR3, "Failed to create a checkpoint file.");
96  goto exit0;
97  }
98 
99  if (tcmi_ckpt_write_hdr(ckpt, is_npm) < 0) {
100  mdbg(ERR3, "Error writing checkpoint header!");
101  goto exit1;
102  }
103  if (tcmi_ckpt_write_rlimit(ckpt, current) < 0) {
104  mdbg(ERR3, "Error writing checkpoint rlimit!");
105  goto exit1;
106  }
107  if (tcmi_ckpt_write_files(ckpt) < 0) {
108  mdbg(ERR3, "Error writing checkpoint files!");
109  goto exit1;
110  }
111  if (tcmi_ckpt_mm_write(ckpt) < 0) {
112  mdbg(ERR3, "Error writing memory descriptor!");
113  goto exit1;
114  }
115 
116  if ( !is_npm ) {
117  if (tcmi_ckpt_write_vmas(ckpt, heavy) < 0) {
118  mdbg(ERR3, "Error writing VM areas type: %d", heavy);
119  goto exit1;
120  }
121  }
122 
123  if (tcmi_ckpt_regs_write(ckpt, regs) < 0) {
124  mdbg(ERR3, "Error writing processor registers descriptor!");
125  goto exit1;
126  }
127  if (tcmi_ckpt_tls_write(ckpt, current) < 0) {
128  mdbg(ERR3, "Error writing process tls!");
129  goto exit1;
130  }
131  if (tcmi_ckpt_fsstruct_write(ckpt, current) < 0) {
132  mdbg(ERR3, "Error writing process fs struct!");
133  goto exit1;
134  }
135  if (tcmi_ckpt_sig_write(ckpt) < 0) {
136  mdbg(ERR3, "Error writing signal data!");
137  goto exit1;
138  }
139 
140  if ( is_npm ) {
141  if (tcmi_ckpt_npm_params_write(ckpt, npm_params) < 0) {
142  mdbg(ERR3, "Error writing npm params!");
143  goto exit1;
144  }
145  }
146 
147  end_time = cpu_clock(smp_processor_id());
148  mdbg(INFO3, "Checkpoint (npm: %d) took '%llu' ms.'", is_npm, (end_time - beg_time) / 1000000);
149  printk("Checkpoint (npm: %d) took '%llu' ms.\n'", is_npm, (end_time - beg_time) / 1000000);
150 
151 
152  tcmi_ckpt_put(ckpt);
153  return 0;
154 
155  /* error handling */
156  exit1:
157  tcmi_ckpt_put(ckpt);
158  exit0:
159  return -ENOEXEC;
160 }

172 /**
173  * <<public>> Restarts a process from a checkpoint (new binfmt
174  * handler). This consists of:
175  *
176  * - creating a new checkpoint instance
177  * - reading a checkpoint header - might fail, if the magic number
178  * doesn't match.
179  * - reading open files
180  * - reading memory descriptor
181  * - reading memory areas along with pages
182  * - reading process state (registers)
183  * - reading signal handlers
184  *
185  * @param *bprm - binary object that is passed to this method by
186  * execve and contains a pointer to the executable file
187  * @param *regs - references the registers of the process, so that
188  * they can be overlayed from the checkpoint.
189  * @return 0 upon success
190  */
191 int tcmi_ckptcom_restart(struct linux_binprm *bprm, struct pt_regs *regs)
192 {
193  struct tcmi_ckpt *ckpt;
194  struct pt_regs* original_regs;
195 // int i;
196  u64 beg_time, end_time;
197 
198  beg_time = cpu_clock(smp_processor_id());
199 
200 
201  memory_sanity_check("Start");
202 
203  original_regs = kmalloc(sizeof(struct pt_regs), GFP_KERNEL);
204  if ( !original_regs )
205  return -ENOMEM;
206 
207  if (!tcmi_ckpt_check_magic(bprm->buf)) {
208  goto exit0;
209  }
210 
211  mdbg(INFO3, "Restarting '%s'", bprm->filename);
212  if (!(ckpt = tcmi_ckpt_new(bprm->file))) {
213  mdbg(ERR3, "Failed to instantiate a checkpoint");
214  goto exit0;
215  }
216  if (tcmi_ckpt_read_hdr(ckpt) < 0) {
217  mdbg(ERR3, "Error reading checkpoint header!");
218  goto exit1;
219  }
220 memory_sanity_check("Post header");
221  /* Flush all traces of the currently running executable */
222  if (flush_old_exec(bprm)) {
223  mdbg(ERR3, "Error flushing the old execution context!");
224  goto exit0;
225  }
226 memory_sanity_check("Pre-rlimit");
227  if (tcmi_ckpt_read_rlimit(ckpt, current) < 0) {
228  mdbg(ERR3, "Error reading checkpoint rlimit!");
229  goto exit1;
230  }
231 memory_sanity_check("Post-rlimit");
232  if ( FD_ISSET(0, current->files->fdt->open_fds) ) {
233  mdbg(INFO3, "Closing open fs.. this should not happend though..");
234  sys_close(0);
235  }
236 memory_sanity_check("Pre files");
237  if (tcmi_ckpt_read_files(ckpt) < 0) {
238  mdbg(ERR3, "Error reading checkpoint files!");
239  goto exit1;
240  }
241 memory_sanity_check("Post - files");
242  if (tcmi_ckpt_mm_read(ckpt) < 0) {
243  mdbg(ERR3, "Error reading memory descriptor!");
244  goto exit1;
245  }
246 memory_sanity_check("Post mm");
247  if ( !ckpt->hdr.is_npm ) {
248  if (tcmi_ckpt_read_vmas(ckpt) < 0) {
249  mdbg(ERR3, "Error reading VM areas");
250  goto exit1;
251  }
252  }
253 
254  *original_regs = *regs;
255  if (tcmi_ckpt_regs_read(ckpt, regs) < 0) {
256  mdbg(ERR3, "Error reading processor registers descriptor!");
257  goto exit1;
258  }
259  if (tcmi_ckpt_tls_read(ckpt, current, regs) < 0) {
260  mdbg(ERR3, "Error reading process tls!");
261  goto exit1;
262  }
263  if (tcmi_ckpt_fsstruct_read(ckpt, current) < 0) {
264  mdbg(ERR3, "Error reading process fsstruct!");
265  goto exit1;
266  }
267  if (tcmi_ckpt_sig_read(ckpt) < 0) {
268  mdbg(ERR3, "Error reading signals informations!");
269  goto exit1;
270  }
271  if ( ckpt->hdr.is_npm ) {
272  struct tcmi_npm_params* params = vmalloc(sizeof(struct tcmi_npm_params));
273  int exec_result = -EFAULT;
274  mm_segment_t old_fs;
275  *regs = *original_regs;
276 
277  if ( !params ) {
278  mdbg(ERR3, "Cannot allocate memory for npm params!");
279  goto exit1;
280  }
281 
282 
283  if (tcmi_ckpt_npm_params_read(ckpt, params) < 0) {
284  mdbg(ERR3, "Error reading npm params!");
285  goto exit1;
286  }
287  tcmi_ckpt_put(ckpt);
288 
289  // TEMPORARY DEBUG!
290  if ( current->mm )
291  mdbg(INFO2, "MM %p nr_ptes: %lu", current->mm, current->mm->nr_ptes);
292  if ( current->active_mm )
293  mdbg(INFO2, "ACTIVE %p MM nr_ptes: %lu", current->active_mm, current->active_mm->nr_ptes);
294 
295  // TODO: This is something REALLY NASTY! Some better solution is appreciated
296  if ( current->mm ) {
297  current->mm->nr_ptes = 0;
298  }
299 
300  old_fs = get_fs();
301  set_fs(KERNEL_DS);
302 
303  // TEMPORARY DEBUG!
304  if ( current->mm )
305  mdbg(INFO2, "MM %p nr_ptes: %lu", current->mm, current->mm->nr_ptes);
306  if ( current->active_mm )
307  mdbg(INFO2, "ACTIVE %p MM nr_ptes: %lu", current->active_mm, current->active_mm->nr_ptes);
308 
309  // We have to unlock temprarily guar to prevent recursive lock (we are calling recursive exceve). TODO: Some better solution?
310  mutex_unlock(&current->cred_guard_mutex);
311  current->fs->in_exec = 0; // Required to pass through 'check_unsafe_exec'
312 
313  mdbg(INFO3, "Going to execute '%s', Args: %d, Envps %d", params->file_name, params->argsc, params->envpc);
314  //mdbg(INFO3, "Arg[0] '%s', Envp[0] '%s'", params->args[0], params->envp[0]);
315 
316  exec_result = do_execve(params->file_name, (char __user *__user *)params->args, (char __user *__user *)params->envp, regs);
317  mdbg(INFO3, "NPM internal execution result %d", exec_result);
318 
319  // And now we relock again as the relock of outer execve will be attempted.
320  if (mutex_lock_interruptible(&current->cred_guard_mutex)) {
321  minfo(ERR3, "Failed to relock cred guard!");
322  }
323 
324  // TODO: How do we release reference to the binmt of the module that was used for this execve.. ?
325  // Do we have to call "set_binfmt(&tcmi_ckptcom_format);" here or can we release it here?
326  set_fs(old_fs);
327 
328  vfree(params);
329  mdbg(INFO3, "NPM after param free");
330 
331  if ( exec_result ) {// In case of error of internal execution, we have to return ENOEXEC
332  return -EFAULT;
333  }
334 
335  end_time = cpu_clock(smp_processor_id());
336  mdbg(INFO3, "Checkpoint NPM took '%llu' ms.'", (end_time - beg_time) / 1000000);
337  printk("Checkpoint NPM took '%llu' ms.\n'", (end_time - beg_time) / 1000000);
338 
339  return 0;
340  } else {
341  // Restart fixup is performed only in PPM
342  tcmi_resolve_restart_block(current, regs, ckpt->hdr.checkpoint_arch, ckpt->hdr.is_32bit_application);
343  }
344  kfree(original_regs);
345 
346  /* flush_signals(current);*/
347  tcmi_ckpt_put(ckpt);
348  /* successul execution of the image - need to set the format */
349  set_binfmt(&tcmi_ckptcom_format);
350 
351  mdbg(INFO3, "Checkpoint PPM took '%llu' ms.'", (end_time - beg_time) / 1000000);
352  printk("Checkpoint PPM took '%llu' ms.\n'", (end_time - beg_time) / 1000000);
353 
354  /* Something went wrong, return the inode and free the argument pages*/
355 /*
356 What is the point of the following code? It does not really seem to free memory, but in any case, it does not compile on latest kernels ;)
357  for (i = 0 ; i < MAX_ARG_PAGES ; i++) {
358  bprm->page[i] = NULL;
359  }
360 */
361 
362  return 0;
363 
364  /* error handling */
365  exit1:
366  tcmi_ckpt_put(ckpt);
367  exit0:
368  return -ENOEXEC;
369 }
