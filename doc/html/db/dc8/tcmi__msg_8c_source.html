41 static int tcmi_msg_init_shared(struct tcmi_msg *self, u_int32_t msg_id, struct tcmi_msg_ops *msg_ops) {
42  mdbg(INFO4, "Initialized message ID=%u, %p", msg_id, self);
43  self->msg_id = msg_id;
44  /* initially no transaction associated with the message */
45  self->trans_id.req = TCMI_TRANSACTION_INVAL_ID;
46  self->trans_id.resp = TCMI_TRANSACTION_INVAL_ID;
47  self->transaction = NULL;
48 
49  self->msg_ops = msg_ops;
50  atomic_set(&self->ref_count, 1);
51 
52  /* initialize list head for message queueing */
53  INIT_LIST_HEAD(&self->node);
54  return 0;
55 }

71 /**
72  * <<public>> Initializes the message for transferring. For
73  * oneway messages sets only message ID and message operations and
74  * associated transaction ID (for responses). If there is a response
75  * expected, sets up a new transaction.
76  *
77  * @param *self - pointer to this message instance
78  * @param msg_id - ID of a particular message type
79  * @param *msg_ops - pointer to the message specific operations
80  * @param *transactions - storage for the new transaction(NULL for one-way messages)
81  * @param resp_msg_id - response message ID - required for the transaction
82  * if the transaction expires(NULL for one-way messages)
83  * @param timeout - transaction time out (0 for one-way messages)
84  * @param resp_trans_id - response transaction ID - used when
85  * this message is a response to a particular request. The user must set
86  * this ID based on the value ofr req_trans_id of the previously received message.
87  * @return 0 upon success
88  */
89 int tcmi_msg_init_tx(struct tcmi_msg *self, u_int32_t msg_id,
90  struct tcmi_msg_ops *msg_ops,
91  struct tcmi_slotvec *transactions, u_int32_t resp_msg_id,
92  u_int timeout,
93  u_int32_t resp_trans_id)
94 {
95  int error = 0;
96  /* basic init is the same as when receiving a message */
97  tcmi_msg_init_shared(self, msg_id, msg_ops);
98  /* transaction ID, needed if this message is a response to a previously sent message */
99  self->trans_id.resp = resp_trans_id;
100  /* setup transaction when needed */
101  if (transactions) {
102  if (!(self->transaction =
103  tcmi_transaction_new(transactions,
104  TCMI_MSG_TYPE(resp_msg_id), timeout))) {
105  mdbg(ERR3, "Can't create a new transaction for message(ID=%x)", msg_id);
106  error = -1;
107  }
108  /* when a transaction was requested, set proper
109  * request ID, so that the receiving is able to reply
110  */
111  else
112  self->trans_id.req = tcmi_transaction_id(self->transaction);
113  }
114  return error;
115 }

118 /**
119  * <<public>> Sends a message via a specified connection. Sending
120  * an anonymous message means, that if there is a transaction
121  * associated with the message, the response message will be delivered
122  * into regular message queue. This is typically used when the sending
123  * thread doesn't want to process the response message itself, but
124  * still wants the response to be checked against the transaction.
125  *
126  * The actual work is delegated to tcmi_msg_send() specifying the
127  * anonymous flag. Then the transaction instance is released as the
128  * sending thread doesn't needed it. The status of the send operation is
129  * communicated to the user.
130  *
131  * This method is also to be used for sending one-way request messages.
132  *
133  * @param *self - pointer to this message instance
134  * @param *sock - KKC socket used for sending message data
135  * be set in the transaction.
136  * @return 0 upon success
137  */
138 int tcmi_msg_send_anonymous(struct tcmi_msg *self, struct kkc_sock *sock)
139 {
140  int err;
141  err = tcmi_msg_send(self, sock, TCMI_TRANS_FLAGS_ANONYMOUS);
142  return err;
143 }

146 /**
147  * <<public>> Sends a message via a specified connection and waits
148  * for the response. This method is used when the message being sent
149  * out has a transaction and a response is expected. The calling
150  * thread blocks until a valid response arrives or the transaction
151  * expires.
152  *
153  * The actual work is delegated to tcmi_msg_send() specifying no extra
154  * flags for the transaction. The thread is then put to sleep waiting
155  * for completion of the transaction. When its woken up again, the transaction
156  * has:
157  * - been aborted - the user gets NULL in the response message parameter
158  * OR
159  * - been completed - the response is extracted from the transaction context
160  * OR
161  * - a signal has arrived, in that case the transaction will be aborted,
162  * but there is still a chance that the response has arrived in the mean time
163  * and completed the transaction. In that rare case, the abort fails
164  * and returns a valid transaction context, which is our response message.
165  * The transaction stays marked as complete.
166  *
167  * The transaction is eventually released.
168  *
169  * The case when there is no transaction associated with the message
170  * is handled correctly - method terminates right after sending out the message.
171  *
172  * @param *self - pointer to this message instance
173  * @param *sock - KKC socket used for sending message data
174  * @param **resp - contains response message or NULL if the transaction
175  * is aborted
176  * @return 0 upon success
177  */
178 int tcmi_msg_send_and_receive(struct tcmi_msg *self, struct kkc_sock *sock,
179  struct tcmi_msg **resp)
180 {
181  int err = 0;
182  int ret;
183  /* increment the reference counter, does nothing when NULL */
184  struct tcmi_transaction *trans = self->transaction;
185  *resp = NULL;
186 
187  /* try sending the data first, quit on error or no transaction */
188  err = tcmi_msg_send(self, sock, 0);
189  if (!self->transaction || err < 0)
190  goto exit0;
191 
192  /* wait till the transaction expires */
193  ret = tcmi_transaction_wait_interruptible(trans);
194  switch (ret) {
195  case TCMI_TRANSACTION_ABORTED:
196  mdbg(INFO3, "Transaction has been aborted, no response arrived");
197  break;
198 
199  case TCMI_TRANSACTION_COMPLETE:
200  mdbg(INFO3, "Transaction complete, picking up the response");
201  *resp = TCMI_MSG(tcmi_transaction_context(trans));
202  break;
203  /* Aborting a transaction upon a signal arrival must
204  * be atomic, it will fail if the transaction has been
205  * completed in the meantime and thus the method returns a valid
206  * transaction context and transaction stays in 'COMPLETE' state */
207  case -ERESTARTSYS:
208  default:
209  mdbg(INFO3, "Signal arrived, result %d. Check if we abort the transaction..", ret);
210  *resp = TCMI_MSG(tcmi_transaction_abort(trans));
211  /* communicate the -ERESTARTSYS to the caller */
212  err = ret;
213  break;
214  }
215 
216  exit0:
217  return err;
218 }

221 /**
222  * <<public>> Receives a message via a specified connection.
223  * - reads the transaction ID from the connection
224  * - calls the receive method of the child class
225  *
226  * The transaction ID is temporarily stored in the instance and will
227  * be used upon delivery when searching for the corresponding
228  * transaction.
229  *
230  * @param *self - pointer to this message instance
231  * @param *sock - KKC socket used for receiving message data
232  * @return >=0 upon success.Unlike tcmi_msg_send() we don't convert a
233  * >=0 result to 0 as this will never get communicated to the user.
234  * The \link tcmi_comm.c::tcmi_comm_thread receiving thread \endlink
235  * is the only component that should use this method.
236  */
237 int tcmi_msg_recv(struct tcmi_msg *self, struct kkc_sock *sock)
238 
239 {
240  int err = 0;
241  struct tcmi_msg_ops *ops = self->msg_ops;
242  if ((err = kkc_sock_rcv_lock_interruptible(sock))) {
243  mdbg(ERR3, "Socket lock - interrupted by a signal %d", err);
244  goto exit0;
245  }
246  /* Receive the transaction ID part */
247  if ((err = kkc_sock_recv(sock, &self->trans_id,
248  sizeof(self->trans_id), KKC_SOCK_BLOCK)) < 0) {
249  mdbg(ERR3, "Failed to receive transaction ID's");
250  goto exit1;
251  }
252 
253  mdbg(INFO3, "Receiving message(ID=%x req=%x resp=%x)",
254  self->msg_id, self->trans_id.req, self->trans_id.resp);
255  /* */
256  if (ops && ops->recv)
257  err = ops->recv(self, sock);
258  kkc_sock_rcv_unlock(sock);
259  return err;
260 
261  /* error handling */
262  exit1:
263  kkc_sock_rcv_unlock(sock);
264  exit0:
265  return err;
266 }

268 /**
269  * <<public>> A message always delivers itself. A delivery of a
270  * message depends on the message type. If it is the request part of
271  * the message or a one-way message, then the receiving party/thread
272  * wants to have it in the associated message queue. If it is a
273  * response to a previously sent message and the receiving
274  * party/thread expects it, it will get associated with its
275  * transaction.
276  *
277  * - a message is of request type if it has a request transaction
278  * ID. In that case, it will be appended to the regular request queue
279  * and the queue processing thread will pick it up.
280  * - a message is of response type - has a response transaction ID. In
281  * that case, it will find its transaction. On success, it will
282  * associate itself with the transaction and complete it. Completing
283  * the transaction is handled by \link
284  * tcmi_transaction.c::tcmi_transaction_complete() the transaction
285  * class \endlink internally.
286  * - in addition, if the transaction is marked anonymous (nobody is
287  * synchronously waiting for the response), the message will also
288  * be delivered into the regular message queue.
289  *
290  * @param *self - pointer to this message instance
291  * @param *requests - queue where the message is stored if it is not a
292  * response to other message.
293  * @param *transactions - contains all active transactions, the message
294  * tries to find its matching transaction.
295  * @return 0 upon success
296  */
297 int tcmi_msg_deliver(struct tcmi_msg *self, struct tcmi_queue *requests,
298  struct tcmi_slotvec *transactions)
299 {
300  int error = 0;
301  /* storage for the potential transaction associated with the message */
302  struct tcmi_transaction *tran;
303  /* request message handling */
304  if (self->trans_id.resp == TCMI_TRANSACTION_INVAL_ID) {
305  mdbg(INFO3, "Adding message(ID=%x req=%x resp=%x) to the request queue",
306  self->msg_id, self->trans_id.req, self->trans_id.resp);
307  tcmi_queue_add(requests, &self->node);
308  }
309  /* response message handling */
310  else {
311  /* search for the transaction */
312  if (!(tran = tcmi_transaction_lookup(transactions, self->trans_id.resp))) {
313  mdbg(ERR3, "Message(ID=%x req=%x resp=%x) - "
314  "can't find response transaction ID among transactions",
315  self->msg_id, self->trans_id.req, self->trans_id.resp);
316  error = -1;
317  goto exit0;
318  }
319  /* store the transaction in the message, reference
320  * counter has been increased by the successful
321  * lookup */
322  self->transaction = tran;
323  /* try completing it - only message type from the
324  * message ID is taken, to verify that the message
325  * being delivered matches the message expected by the
326  * transaction */
327  if (tcmi_transaction_complete(tran, self, TCMI_MSG_TYPE(self->msg_id))) {
328  mdbg(ERR3, "Error completing transaction %x",
329  tcmi_transaction_id(tran));
330  error = -EINVAL;
331  goto exit0;
332  }
333  if (tcmi_transaction_is_anon(tran)) {
334  mdbg(INFO3, "Detected anonymous transaction, queueing the message too..");
335  tcmi_queue_add(requests, &self->node);
336  }
337  mdbg(INFO3, "Message (ID=%x req=%x resp=%x) delivered, transaction completed",
338  self->msg_id, self->trans_id.req, self->trans_id.resp);
339  }
340 
341  /* error handling */
342  exit0:
343  return error;
344 }

346 /**
347  * <<public>> Class method - creates a message based on a
348  * descriptor. Verifies, that the message type in the message ID
349  * matches the one in the selected descriptor. Then, based on message
350  * ID flags calles the regular rx constructor or error rx constructor.
351  * This method is meant to be called by the message factory.
352  *
353  * @param msg_id - this message ID is for verification
354  * @param *dsc - descriptor of a message that is to be built
355  * @return new message instance or NULL
356  */
357 struct tcmi_msg* tcmi_msg_new_rx(u_int32_t msg_id, struct tcmi_msg_dsc *dsc)
358 {
359  struct tcmi_msg *msg = NULL;
360 
361  if (TCMI_MSG_TYPE(msg_id) != TCMI_MSG_TYPE(dsc->msg_id)) {
362  mdbg(ERR3, "Requested message ID type(%x) doesn't match the descriptor(%x)",
363  TCMI_MSG_TYPE(msg_id), TCMI_MSG_TYPE(dsc->msg_id));
364  goto exit0;
365  }
366  /* based on message flags call the regular or error rx constructor */
367  switch (TCMI_MSG_FLG(msg_id)) {
368  case TCMI_MSG_REGFLGS:
369  if (!dsc->new_rx) {
370  mdbg(ERR3, "Regular rx message constructor not present, bailing out..");
371  goto exit0;
372  }
373  msg = dsc->new_rx(msg_id);
374  break;
375  case TCMI_MSG_ERRFLGS:
376  if (!dsc->new_rx_err) {
377  mdbg(ERR3, "Error rx message constructor not present, bailing out..");
378  goto exit0;
379  }
380  msg = dsc->new_rx_err(msg_id);
381  break;
382  default:
383  mdbg(ERR3, "Unexpected message flags value in message ID %x", msg_id);
384  }
385  return msg;
386  /* error handling */
387  exit0:
388  return NULL;
389 }

396 /**
397  * <<private>> Sends a message via a specified connection.
398  * Sending a message requires following steps:
399  * - activate the transaction(if any)
400  * - send message ID,
401  * - send transaction ID(or TCMI_TRANSACTION_INVAL_ID for oneway messages)
402  * - call a specific message method
403  *
404  * It is necessary to lock the socket, so that the message won't get
405  * intermixed with some other.
406  *
407  * @param *self - pointer to this message instance
408  * @param *sock - KKC socket used for sending message data
409  * @param flags - if the message has a transaction, these are additional flags to
410  * be set in the transaction.
411  * @return 0 upon success
412  */
413 static int tcmi_msg_send(struct tcmi_msg *self, struct kkc_sock *sock, int flags)
414 {
415  int err = 0;
416  struct tcmi_msg_ops *ops = self->msg_ops;
417  struct tcmi_transaction *trans = self->transaction;
418 
419  /** try to start the transaction */
420  if (trans && (err = tcmi_transaction_start(trans, flags))) {
421  mdbg(ERR3, "Failed to start the transaction %d", err);
422  goto exit0;
423  }
424 
425  /* Send message ID*/
426  mdbg(INFO3, "Sending message(Msg=%p ID=%x req=%x resp=%x), transaction started: %d",
427  self, self->msg_id, self->trans_id.req, self->trans_id.resp, (trans!=NULL));
428  if ((err = kkc_sock_snd_lock_interruptible(sock))) {
429  mdbg(ERR3, "Socket lock - interrupted by a signal %d", err);
430  goto exit0;
431  }
432  if ((err = kkc_sock_send(sock, &self->msg_id,
433  sizeof(self->msg_id), KKC_SOCK_BLOCK)) < 0) {
434  mdbg(ERR3, "Failed to send message ID: %d. Err: %d", self->msg_id, err);
435  goto exit1;
436  }
437  /* Send transaction ID's */
438  if ((err = kkc_sock_send(sock, &self->trans_id,
439  sizeof(self->trans_id), KKC_SOCK_BLOCK)) < 0) {
440  mdbg(ERR3, "Failed to send transaction ID's");
441  goto exit1;
442  }
443  /* Send the actual message content */
444  if (ops && ops->send) {
445  err = ops->send(self, sock);
446  }
447 
448  /* convert the result to 0 if no error has occured this is
449  * because the number of bytes sent has no meaning for us anymore */
450  err = min(err, 0);
451  kkc_sock_snd_unlock(sock);
452 
453  return err;
454  /* error handling */
455  exit1:
456  kkc_sock_snd_unlock(sock);
457  exit0:
458  return err;
459 }
