40 /** <<public>> Get file status
41  * @param self - pointer to this file instance
42  * @param status - status we asking for
43  *
44  * @return true if status is set for file
45  * */
46 unsigned proxyfs_file_get_status(struct proxyfs_file_t* self, unsigned status)
47 {
48  unsigned rtn = 0;
49  if( status & PROXYFS_FILE_RO_STATUS ){
50  if( status & PROXYFS_FILE_CAN_WRITE ){
51  if( circ_buf_get_write_size( & self->write_buf ) )
52  rtn |= PROXYFS_FILE_CAN_WRITE;
53  }
54  if( status & PROXYFS_FILE_CAN_READ ){
55  if( circ_buf_count( & self->read_buf ) )
56  rtn |= PROXYFS_FILE_CAN_READ;
57  }
58  if( status & PROXYFS_FILE_ALL_WRITTEN ){
59  mdbg(INFO3, "file_get_status[%lu] all_written.. %d",proxyfs_file_get_file_ident(PROXYFS_FILE(self)),circ_buf_get_read_size(&self->write_buf));
60  if( circ_buf_get_read_size( & self->write_buf ) == 0 )
61  rtn |= PROXYFS_FILE_ALL_WRITTEN;
62  }
63  if( status & PROXYFS_FILE_ALL_READ ){
64  mdbg(INFO3, "file_get_status[%lu] all_read.. %d",proxyfs_file_get_file_ident(PROXYFS_FILE(self)),circ_buf_get_read_size( & self->read_buf ));
65  if( circ_buf_get_read_size( & self->read_buf ) == 0 )
66  rtn |= PROXYFS_FILE_ALL_READ;
67  }
68  }
69  if( status & PROXYFS_FILE_RW_STATUS ){
70  rtn |= self->state & (status & PROXYFS_FILE_RW_STATUS);
71  }
72  return rtn;
73 }

131 /** <<public>> Initialize proxyfs_file struct
132  * @param self - pointer to this file instance
133  *
134  * @return zero on success
135  * */
136 int proxyfs_file_init(struct proxyfs_file_t* self)
137 {
138  if( circ_buf_init( & self->write_buf, WRITE_BUF_SIZE ) != 0 ){
139  mdbg(ERR3, "Allocating write buffer for proxy_file failed");
140  goto exit0;
141  }
142  if( circ_buf_init( & self->read_buf, READ_BUF_SIZE ) != 0 ){
143  mdbg(ERR3, "Allocating read buffer for proxy_file failed");
144  goto exit1;
145  }
146 
147  self->state = 0;
148  self->write_buf_unconfirmed = 0;
149  self->read_was_requested = 0;
150  self->peer = NULL;
151  return 0;
152 exit1:
153  circ_buf_destroy( & self->write_buf );
154 exit0:
155  return -1;
156 }
