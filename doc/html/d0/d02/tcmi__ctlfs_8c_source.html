52 /**
53  * <<public>> Root entry accessor. If the filesystem has not been
54  * activated yet, a kernel mount is performed. The kernel mount in
55  * turn ensures instantiation of the root directory. A user might have
56  * already mounted the filesystem from userspace. This is not a
57  * problem as we allow only one instance of the filesystem in
58  * memory(enforced by get_sb_single()), so only a reference counter of
59  * the root directory and the associated superblock is
60  * incremented. The last thread that releases the last reference to
61  * the super block and eventually to the root directory causes the
62  * whole filesystem to be released. See tcmi_ctlfs_kill_super() for
63  * details on this.
64  *
65  * If the filesystem is already active (the vfs_mnt is valid), the
66  * mount reference count is adjusted.
67  *
68  * It is necessary to protect the vfs_mnt variable as it can be
69  * manipulated by other threads that might issue tcmi_ctlfs_put_root()
70  * and release the vfs mount from under us.
71  *
72  * @return pointer to the root entry (reference counter adjusted by
73  * tcmi_ctlfs_entry_get()) or NULL
74  */
75 struct tcmi_ctlfs_entry* tcmi_ctlfs_get_root(void)
76 {
77  down(&self.vfs_mnt_sem);
78  /* when the VFS mount is currently inactive we have to create
79  * a new one */
80  if (!self.vfs_mnt) {
81  mdbg(INFO3, "Mounting");
82  self.vfs_mnt = kern_mount(&self.fs_type);
83  if (IS_ERR(self.vfs_mnt)) {
84  minfo(ERR1, "Could not mount!");
85  self.vfs_mnt = NULL;
86  /* if the mount failed, the root directory
87  * instance doesn't exist too, so we don't
88  * need to explicitely delete it*/
89  self.root_dir = NULL;
90  }
91  }
92  /* VFS mount still active, just increment its ref. counter */
93  else
94  mntget(self.vfs_mnt);
95  up(&self.vfs_mnt_sem);
96 
97  mdbg(INFO3, "sb c_count=%d, s_active=%d", self.sb->s_count,
98  atomic_read(&self.sb->s_active));
99  mdbg(INFO3, "mount mnt_count=%d", atomic_read(&self.vfs_mnt->mnt_count));
100  return tcmi_ctlfs_entry_get(self.root_dir);
101 }

103 /**
104  * <<public>> Releases the root directory and then the mount
105  * object. VFS takes care of releasing the object in case their
106  * reference count reaches 0. We need to take care of setting the
107  * vfs_mnt to NULL.
108  *
109  * The last 2 steps don't need to be semaphore protected as we are
110  * working on copies of both pointers and we wanted to protect the
111  * access to vfs_mnt only. If anybody tries to get a root from now on
112  * a new VFS mount will be created. If the root directory has been
113  * also destroyed, the read_super method takes care of instantiating a
114  * new super block along with the root directory.
115  */
116 void tcmi_ctlfs_put_root(void)
117 {
118  struct vfsmount *cur_vfs_mnt;
119  struct tcmi_ctlfs_entry *cur_root_dir;
120  mdbg(INFO3, "sb c_count=%d, s_active=%d", self.sb->s_count,
121  atomic_read(&self.sb->s_active));
122  mdbg(INFO3, "mount mnt_count=%d", atomic_read(&self.vfs_mnt->mnt_count));
123 
124  /* mdbg(INFO3, "root d_count=%d", atomic_read(&self.root_dir->super.dentry->d_count));*/
125 
126  down(&self.vfs_mnt_sem);
127  cur_vfs_mnt = self.vfs_mnt;
128  cur_root_dir = self.root_dir;
129  /* If we are the last thread accesing the kernel mount */
130  if (atomic_read(&self.vfs_mnt->mnt_count) == 1)
131  self.vfs_mnt = NULL;
132 
133  up(&self.vfs_mnt_sem);
134 
135  /* no locking needed since working on copies of both
136  * pointers */
137  tcmi_ctlfs_entry_put(cur_root_dir);
138  mntput(cur_vfs_mnt);
139 }

181 static int init_inodecache(void)
182 {
183  ctlfs_inode_cachep = kmem_cache_create("ctlfs_inode_cache",
184  sizeof(struct inode),
185  0, (SLAB_RECLAIM_ACCOUNT|
186  SLAB_MEM_SPREAD),
187  init_once);
188  if (ctlfs_inode_cachep == NULL)
189  return -ENOMEM;
190 
191  return 0;
192 }

206 /**
207  * <<private>> Sets up super block and instantiates the root
208  * directory. This method is called only once through out the super
209  * block life.
210  *
211  * @param sb - pointer to the super block
212  * @param *data - custom data supplied when reading the superblock(not used)
213  * @param silent - required by VFS, no idea what it means.
214  *
215  * @return 0 upon success
216  */
217 static int tcmi_ctlfs_fill_super(struct super_block *sb, void *data, int silent)
218 {
219  sb->s_blocksize = PAGE_CACHE_SIZE;
220  sb->s_blocksize_bits = PAGE_CACHE_SHIFT;
221  sb->s_magic = TCMI_CTLFS_MAGIC;
222  sb->s_op = &s_ops;
223  self.sb = sb;
224 mdbg(INFO4, "TMPDBG: GOING TO REQUEST ROOT");
225  self.root_dir = tcmi_ctlfs_rootdir_new(sb);
226 mdbg(INFO4, "TMPDBG: ROOT SUCCEEDED");
227  return 0;
228 }

230 /**
231  * <<private>> Super block as assigned a nonexistent device and
232  * the super block is obtained via get_sb_single(). The kernel
233  * function get_sb_single() ensures that only 1 instance of the
234  * filesystem can be mounted at a time. Custom method to fill the
235  * super block is passed as parameter.
236  *
237  * @param *fst - points to the file system type, that is being mounted
238  * @param flags -
239  * @param *devname - device name, on which the file system resides (it has no meaning here)
240  * @param *data - pointer to private filesystem data - not used
241  *
242  * @return - pointer to the super block object that has been created
243  */
244 static int tcmi_ctlfs_get_super(struct file_system_type *fst,
245  int flags, const char *devname, void *data, struct vfsmount *mnt)
246 {
247  return get_sb_single(fst, flags, data, tcmi_ctlfs_fill_super, mnt);
248 }
