36 /** @defgroup kkc_arch_class kkc_arch class
37  *
38  * @ingroup kkc_class
39  *
40  * This class describes a particular network architecture. Each
41  * architecture has its unique name. Based on this name, the main KKC
42  * module will lookup a particular architecture to create a user
43  * specified object.
44  *
45  * The instances of kkc_arch are to be created at compile time by the
46  * corresponding architecture socket class(e.g. \link
47  * kkc_sock_tcp_class kkc_sock_tcp \endlink). The socket class then
48  * provides architecture specific operations. Currently, there is only
49  * one operation needed - the socket constructor.
50  *
51  * Also each particular architecture instance retains a reference to a
52  * module where it has been declared. Each object created via the
53  * architecture owns a reference to the architecture. That way, the
54  * module can't get unloaded when instance of its sockets still exist.
55  *
56  * @{
57  */
58 /** Maximum architecture string length. */
59 #define KKC_MAX_ARCH_LENGTH 20
60

72 struct kkc_arch {

145 /**
146  * <<public>> Retrieves an architecture by name. This requires
147  * merging the architecture name with the architecture prefix and
148  * asking the kernel to retrieve the symbol specified by this string.
149  *
150  * This is a class method
151  * @param arch_name - name of the architecture requested (e.g. 'tcp')
152  * @return - instance of the kkc_arch that handles the specified type
153  */
154 static inline struct kkc_arch* kkc_arch_get_by_name(char *arch_name)
155 {
156 
157  char sym[sizeof(KKC_ARCH_PREFIX_STR) + KKC_MAX_ARCH_LENGTH];
158  strlcpy(sym, KKC_ARCH_PREFIX_STR, sizeof(KKC_ARCH_PREFIX_STR) + KKC_MAX_ARCH_LENGTH);
159  strlcat(sym, arch_name, sizeof(KKC_ARCH_PREFIX_STR) + KKC_MAX_ARCH_LENGTH);
160  mdbg(INFO3, "Getting architecture '%s', prefix: '%s'", sym, MODULE_SYMBOL_PREFIX);
161 
162  return (struct kkc_arch*) __symbol_get((const char*)sym);
163 }
