145 /**
146  * <<public>> This method unregisters the file uses the semaphore
147  * lock as protection from races. This is the prefered way of
148  * detaching the file from an object. If the file is to unregister
149  * itself (e.g. from inside the registered object's methods), the
150  * previous version is recommended to prevent deadlocks.
151  *
152  * @param *self - pointer to this file instance
153  */
154 void tcmi_ctlfs_file_unregister(struct tcmi_ctlfs_entry *self)
155 {
156  struct tcmi_ctlfs_file *self_f = TCMI_CTLFS_FILE(self);
157  if (self_f) {
158  tcmi_ctlfs_file_lock(self_f);
159  __tcmi_ctlfs_file_unregister(self);
160  tcmi_ctlfs_file_unlock(self_f);
161 
162  }

167  * <<public>> This method unregisters the file and allows the user
168  * to specify the preferred way of unregistration (w/ or w/o the
169  * lock). It is typically used if owning object uses this file is for
170  * unregistering other files including this file. All work is
171  * delegate to either __tcmi_ctlfs_file_unregister() or
172  * tcmi_ctlfs_file_unregister()
173  *
174  * @param *self - pointer to this file instance
175  * @param unreg - describes a unregistration type, if a regular shutdown

182 {
183  /* unregistration issued from owning object's method, no locking */
184  if (unreg == TCMI_CTLFS_FILE_FROM_METHOD)
185  __tcmi_ctlfs_file_unregister(self);

187  else
188  tcmi_ctlfs_file_unregister(self);

196 /**
197  * <<private>> This is a generic method, that creates a new
198  * instance of a file as follows:
199  *
200  * - allocates a new instance and delegates the entry initialization to
201  * the parent class. There is a custom tcmi_ctlfs_ops vector as file
202  * destruction needs special handling. There are also file and inode
203  * specific operations.
204  * - allocates file associated data
205  * - registers the object and its 2 methods with the file. None of
206  * the methods are mandatory
207  * - there is a custom entry_free operation that releases the
208  * above allocated file data upon file destruction
209  * - proc_handler is associated with the data, so that ascii data
210  * can be parsed when passing from/to the object's read/write method
211  *
212  *
213  * @param *parent - pointer to the parent entry - has to be a directory
214  * @param *mode - access rights for the inode
215  * @param *object - points to the object that creates the file
216  * @param *read_method - method of the file creator that will be called upon file read
217  * @param *write_method - method of the file creator that will be called upon file write
218  * @param maxlen - maximum length of the data in bytes
219  * @param *proc_handler - data parsing handler that does the conversion from/to ascii
220  * @param namefmt - nameformat string (printf style)
221  * @param args - arguments to the format string
222  * @return pointer to the new file or NULL
223  */
224 static struct tcmi_ctlfs_entry* tcmi_ctlfs_genericfile_new(struct tcmi_ctlfs_entry *parent,
225  mode_t mode,
226  void *object,
227  object_method *read_method,
228  object_method *write_method,
229  int maxlen,
230  proc_handler *proc_handler,
231  const char namefmt[], va_list args)
232 {
233  struct tcmi_ctlfs_file *file;
234  mdbg(INFO4, "Creating new file");
235 
236  if (!(file = kmalloc(sizeof(struct tcmi_ctlfs_file), GFP_ATOMIC))) {
237  mdbg(ERR3, "Failed to allocate memory for file");
238  goto exit0;
239  }
240  if (tcmi_ctlfs_entry_init(TCMI_CTLFS_ENTRY(file),
241  TCMI_CTLFS_ENTRY(parent),
242  S_IFREG | mode,
243  &tcmi_ctlfs_file_ops,
244  &tcmi_ctlfs_file_inode_operations,
245  &tcmi_ctlfs_file_operations, namefmt, args)) {
246  mdbg(ERR3, "Failed to create the file");
247  goto exit1;
248  }
249  if (!(file->data = kmalloc(maxlen, GFP_ATOMIC))) {
250  mdbg(ERR3, "Failed to allocate %d bytes of file data", maxlen);
251  goto exit2;
252  }
253  file->maxlen = maxlen;
254  init_MUTEX(&file->f_sem);
255  /* object associated with the file */
256  file->object = object;
257  file->read_method = read_method; /*TESTING */
258  file->write_method = write_method; /*TESTING */
259  file->proc_handler = proc_handler; /* TESTING */
260  /* indicates a valid object has registered with the file */
261  file->unregistered = 0;
262  return TCMI_CTLFS_ENTRY(file);
263 
264  /* error handling */
265  exit2:
266  kfree(file->data);
267  exit1:
268  kfree(file);
269  exit0:
270  return NULL;
271 }

295 /**
296  * <<private>> The read file operation, called by the read system call.
297  * The access to the file associated data is serialized, so that the
298  * object method always receives consistent data.
299  * - Read requests are possible only from offset (ppos) 0.
300  * - Initial check whether an object is already unregistered with the file is
301  * performed. This is to prevent calling the method when the the object is
302  * currently detaching from the file. The associated object might have already
303  * taken steps that don't allow calling the read method - E.g. has already
304  * woken up the sleeping process and doesn't want another process to call
305  * the method again.
306  * - The registered object is first sent a read request and data
307  * is returned
308  * - The data is then processed by the proc_handler. It is necessary
309  * to setup the ctl_table, so that the we can reuse the sysctl.c
310  * data parsing code for integers and strings.
311  *
312  * @param *filp - pointer to the kernel file object that is being read
313  * @param *buf - user buffer where to store the result
314  * @param count - number of bytes to be read
315  * @param *ppos - pointer to the current position in the file
316  * @return - when >= 0 - number of bytes written, the result from the proc_handler
317  * is usually 0, just to denote success converting from binary to ascii
318  * when < 0 - error
319  */
320 static ssize_t tcmi_ctlfs_file_read(struct file *filp, char __user *buf,
321  size_t count, loff_t *ppos)
322 {
323  struct tcmi_ctlfs_file *file =
324  TCMI_CTLFS_FILE(TCMI_CTLFS_DENTRY_TO_ENTRY(filp->f_dentry));
325  size_t res;
326  ssize_t error = 0;
327 
328  struct ctl_table tbl = {
329  .data = file->data,
330  .maxlen = file->maxlen,
331  };
332  mdbg(INFO4, "file read in progress count=%lu, ppos=%Ld", (unsigned long)count, *ppos);
333  if (tcmi_ctlfs_file_lock_interruptible(file)) {
334  error = -ERESTARTSYS;
335  goto exit0;
336  }
337  /* when object unregistering is in progress don't start any operation */
338  if (file->unregistered)
339  goto exit1;
340  /* when trying from other position than 0, reset the file position and quit */
341  if (*ppos) {
342  *ppos = 0;
343  goto exit1;
344  }
345  if (file->read_method)
346  if ((error = file->read_method(file->object, file->data)))
347  goto exit1;
348  if (file->proc_handler) {
349  /* res is in/out parameter, initially stores the number of requested bytes.
350  The handler then returns the number of bytes really processed */
351  res = count;
352  if ((error = file->proc_handler(&tbl, 0, buf, &res, ppos)))
353  goto exit1;
354  error = (ssize_t)res;
355  }
356 
357  exit1:
358  tcmi_ctlfs_file_unlock(file);
359  /* signal arrived, semaphore still busy */
360  exit0:
361  return error;
362 }

364 /**
365  * <<private>> The write file operation, called by the write system call.
366  * The access to the file associated data is serialized, so that the
367  * object method always receives consistent data.
368  * - Initial check whether an object is already unregistered with the file is
369  * performed. This is to prevent calling the method when the the object is
370  * currently detaching from the file. The associated object might have already
371  * taken steps that don't allow calling the write method - E.g. has already
372  * woken up the sleeping process and doesn't want another process to call
373  * the method again.
374  * - The requested file offset can be arbitrary, but must not exceed
375  * the size of the data assoicated with the file - handled by the
376  * proc_handler
377  * - The data is first processed by the proc_handler. It is necessary
378  * to setup the ctl_table, so that the we can reuse the sysctl.c
379  * data parsing code for integers and strings.
380  * - Finally, the registered object is notified about the write event
381  * and the data is passed to it.
382  *
383  * @param *filp - pointer to the kernel file object that is being written
384  * @param *buf - user buffer where to take the source data
385  * @param count - number of bytes to be written
386  * @param *ppos - pointer to the current position in the file
387  * @return - when >= 0 - number of bytes written, the result from the proc_handler
388  * is usually 0, just to denote success converting the ascii data to binary
389  * when < 0 - error
390  */
391 static ssize_t tcmi_ctlfs_file_write(struct file *filp, const char __user *buf,
392  size_t count, loff_t *ppos)
393 {
394  struct tcmi_ctlfs_file *file =
395  TCMI_CTLFS_FILE(TCMI_CTLFS_DENTRY_TO_ENTRY(filp->f_dentry));
396  size_t res;
397  ssize_t error = 0;
398 
399  struct ctl_table tbl = {
400  .data = file->data,
401  .maxlen = file->maxlen,
402  };
403  mdbg(INFO4, "file write in progress count=%lu, ppos=%Ld", (unsigned long)count, *ppos);
404  if (tcmi_ctlfs_file_lock_interruptible(file)) {
405  error = -ERESTARTSYS;
406  goto exit0;
407  }
408  /* when object unregistering is in progress don't start any operation */
409  if (file->unregistered)
410  goto exit1;
411  if (file->proc_handler) {
412  res = count;
413  if ((error = file->proc_handler(&tbl, 1, (char __user *)buf, &res, ppos)))
414  goto exit1;
415  }
416  if (file->write_method) {
417  if ((error = file->write_method(file->object, file->data)))
418  goto exit1;
419  error = (ssize_t)res;
420  }
421  exit1:
422  tcmi_ctlfs_file_unlock(file);
423  /* signal arrived, semaphore still busy */
424  exit0:
425  return error;
426 }

428 /**
429  * This method is used when object associated with the file wants to
430  * unregister with the file. The calling object is responsible for waking
431  * up a user process currently using this file (e.g. that have it open
432  * and might be sleeping in the read/write methods of the object).
433  *
434  * This method assumes, that the file lock is already held by the
435  * caller. A typical use can be when the file is supposed to destroy
436  * itself upon user reading from or writing into the file (from inside
437  * the registered object's methods).
438  *
439  * The unregistration requires resetting read/write methods and
440  * associated proc_handler and object pointer. The unregistered flag
441  * ensures that the methods will be ignored on the next read/write issued
442  * by any user process.
443  *
444  * @param *self - pointer to this file instance
445  */
446 static void __tcmi_ctlfs_file_unregister(struct tcmi_ctlfs_entry *self)
447 {
448  struct tcmi_ctlfs_file *self_f = TCMI_CTLFS_FILE(self);
449  if (self_f) {
450  /* in case the object hasn't set the unregistered flag in advance */
451  tcmi_ctlfs_file_unregister_prepare(self);
452  self_f->object = NULL;
453  self_f->read_method = NULL;
454  self_f->write_method = NULL;
455  self_f->proc_handler = NULL;
456  }
457 }
