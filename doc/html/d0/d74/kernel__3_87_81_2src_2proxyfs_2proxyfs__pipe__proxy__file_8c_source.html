39 /** <<private>> Write to file from user space buffer
40  * @param *self - pointer to this file instance
41  * @param *buf - pointer to data in userspace
42  * @param count - size of data
43  *
44  * @return the number of bytes written or error (negative number)
45  * */
46 static int proxyfs_pipe_proxy_file_write_user(struct proxyfs_proxy_file_t *self, const char *buf, size_t count)
47 {
48  int lenght = 0;
49  uint32_t write_size;
50 
51  if( ! access_ok(VERIFY_WRITE, buf, count) )
52  return -EFAULT;
53 
54  down( & self->write_buf_sem );
55  write_size = circ_buf_free_space( & PROXYFS_FILE(self)->write_buf );
56 
57  mdbg(INFO2, "[File %lu] Space in buf: %lu, writting %lu", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), (unsigned long)write_size, (unsigned long)count);
58 
59  if( write_size < 2*sizeof(write_size) ) // Won't write less then 4 bytes
60  lenght = -(2*(int)sizeof(write_size));
61  else if( count <= PIPE_BUF && write_size < sizeof(write_size) + count )
62  lenght = -(sizeof(write_size) + count); // Not enough space
63  else
64  {
65  write_size -= sizeof( write_size ); // Never overflow
66  write_size = write_size < count ? write_size : count;
67  circ_buf_write_user( & PROXYFS_FILE(self)->write_buf, (void*)&write_size, sizeof(write_size) );
68  lenght = circ_buf_write_user( & PROXYFS_FILE(self)->write_buf, buf, write_size );
69  }
70  up( & self->write_buf_sem );
71 
72  mdbg(INFO2, "written %d", lenght);
73 
74  return lenght;
75 }
