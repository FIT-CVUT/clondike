130 void proxyfs_server_loop(void){
131  while(!kthread_should_stop()){
132  proxyfs_task_wait_for_data_ready(PROXYFS_TASK(self));
133 
134  // Accept all incoming conections
135  while( proxyfs_server_accept() == 1 );
136  // Try to recv from all socket.
137  proxyfs_task_recv_peers( PROXYFS_TASK(self) );
138  //mdbg(INFO3, "STATE: %lx", current->state);

153  * @param self - pointer to this proxyfs_task instance

156 static void proxyfs_server_handle_msg(struct proxyfs_task *self, struct proxyfs_peer_t *peer )
157 {
158  struct proxyfs_msg *msg = proxyfs_peer_get_msg(peer);
159  struct proxyfs_file_t *real_file;
160  u_int32_t exitting;
161  //long rtn;
162 
163  mdbg(INFO1, "Msg%lu DATA: (%p)",msg->header.file_ident ,msg);
164 
165  switch( msg->header.msg_num ){
166  case MSG_OPEN:
167  mdbg(INFO1, "MSG_OPEN ident=%lu", msg->header.file_ident);
168  real_file = proxyfs_task_find_file( self, msg->header.file_ident);
169  if( real_file != NULL ){
170  mdbg(INFO1, "File found");
171  if( 0 && real_file->peer != NULL )
172  mdbg(INFO1, "Atempt to open already opened file, rejecting...");
173  else{
174  proxyfs_real_file_open( PROXYFS_REAL_FILE(real_file), peer );
175  return;
176  }
177  }
178  mdbg(INFO1, "File not found");
179  msg = proxyfs_msg_new(MSG_OPEN_RESP_FAILED, msg->header.file_ident, 0, NULL );
180  proxyfs_peer_send_msg( peer, msg );
181  break;
182  case MSG_IOCTL:
183  mdbg(INFO1, "MSG_IOCTL");
184  real_file = proxyfs_task_find_file( self, msg->header.file_ident);
185  if( real_file != NULL ){
186  u_int32_t ioctl_info = ioctl_get_info(msg->header.data[0]);
187  char* ioctl_buffer = kmalloc(IOCTL_ARG_SIZE(ioctl_info), GFP_KERNEL);
188  if ( ioctl_buffer ) {
189  u_int32_t result;
190  mdbg(INFO2, "MSG_IOCTL execute command: %d Arg size: %d", msg->header.data[0], IOCTL_ARG_SIZE(ioctl_info));
191  result = proxyfs_real_file_do_ioctl(PROXYFS_REAL_FILE(real_file),
192  msg->header.data[0], (unsigned long)ioctl_buffer);
193 
194 
195  mdbg(INFO2, "MSG_IOCTL performed with result: %d", result);
196  if( IOCTL_WILL_READ(ioctl_info) )
197  msg = proxyfs_msg_compose_new(MSG_IOCTL_RESP, proxyfs_file_get_file_ident(real_file),
198  sizeof(u_int32_t), &result,
199  IOCTL_ARG_SIZE(ioctl_info), ioctl_buffer,
200  (u_int32_t)0);
201  else
202  msg = proxyfs_msg_compose_new(MSG_IOCTL_RESP, proxyfs_file_get_file_ident(real_file),
203  sizeof(u_int32_t), &result,
204  (u_int32_t)0);
205 
206  kfree(ioctl_buffer);
207  if( msg )
208  proxyfs_peer_send_msg( peer, msg );
209  else
210  mdbg(ERR2, "Failed creating response");
211  } else {
212  mdbg(ERR2, "Not enough memory to create ioctl read buffer of size: %d", IOCTL_ARG_SIZE(ioctl_info));
213  }
214  return;
215  }
216  mdbg(INFO1, "File not found");
217  /*msg = proxyfs_msg_new(MSG_OPEN_RESP_FAILED, msg->header.file_ident, 0, NULL );
218  proxyfs_peer_send_msg( peer, msg );*/
219  break;
220  case MSG_CLOSE:
221  exitting = msg->header.data[0];
222  mdbg(INFO1, "MSG_CLOSE ident=%lu was exitting: %d", msg->header.file_ident, exitting);
223  real_file = proxyfs_task_find_file( self, msg->header.file_ident);
224  if( real_file != NULL ){
225  mdbg(INFO1, "File found");
226  proxyfs_real_file_close( PROXYFS_REAL_FILE(real_file), peer );
227  if ( !exitting ) {
228  /* If the process was not exitted, the close was ordinary close request made
229  by the process, so we can mark the file closed, since nobody is going to
230  open it again */
231  mdbg(INFO1, "Marking physical file closed");
232  proxyfs_file_set_status( real_file, PROXYFS_FILE_CLOSED );
233  }
234  return;
235  }
236  msg = proxyfs_msg_new(MSG_CLOSE_RESP, msg->header.file_ident, 0, NULL );
237  proxyfs_peer_send_msg( peer, msg );
238  break;
239  case MSG_FSYNC:
240  mdbg(INFO1, "MSG_FSYNC ident=%lu", msg->header.file_ident);
241  real_file = proxyfs_task_find_file( self, msg->header.file_ident);
242  if( real_file != NULL ){
243  mdbg(INFO1, "File found");
244  proxyfs_real_file_fsync( PROXYFS_REAL_FILE(real_file), peer );
245  return;
246  }
247  msg = proxyfs_msg_new(MSG_FSYNC_RESP, msg->header.file_ident, 0, NULL );
248  proxyfs_peer_send_msg( peer, msg );
249  break;
250  case MSG_READ_REQUEST:
251  mdbg(INFO1, "MSG_READ_REQUEST ident=%lu", msg->header.file_ident);
252  real_file = proxyfs_task_find_file( self, msg->header.file_ident);
253  if( real_file != NULL ){
254  mdbg(INFO1, "File found");
255  proxyfs_real_file_mark_read_request( PROXYFS_REAL_FILE(real_file), peer );
256  return;
257  }
258  msg = proxyfs_msg_new(MSG_READ_REQUEST_RESP, msg->header.file_ident, 0, NULL );
259  proxyfs_peer_send_msg( peer, msg );
260  break;
261  default:
262  mdbg(ERR3, "Unknown msg number");
263  }

271 void proxyfs_server_read_and_write_real_files(void){
272  struct list_head *l, *nxt;
273  struct proxyfs_msg *msg;
274  struct proxyfs_real_file *real_file;
275  struct proxyfs_peer_t *peer;
276  u_int32_t length;
277 
278  down( & self->files_sem ); // Serialize access
279  list_for_each_safe( l, nxt, & PROXYFS_TASK(self)->files ){
280  // up( & self.files_sem ); // Adding new file can wait :-)
281  real_file = PROXYFS_REAL_FILE( list_entry(l, struct proxyfs_file_t, files) );
282  peer = PROXYFS_FILE(real_file)->peer;
283  mdbg(INFO4, "Processing %lu, status %x", proxyfs_file_get_file_ident(PROXYFS_FILE(real_file)),
284  proxyfs_file_get_status( PROXYFS_FILE(real_file), 0xffff ) );
285  // write to file
286  if( (length = proxyfs_real_file_write(real_file)) > 0 ){
287  mdbg(INFO3, "Sending WRITE_RESP msg for file %lu, ack %d",
288  proxyfs_file_get_file_ident(PROXYFS_FILE(real_file)), length );
289 
290  if ( peer ) {
291  msg = proxyfs_msg_compose_new(MSG_WRITE_RESP, proxyfs_file_get_file_ident(PROXYFS_FILE(real_file)),
292  sizeof(u_int32_t), &length, 0 );
293  proxyfs_peer_send_msg( peer, msg );
294  }
295  }
296  // Close file if it should be closed
297  if( proxyfs_file_get_status( PROXYFS_FILE(real_file), PROXYFS_FILE_CLOSED ) &&
298  proxyfs_file_get_status( PROXYFS_FILE(real_file), PROXYFS_FILE_ALL_READ )){
299  struct file* physical_file = real_file->file;
300  mdbg(INFO3, "Closing file %lu. Ref count: %ld", proxyfs_file_get_file_ident(PROXYFS_FILE(real_file)), file_count(real_file->file) );
301  //l = l->next;
302  list_del( & PROXYFS_FILE(real_file)->files ); // Delete from list
303  proxyfs_real_file_destroy( real_file );
304  kfree( real_file );
305  // Close file AFTER we de-list proxy file.. fput may actually destroy the file, if we hold last reference
306  fput( physical_file );
307  continue;
308  }
309  // read from file
310  if ( peer && !proxyfs_file_get_status( PROXYFS_FILE(real_file), PROXYFS_FILE_READ_EOF ) &&
311  proxyfs_real_file_read(real_file) == 0 ){ // EOF
312  msg = proxyfs_msg_compose_new(MSG_WRITE, proxyfs_file_get_file_ident(PROXYFS_FILE(real_file)), 0);
313  proxyfs_peer_send_msg(peer , msg );
314  proxyfs_file_set_status( PROXYFS_FILE(real_file), PROXYFS_FILE_READ_EOF );
315  }
316  // down( & self.files_sem );
317  }
318  up( & self->files_sem );
319 
320 }

440 static void proxyfs_server_duplicate_overtaken_file(struct proxyfs_real_file* file, struct task_struct* new_owner) {
441  struct proxyfs_real_file* clone = NULL;
442 
443  if ( file->ops == NULL || file->ops->duplicate == NULL ) {
444  mdbg(ERR1, "Cannot duplicate file [%lu], it does not have duplicate method defined", proxyfs_file_get_file_ident(PROXYFS_FILE(file)));
445  return;
446  }
447 
448  clone = file->ops->duplicate(file);
449  if ( clone == NULL ) {
450  mdbg(ERR1, "Failed to duplicate file [%lu].", proxyfs_file_get_file_ident(PROXYFS_FILE(file)));
451  return;
452  }
453 
454  PROXYFS_FILE(clone)->file_identifier.file_owner_pid = task_tgid_vnr(new_owner);
455  proxyfs_real_file_add_shadow(clone, new_owner);
456 
457  if ( proxyfs_server_register_poll_callback(clone) )
458  goto exit;
459 
460  list_add_tail( & PROXYFS_FILE(clone)->files, &PROXYFS_TASK(self)->files );
461 
462  return;
463 
464 exit:
465  fput(clone->file);
466  proxyfs_real_file_destroy(clone);
467 }

492  }
493  }
