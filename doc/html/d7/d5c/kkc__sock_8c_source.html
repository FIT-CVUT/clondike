35 /**
36  * <<public>> Initializes generic socket. This method should be
37  * called back from the architecture specific constructor to
38  * initialize common socket fields. Architecture is set.
39  *
40  * @param *self - pointer to this socket instance
41  * @param *sock_ops - operations vector of the architecture specific socket
42  * @param *arch - socket specific architecture. This allows the user
43  * to read architecture name.
44  *
45  * @return 0 upon successful connection
46  */
47 int kkc_sock_init(struct kkc_sock *self, struct kkc_arch *arch,
48  struct kkc_sock_ops *sock_ops)
49 {
50  if (!arch) {
51  mdbg(ERR4, "No architecture specified, can't init socket");
52  return -EINVAL;
53  }
54  self->arch = arch;
55  self->sock_ops = sock_ops;
56  atomic_set(&self->ref_count, 1);
57  init_MUTEX(&self->sock_send_sem);
58  init_MUTEX(&self->sock_recv_sem);
59  INIT_LIST_HEAD(&self->pub_list);
60 
61  mdbg(INFO4, "Initialized KKC socket: %p", self);
62 
63  return 0;
64 }

123 /**
124  * <<public>> Sends out specified data. Sends the entire content
125  * of the specified buffer. This might require multiple iterations -
126  * the method will try until all data is sent or an error occurs.
127  *
128  * @param *self - pointer to this socket instance
129  * @param *buf - buffer with the data to be sent
130  * @param buflen - length of the buffer to be sent
131  * @param *buflen - number of bytes actually sent
132  * @param flags - specify blocking or non-blocking mode.
133  * @return number of bytes actually sent or error when < 0
134  */
135 int kkc_sock_send(struct kkc_sock *self, void *buf, int buflen,
136  kkc_sock_flags_t flags)
137 {
138  int result = 0;
139  mdbg(INFO4, "Sending buffer %p length: %d bytes", buf, buflen);
140  if (self->sock_ops && self->sock_ops->send) {
141  result = kkc_sock_send_recv(self, buf, buflen, flags,
142  self->sock_ops->send);
143  }
144 
145  return result;

149 /**
150  * <<public>> Receives requested number of bytes of data. The
151  * behavior of the function depends on the mode. In blocking mode, it
152  * will try to receive all buflen bytes via kkc_sock_send_recv(). In
153  * non-blocking mode it will perform only 1 attempt to receive data,
154  * using the receive operation of the socket instance directly.
155  *
156  * @param *self - pointer to this socket instance
157  * @param *buf - buffer where to store received data.
158  * @param buflen - length of the buffer to be sent
159  * @param flags - specify blocking or non-blocking mode.
160  * @return number of bytes actually received or error when < 0
161  */
162 int kkc_sock_recv(struct kkc_sock *self, void *buf, int buflen,
163  kkc_sock_flags_t flags)
164 {
165  int result = 0;
166  mdbg(INFO4, "Receiving into buffer %p length: %d bytes", buf, buflen);
167  if (self->sock_ops && self->sock_ops->recv) {
168  if (flags == KKC_SOCK_BLOCK)
169  result = kkc_sock_send_recv(self, buf, buflen, flags,
170  self->sock_ops->recv);
171  else
172  result = self->sock_ops->recv(self, buf, buflen, flags);
173 
174  }
175 
176  return result;
177 }

251 /**
252  * <<private>> Transfers a required number of bytes. This method
253  * is intended for receiving and sending in blocking mode. The caller
254  * specifies only which method should be used. The method tries to
255  * transfer all the requested bytes until an error occurs.
256  *
257  * @param *self - pointer to this socket instance
258  * @param *buf - buffer for transferred data (received or sent)
259  * @param buflen - length of the buffer to be sent/received
260  * @param flags - specify blocking or non-blocking mode.
261  * @param *kkc_sock_method - send or receive method to be executed
262  * @return number of bytes actually sent or received or error when < 0
263  */
264 static int kkc_sock_send_recv(struct kkc_sock *self, void *buf, int buflen,
265  kkc_sock_flags_t flags,
266  int (*kkc_sock_method)(struct kkc_sock *, void *,
267  int, kkc_sock_flags_t))
268 {
269  int error = 0;
270  int total = 0;
271  while ((buflen > 0) && (error >= 0)) {
272  error = kkc_sock_method(self, buf, buflen, flags);
273  mdbg(INFO4, "Buffer starting at %p, xferred %d bytes", buf, error);
274  buf += error;
275  buflen -= error;
276  total += error;
277  }
278  /* When error occured, return error code, otherwise the total */
279  return ((error >= 0) ? total : error);
280 }
