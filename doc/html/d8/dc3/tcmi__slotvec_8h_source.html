39 /** @defgroup tcmi_slotvec_class tcmi_slotvec class
40  *
41  * @ingroup tcmi_lib_group
42  *
43  * This class represents a data structure widely used in
44  * implementation of TCMI. It is a vector of slots where each slot is
45  * head of a doubly linked list of the respective objects.
46  \verbatim
47  +-------+
48  0| slot | -> object1 -> object 2 -> object 3...
49  +-------+
50  1| slot | empty
51  +-------+
52  2| slot | -> object 5 -> object 6...
53  +-------+
54  3| slot | empty
55  +-------+
56  . .
57  . .
58  . .
59  +-------+
60  | slot | -> objectX
61  +-------+
62  n| slot | empty
63  +-------+
64  \endverbatim
65  *
66  * - The vector has limited size specified during initialization by
67  * the user. The benefit is that it provides O(1) access to all its
68  * slots based on index.
69  * - Allows iterating through non-empty slots
70  * - Allows allocating an empty slot with O(1) time complexity
71  *
72  * The data structure internally keeps track of used and unused slots.
73  * The user can exclusively insert objects into empty slots
74  * only. This way, it can be used as a data structure describing some
75  * limited resource. On the other hand(as seen in the figure), the
76  * user is able to insert new elements into any slots, thus yielding a
77  * hashtable. The user needs to select only a suitable hashing
78  * function. The general requirement on the hash function is to yield
79  * the index 0 to n-1 with an approximately uniform distribution.
80  *
81  * Since the datastructure is expected to be accessed from multiple
82  * threads of execution a reference counting is implemented. The user
83  * should always access the slot vector by means of tcmi_slotvec_get()
84  * method and release the instance by means of tcmi_slotvec_put().
85  * The last accessing thread causes the slot vector to be freed from
86  * memory.
87  *
88  *@{
89  */

93  * intended for traversals only, no removals should take place. We

141 /**
142  * A slot vector compound structure. Contains the used/unused slot lists
143  * and the actual vector.
144  */
145 struct tcmi_slotvec {
146  /** The actual slot vector */
147  struct tcmi_slot *slots;
148  /** Number of slots in the vector */
149  u_int slot_count;
150  /** Hash mask - valid only when slot_count is \$2^order\$ and
151  * the slot vector has been created by tcmi_slotvec_hnew(). */
152  u_int hashmsk;
153 
154  /** list of unused slots */
155  struct list_head unused_slots;
156  /** list of used slots */
157  struct list_head used_slots;
158 
159  /** reference counter */
160  atomic_t ref_count;
161 
162  /** current instance reference - used when freeing */
163  /* struct tcmi_slotvec *this;*/
164  /** Protects the slotvector from concurrent modification */
165  spinlock_t s_lock;
166  /** Technical field used for irq safe spinlocks */
167  unsigned long lock_flags;
168 };

176 /**
177  * Slot vector instance accessor.
178  * Checks whether the instance is not currently being freed - if so,
179  * returns NULL. Otherwise increments the reference counter and
180  * returns self.
181  *
182  * @param *self - pointer to this vector instance
183  * @return - self
184  */
185 static inline struct tcmi_slotvec* tcmi_slotvec_get(struct tcmi_slotvec *self)
186 {
187  if (!self)
188  return NULL;
189  atomic_inc(&self->ref_count);
190  return self;
191 }
