51 /**
52  * <<public>> Checkpoint constructor.
53  * - allocates a new instance
54  * - resets the byte counter
55  * - reserves an extra reference for the file
56  * - reset position in the file to the beginning
57  *
58  *
59  * @param *file - points to the file where the checkpoint is to be
60  * written/read
61  * @return tcmi_ckpt instance or NULL
62  */
63 struct tcmi_ckpt* tcmi_ckpt_new(struct file *file)
64 {
65  struct tcmi_ckpt *ckpt;
66  loff_t seek_res;
67 
68  if (!(ckpt = TCMI_CKPT(kmalloc(sizeof(struct tcmi_ckpt),
69  GFP_KERNEL)))) {
70  mdbg(ERR3, "Can't allocate memory for TCMI ckpt");
71  goto exit0;
72  }
73  get_file(file);
74  ckpt->file = file;
75  ckpt->fdcache = NULL;
76  atomic_set(&ckpt->ref_count, 1);
77  mdbg(INFO2, "Created new checkpoint %p", ckpt);
78  /* reset the file position */
79  seek_res = tcmi_ckpt_seek(ckpt, 0, 0);
80 
81  if ( seek_res < 0 ) {
82  kfree(ckpt);
83  return ERR_PTR(seek_res);
84  }
85 
86  return ckpt;
87 
88  /* error handling */
89  exit0:
90  return NULL;
91 
92 }

94 /**
95  * <<public>> Writes a checkpoint header.
96  * The checkpoint header consists of:
97  * - a magic number
98  * - total VM area count
99  * - total file count
100  *
101  * @param *self - this checkpoint instance
102  * @param is_npm - Is this a non-preemptive checkpoint (otherwise preemptive is assumed)
103  * @return 0 upon success
104  */
105 int tcmi_ckpt_write_hdr(struct tcmi_ckpt *self, int is_npm)
106 {
107  self->hdr.checkpoint_arch = ARCH_CURRENT;
108  self->hdr.is_32bit_application = check_is_application_32bit();
109  self->hdr.magic = TCMI_CKPT_MAGIC;
110  self->hdr.is_npm = is_npm;
111  if ((self->hdr.map_count = tcmi_ckpt_map_count(self)) < 0) {
112  mdbg(ERR3, "Failed counting mmap's");
113  goto exit0;
114  }
115  if ((self->hdr.file_count = tcmi_ckpt_file_count(self)) < 0) {
116  mdbg(ERR3, "Failed counting open files");
117  goto exit0;
118  }
119 
120  if (tcmi_ckpt_fdcache_init(self) < 0) {
121  mdbg(ERR3, "Error initializing file descriptor cache!!");
122  goto exit0;
123  }
124  // No need to lock, exclusive access
125  strncpy(self->hdr.comm, current->comm, sizeof(current->comm));
126 
127  mdbg(INFO3, "Writing header of size: %lu Address: %p First 4 bytes: %8x", (unsigned long)sizeof(self->hdr), &self->hdr, ((uint32_t*)(&self->hdr))[0]);
128 
129  return tcmi_ckpt_write(self, &self->hdr, sizeof(self->hdr));
130 
131  /* error handling */
132  exit0:
133  return -EINVAL;
134 }

136 /**
137  * <<public>> Reads the checkpoint header.
138  * The magic number has to match TCMI_CKPT_MAGIC.
139  *
140  * @param *self - this checkpoint instance
141  * @return 0 upon success
142  */
143 int tcmi_ckpt_read_hdr(struct tcmi_ckpt *self)
144 {
145  int err;
146 
147  if ((err = tcmi_ckpt_read(self, &self->hdr, sizeof(self->hdr))) < 0) {
148  mdbg(ERR3, "Can't read checkpoint header %d", err);
149  goto exit0;
150  }
151  if (self->hdr.magic != TCMI_CKPT_MAGIC) {
152  mdbg(ERR3, "Invalid checkpoint magic: %08x, expected: %08x",
153  self->hdr.magic, TCMI_CKPT_MAGIC);
154  goto exit0;
155  }
156  mdbg(INFO4, "Read ckpt header - magic: %08x, map_count: %d, files: %d (Header size: %lu)",
157  self->hdr.magic, self->hdr.map_count, self->hdr.file_count, (unsigned long)sizeof(self->hdr));
158 
159  if (tcmi_ckpt_fdcache_init(self) < 0) {
160  mdbg(ERR3, "Error initializing file descriptor cache!!");
161  goto exit0;
162  }
163  tcmi_ckpt_file_count(self);
164 
165  strncpy(current->comm, self->hdr.comm, sizeof(current->comm));
166 
167  /* TODO: To some arch file ? */
168  #if defined(__x86_64__)
169  if ( self->hdr.is_32bit_application )
170  set_thread_flag(TIF_IA32); /* This flag must be set for 32bit applications running on 64 bit kernel */
171 
172  if ( self->hdr.checkpoint_arch == ARCH_I386 )
173  set_personality(PER_LINUX32_3GB); /* Address space must be limited by 3GB, so that we are possibly able to migrate back */
174  #endif
175 
176  mdbg(INFO3, "Read header. Architecture of checkpoint %d Is 32 bit app: %d", self->hdr.checkpoint_arch, self->hdr.is_32bit_application);
177  return 0;
178 
179  /* error handling */
180  exit0:
181  return -EINVAL;
182 }

184 /**
185  * <<public>> Writes open files into the checkpoint. Iterates through all open
186  * files of the current process and writes each file details into the
187  * checkpoint. Since some files might be just a duplicate file
188  * descriptor, we always check with the file descriptor cache.
189  * For duplicate file descriptors only simplified record is needed.
190  *
191  * @param *self - this checkpoint instance
192  * @return 0 upon success
193  */
194 int tcmi_ckpt_write_files(struct tcmi_ckpt *self)
195 {
196  int fd = 0;
197  int err = 0;
198  struct file *file;
199 
200  /* Check each open file */
201  tcmi_ckpt_foreach_openfile(file, fd) {
202  mdbg(INFO4, "Processing '%s' fd=%d, type=%0o flags %08o, mode %08o, %p",
203  file->f_dentry->d_name.name, fd, file->f_dentry->d_inode->i_mode & S_IFMT,
204  file->f_flags, file->f_mode, file);
205  if ( (err = tcmi_ckpt_openfile_write(self, fd, file)) < 0) {
206  mdbg(INFO3, "Error writing file into checkpoint. Error %d", err);
207  goto exit0;
208  }
209  }
210  return 0;
211 
212  /* error handling */
213  exit0:
214  return err;
215 }

260  }
261 
262  return 0;

294 /**
295  * <<private>> Calculates the number of memory regions of the
296  * current process. We can't simply return current->map_count. We
297  * need to check for special regions that would prevent creating a
298  * valid checkpoint
299  *
300  * For now, we have to verify that the memory region is not:
301  * - VM_IO - I/O mapped devices
302  * - VM_SHARED - shared memory - this will be included later on,
303  * when we support checkpointing and migration of LWP's
304  * - VM_RESERVED - special mappings
305  *
306  * @param *self - this checkpoint instance
307  * @return number of memory regions, or error (< 0)
308  */
309 static int tcmi_ckpt_map_count(struct tcmi_ckpt *self)
310 {
311  struct vm_area_struct *vma;
312  int count = 0;
313  int err = 0;
314 
315  tcmi_ckpt_foreach_vma(vma) {
316  if ( vma_ignore(self, vma) )
317  continue;
318 
319  count++;
320  /* Do not dump I/O mapped devices, shared memory, or special mappings */
321  if (vma->vm_flags & (VM_IO | VM_SHARED | VM_DONTDUMP)) // VM_RESERVED replaced to VM_DONTDUMP by Jiri Rakosnik
322  err = -EINVAL;
323  mdbg(INFO4, "Scanning area start:%08lx, end: %08lx, flags: %08lx, pageprot %08lx",
324  vma->vm_start, vma->vm_end, vma->vm_flags, vma->vm_page_prot.pgprot);
325  }
326  mdbg(INFO4, "Found %d areas, mm indicates: %d", count, current->mm->map_count);
327  return (err ? err : count);
328 }

330 /**
331  * <<private>> Calculates the number of open files of the current process.
332  * For each file check if the file is supported if so,
333  * adjust the counter.
334  *
335  * An error is indicated if any of the files is:
336  * - socket
337  * - pipe
338  * - symlink
339  * - special file
340  *
341  * @param *self - this checkpoint instance
342  * @return number of open files, or error (< 0)
343  */
344 static int tcmi_ckpt_file_count(struct tcmi_ckpt *self)
345 {
346  /* file descriptor count */
347  int fd;
348  int count = 0;
349  int err = 0;
350  struct file *file;
351  struct inode *inode;
352 
353  /* Check each open file */
354  tcmi_ckpt_foreach_openfile(file, fd) {
355  inode = file->f_dentry->d_inode;
356  if (!tcmi_ckpt_openfile_supported(inode)) {
357  mdbg(ERR3, "Unsupported file mode detected %o [FD: %d]", inode->i_mode, fd);
358  err = -EINVAL;
359  goto exit0;
360  }
361  count++;
362  /*mdbg(INFO4, "Scanned file fd=%d, type=%0o '%s'", fd, file->f_flags,
363  file->f_dentry->d_name.name); */
364  }
365 
366  mdbg(INFO4, "Found %d files, err=%d", count, err);
367  return count;
368 
369  /* error handling */
370  exit0:
371  return -EINVAL;
372 }
