140 /**
141  * <<private>> Getgroups system call hook
142  *
143  * @return the session ID of the calling process
144  */
145 static long tcmi_syscall_hooks_sys_getgroups(int size, gid_t *groups)
146 {
147  if( size == 0 )
148  return tcmi_rpc_call2(tcmi_guest_rpc, TCMI_RPC_SYS_GETGROUPS, 0, (unsigned long)NULL);
149  else{
150  if( !access_ok( VERIFY_WRITE, groups, size * sizeof(gid_t) ) )
151  return -EFAULT;
152  return tcmi_rpc_call2(tcmi_guest_rpc, TCMI_RPC_SYS_GETGROUPS, size, (unsigned long)groups);
153  }
154 }

162 {
163  if( size > NGROUPS_MAX )

165  else{

167  return -EFAULT;
168  return tcmi_rpc_call2(tcmi_guest_rpc, TCMI_RPC_SYS_SETGROUPS,
169  size, (unsigned long)groups );
