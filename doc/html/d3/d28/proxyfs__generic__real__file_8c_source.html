46  * @param *self - this real_file class instance
47  *
48  * @return amount of bytes written

53  mm_segment_t old_fs;
54  void *buf_addr;

57

59 
60  mdbg(INFO3, "File %lu has at least %lu bytes pending in buffer.", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), (unsigned long)buf_data );
61 
62  if( buf_data ){
63  buf_addr = circ_buf_get_read_addr( & PROXYFS_FILE(self)->read_buf );
64  pos = self->file->f_pos;
65  old_fs = get_fs();
66

68 
69  lenght = vfs_write( self->file, buf_addr, buf_data, &pos );
70  mdbg(INFO3, "vfs_write returned %d", lenght );
71 
72  if( lenght > 0 ){

74  total = lenght;
75  // If we wrote everything, we try to write one more time,
76  // because we are using circular buffer
77  if( lenght == buf_data ){
78  buf_data = circ_buf_get_read_size( & PROXYFS_FILE(self)->read_buf );
79  if( buf_data ){

81  lenght = vfs_write(self->file, buf_addr, buf_data, &pos);

84  circ_buf_adjust_read_ptr( & PROXYFS_FILE(self)->read_buf, lenght );
85  total += lenght;
86  }
87  }
88  }
89  }
90  else
91 
92  self->file->f_pos = pos;
93  set_fs(old_fs);
94  }
