49 /** @defgroup tcmi_ckpt_class tcmi_ckpt class
50  *
51  * @ingroup tcmi_ckptcom_class
52  *
53  * This is a helper class used by the checkpointing component that
54  * provides functionality to:
55  * - write current process into a checkpoint file
56  * - restore a current process from a checkpoint file
57  *
58  * It doesn't do the whole job at once, it relies on its controlling
59  * class (\link tcmi_ckptcom_class checkpointing component \endlink).
60  * The checkpointing component instructs the checkpoint instance what
61  * to do in each step (this applies to creating the checkpoint as
62  * well as restoring the process execution).
63  *
64  * Implementation is Linux kernel specific.
65  *
66  * @{

163 /**
164  * <<public>> Releases the instance.
165  *
166  * @param *self - pointer to this instance
167  */
168 static inline void tcmi_ckpt_put(struct tcmi_ckpt *self)
169 {
170  if (self && atomic_dec_and_test(&self->ref_count)) {
171  mdbg(INFO4, "Destroying TCMI ckpt, %p", self);
172  fput(self->file);
173  tcmi_fdcache_put(self->fdcache);
174  kfree(self);
175  }
176 }

183 /**
184  * <<private>> Executes a specified VFS method on the checkpoint
185  * file.
186  *
187  * @param *self - this checkpoint instance
188  * @param *data - buffer for the data
189  * @param count - number of bytes to be processed
190  * @param *vfs_method - VFS method to be executed on the checkpoint
191  * and data
192  * @return 0 upon successful vfs method execution or an error (< 0)
193  */
194 static inline int tcmi_ckpt_read_write(struct tcmi_ckpt *self,
195  void *data, int count,
196  vfs_method_t *vfs_method)
197 {
198  mm_segment_t old_fs;
199  int result = 0;
200 
201  old_fs = get_fs();
202  set_fs(get_ds());
203  /* The cast to a user pointer is valid due to the set_fs() */
204  result = vfs_method(self->file, (void __user *)data, count, &self->file->f_pos);
205  set_fs(old_fs);
206 
207  return (result == count ? 0 : -EINVAL);
208 }
