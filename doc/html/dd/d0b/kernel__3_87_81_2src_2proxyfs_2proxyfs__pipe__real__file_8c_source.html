44 /** <<private>> Write to file from read buffer
45  * @param *self - this real_file class instance
46  *
47  * @return amount of bytes written
48  * */
49 static int proxyfs_pipe_real_file_write(struct proxyfs_real_file *self)
50 {
51  int lenght, total= 0, poll_result, nread;
52  mm_segment_t old_fs;
53  void *buf_addr;
54  size_t buf_data;
55  loff_t pos;
56 
57  mdbg(INFO3, "[File %lu] fmode %d", proxyfs_file_get_file_ident(PROXYFS_FILE(self)),
58  self->file->f_mode);
59 
60 restart_write:
61  //nread = proxyfs_real_file_ioctl_nread(self); // TODO: Remove this, just for debugging now
62  poll_result = proxyfs_real_file_poll_write(self);
63  mdbg(INFO3, "[File %lu] polling result: %d (nread: %d)", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), poll_result, nread);
64 
65  if ( poll_result < 1 ) // No space for writing (in pipe it means no free buffer)
66  return total;
67 
68  buf_data = circ_buf_get_read_size( & PROXYFS_FILE(self)->read_buf );
69 
70  mdbg(INFO3, "[File %lu] At least %lu bytes pending in buffer", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), (unsigned long)buf_data);
71  circ_buf_dump("Pipe real file write", & PROXYFS_FILE(self)->read_buf);
72 
73  if( buf_data ){
74  pos = self->file->f_pos;
75  if( PROXYFS_PIPE_REAL_FILE(self)->write_counter == 0 ){
76  if( circ_buf_look_at( &PROXYFS_FILE(self)->read_buf,
77  (void*)&PROXYFS_PIPE_REAL_FILE(self)->write_counter,
78  sizeof(PROXYFS_PIPE_REAL_FILE(self)->write_counter), 0)
79  == sizeof(PROXYFS_PIPE_REAL_FILE(self)->write_counter) )
80  {
81  mdbg(INFO3, "[File %lu] New write block size %d.",
82  proxyfs_file_get_file_ident(PROXYFS_FILE(self)),
83  PROXYFS_PIPE_REAL_FILE(self)->write_counter );
84 
85  if( PROXYFS_PIPE_REAL_FILE(self)->write_counter <= PIPE_BUF )
86  PROXYFS_PIPE_REAL_FILE(self)->atomic_write = 1;
87  else
88  PROXYFS_PIPE_REAL_FILE(self)->atomic_write = 0;
89 
90  circ_buf_adjust_read_ptr( & PROXYFS_FILE(self)->read_buf,
91  sizeof(PROXYFS_PIPE_REAL_FILE(self)->write_counter) );
92 
93  total += sizeof(PROXYFS_PIPE_REAL_FILE(self)->write_counter);
94  goto restart_write;
95  }
96  else
97  PROXYFS_PIPE_REAL_FILE(self)->write_counter = 0;
98  }
99  else if( PROXYFS_PIPE_REAL_FILE(self)->atomic_write ){ // Write must be atomic
100  int total_available = circ_buf_count(& PROXYFS_FILE(self)->read_buf);
101  if ( total_available < PROXYFS_PIPE_REAL_FILE(self)->write_counter )
102  goto done;
103 
104  if( buf_data < PROXYFS_PIPE_REAL_FILE(self)->write_counter ){ // Data overlaps
105  circ_buf_look_at( &PROXYFS_FILE(self)->read_buf,
106  PROXYFS_PIPE_REAL_FILE(self)->atomic_buf,
107  PROXYFS_PIPE_REAL_FILE(self)->write_counter, 0);
108 
109  old_fs = get_fs();
110  set_fs(KERNEL_DS);
111  lenght = vfs_write( self->file, PROXYFS_PIPE_REAL_FILE(self)->atomic_buf,
112  PROXYFS_PIPE_REAL_FILE(self)->write_counter, &pos );
113  set_fs(old_fs);
114 
115  }
116  else{ // Data doesn't overlap
117  buf_addr = circ_buf_get_read_addr( & PROXYFS_FILE(self)->read_buf );
118  old_fs = get_fs();
119  set_fs(KERNEL_DS);
120  lenght = vfs_write( self->file, buf_addr,
121  PROXYFS_PIPE_REAL_FILE(self)->write_counter, &pos );
122  set_fs(old_fs);
123  }
124 
125  if( lenght > 0 ){ // We wrote something
126  mdbg(INFO3, "[File %lu] Atomicaly written %d Bytes",
127  proxyfs_file_get_file_ident(PROXYFS_FILE(self)), lenght);
128 
129  if( lenght != PROXYFS_PIPE_REAL_FILE(self)->write_counter )
130  mdbg(ERR1, "[File %lu] Fatal error in atomic write",
131  proxyfs_file_get_file_ident(PROXYFS_FILE(self)));
132 
133  PROXYFS_PIPE_REAL_FILE(self)->write_counter = 0;
134  self->file->f_pos = pos;
135  total += lenght;
136  circ_buf_adjust_read_ptr( & PROXYFS_FILE(self)->read_buf, lenght );
137  goto restart_write;
138  }
139  else
140  mdbg(INFO3, "[File %lu] Write failed with %d",
141  proxyfs_file_get_file_ident(PROXYFS_FILE(self)), lenght);
142  }
143  else{ // Write hasn't to be atomic
144  buf_addr = circ_buf_get_read_addr( & PROXYFS_FILE(self)->read_buf );
145  if( buf_data > PROXYFS_PIPE_REAL_FILE(self)->write_counter )
146  buf_data = PROXYFS_PIPE_REAL_FILE(self)->write_counter;
147 
148  old_fs = get_fs();
149  set_fs(KERNEL_DS);
150  lenght = vfs_write( self->file, buf_addr, buf_data, &pos );
151  set_fs(old_fs);
152 
153  if( lenght > 0 ){
154  circ_buf_adjust_read_ptr( & PROXYFS_FILE(self)->read_buf, lenght );
155  self->file->f_pos = pos;
156  PROXYFS_PIPE_REAL_FILE(self)->write_counter -= lenght;
157  total += lenght;
158  goto restart_write;
159  }
160  }
161  }
162 
163 done:
164  return total;
165 }

167 /** <<private>> Reads from file and write to writte buffer
168  * @param *self - this real_file class instance
169  *
170  * @return amount of bytes read
171  * */
172 static int proxyfs_pipe_real_file_read(struct proxyfs_real_file *self)
173 {
174  int lenght = 0, total_read = 0, poll_result;
175  mm_segment_t old_fs;
176  void *buf_addr;
177  size_t buf_data, total_buffer_size;
178  loff_t pos;
179 
180  // We read from pipe ONLY in case the remote process ever used it as a read pipe..
181  // Workaround for inpolite processes like Makefile.. see MSG_READ_REQUEST comments for details
182  if (!proxyfs_file_was_read_requested(PROXYFS_FILE(self)) )
183  return -EAGAIN;
184 
185  poll_result = proxyfs_real_file_poll_read(self);
186  mdbg(INFO3, "[File %lu] polling result: %d", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), poll_result);
187 
188  if ( poll_result < 1 ) // Are there some date pending?
189  return -EAGAIN;
190 
191  total_buffer_size = circ_buf_free_space( &PROXYFS_FILE(self)->write_buf );
192  if ( total_buffer_size == 0 ) // We do not have free space in the bufer at the moment
193  return -EAGAIN;
194 
195  buf_data = circ_buf_get_write_size( & PROXYFS_FILE(self)->write_buf );
196 
197  mdbg(INFO3, "Trying read file %lu, buffer space is at least %lu", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), (unsigned long)buf_data);
198  circ_buf_dump("Pipe real file read", & PROXYFS_FILE(self)->write_buf);
199 
200  if( buf_data ){
201  buf_addr = circ_buf_get_write_addr( & PROXYFS_FILE(self)->write_buf );
202  pos = self->file->f_pos;
203  old_fs = get_fs();
204 
205  set_fs(KERNEL_DS);
206 
207  lenght = vfs_read(self->file, buf_addr, buf_data, &pos);
208 
209  if( lenght > 0 ){
210  total_read = lenght;
211  mdbg(INFO3, "[File %lu] Read from pipe content %.*s.", proxyfs_file_get_file_ident(PROXYFS_FILE(self)), (int)lenght, (char*)buf_addr );
212 
213 
214  circ_buf_adjust_write_ptr( & PROXYFS_FILE(self)->write_buf, lenght );
215  // If we wrote everything, we try to write one more time,
216  // because we are using circular buffer
217  if( lenght == buf_data ){
218  buf_data = circ_buf_get_write_size( & PROXYFS_FILE(self)->write_buf );
219  if( buf_data ){
220  // We have to poll once more, since now there may be no more data!
221  if ( !proxyfs_real_file_poll_read(self) ) {
222  self->file->f_pos = pos;
223  set_fs(old_fs);
224  goto out;
225  }
226 
227  buf_addr = circ_buf_get_write_addr( & PROXYFS_FILE(self)->write_buf );
228  lenght = vfs_read(self->file, buf_addr, buf_data, &pos);
229  if( lenght > 0 ) {
230  circ_buf_adjust_write_ptr( & PROXYFS_FILE(self)->write_buf, lenght );
231  total_read += lenght;
232  }
233  }
234  }
235  }
236 
237  self->file->f_pos = pos;
238  set_fs(old_fs);
239  }
240 
241 out:
242  return total_read;
243 }

309 /** <<public>> Duplicates a proxyfs real file
310  *
311  * @param *self - pointer to the instance to be duplicated
312  * @return Duplicated file instance
313  */
314 static struct proxyfs_real_file* proxyfs_pipe_real_file_duplicate(struct proxyfs_real_file *self)
315 {
316 
317  struct proxyfs_pipe_real_file *clone;
318  mdbg(INFO4, "Duplicating file %lu", proxyfs_file_get_file_ident(PROXYFS_FILE(self)));
319  clone = (struct proxyfs_pipe_real_file*)kmalloc(sizeof(struct proxyfs_pipe_real_file), GFP_KERNEL);
320  if ( !clone )
321  return NULL;
322 
323  if( proxyfs_real_file_init( PROXYFS_REAL_FILE(clone) ) != 0 )
324  goto exit1;
325 
326  if( proxyfs_pipe_real_file_init( PROXYFS_PIPE_REAL_FILE(clone) ) != 0 )
327  goto exit2;
328 
329  PROXYFS_FILE(clone)->file_identifier.file_ident = PROXYFS_FILE(self)->file_identifier.file_ident;
330  PROXYFS_FILE(clone)->file_identifier.file_owner_pid = PROXYFS_FILE(self)->file_identifier.file_owner_pid;
331  PROXYFS_REAL_FILE(clone)->file = PROXYFS_REAL_FILE(self)->file;
332  // We can do this here as we own a reference to file (the file being clonned) and we have a lock holding the file so it cannot be released
333  get_file(PROXYFS_REAL_FILE(self)->file);
334 
335  return PROXYFS_REAL_FILE(clone);
336 
337 exit2:
338  proxyfs_real_file_destroy( PROXYFS_REAL_FILE(self) );
339 exit1:
340  kfree(self);
341  return NULL;
342 }
