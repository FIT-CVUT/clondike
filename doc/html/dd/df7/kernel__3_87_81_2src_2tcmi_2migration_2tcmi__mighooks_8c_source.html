120 /**
121  * <<private>> Exit system call hook. Delegates work to the task
122  * helper that notifies current process by submitting a task_exit
123  * method and supplying the exit code. The task handles the rest on
124  * its own.
125  *
126  * The priority notification is used, so that the exit follows
127  * immediately.
128  *
129  * This hook is called for every process in the system, but does
130  * something only for TCMI tasks.
131  *
132  * @param code - process exit code
133  * @return - status of how the notification went.
134  */
135 static long tcmi_mighooks_do_exit(long code)
136 {
137  int prio = 1, is_shadow;
138  struct rusage rusage;
139  mm_segment_t old_fs;
140  long res;
141 
142  is_shadow = (current->tcmi.task_type == shadow || current->tcmi.task_type == shadow_detached);
143 
144  old_fs = get_fs();
145  set_fs(get_ds());
146  res = getrusage(current, RUSAGE_SELF, &rusage);
147  set_fs(old_fs);
148  director_task_exit(current->pid, code, &rusage);
149 
150  res = tcmi_taskhelper_notify_current(tcmi_task_exit, &code,
151  sizeof(code), prio);
152 
153  if ( is_shadow ) { // If the exitting task is a shadow task, we release all its proxy files as they won't be needed
154  proxyfs_sync_files_on_exit(); // We have to synchronize all files first so that no new writes occur after server release
155  proxyfs_server_release_all();
156  }
157 
158  return res;
159 }

177 static void tcmi_try_npm_on_exec(const char *filename, const char __user * const __user * argv,
178  const char __user * const __user * envp, struct pt_regs *regs,
179  struct tcmi_man* man, int is_guest)
180 {
181  struct rusage rusage;
182  mm_segment_t old_fs;
183  int migman_to_use = -1;
184  int migrate_home = 0;
185 
186  old_fs = get_fs();
187  set_fs(get_ds());
188  getrusage(current, RUSAGE_SELF, &rusage);
189  set_fs(old_fs);
190  if (director_npm_check(current->pid, current_euid(), is_guest, filename, argv, envp, &migman_to_use, &migrate_home, &rusage) != 1 )
191  return;
192 
193  migrate_home = is_guest ? migrate_home : 0;
194 
195  // Check if we should emigrate and if so where
196  if ( migman_to_use != -1 || migrate_home ) {
197  struct tcmi_npm_params* npm_params = vmalloc(sizeof(struct tcmi_npm_params));
198  int err = extract_tcmi_npm_params(npm_params, filename, argv, envp);
199  mdbg(INFO4, "AFTER EXTRACT Filename: %s, Argc: %d, Evnpc: %d.", npm_params->file_name, npm_params->argsc, npm_params->envpc);
200  if ( err ) {
201  mdbg(ERR3, "Error extracting npm params: %d. Skipping npm emigration attempt.", err);
202  vfree(npm_params);
203  } else {
204  if ( migrate_home )
205  tcmi_penman_migrateback_npm(man, regs, npm_params);
206  else
207  tcmi_man_emig_npm(man, current->pid, migman_to_use, regs, npm_params);
208  // If we got here, the migration was not performed => free npm params
209  mdbg(INFO3, "Freeing npm_params after unsuccessful npm migration attempt.");
210  vfree(npm_params);
211  }
212  }
213 }

215 static long tcmi_mighooks_execve(const char *filename, const char __user * const __user * argv, const char __user * const __user * envp, struct pt_regs *regs) {
216  if ( !current->tcmi.tcmi_task ) {
217  // The task is not controlled by clondike => we can non-preemptively migrate it ONLY if there is CCN registered on current node
218  struct tcmi_ccnman* ccn_man = tcmi_ccnman_get_instance();
219  if ( ccn_man != NULL )
220  tcmi_try_npm_on_exec(filename, argv, envp, regs, TCMI_MAN(ccn_man), 0);
221  } else if ( tcmi_task_get_type(current->tcmi.tcmi_task) == guest ) {
222  // We allow npm only after we get initial execve executed.. not during the initial image load!
223  if ( tcmi_task_get_execve_count(current->tcmi.tcmi_task) > 1 ) {
224  // Task is guest, so DN has to take care of its possible NPM
225  struct tcmi_penman* pen_man = tcmi_penman_get_instance();
226 
227  mdbg(INFO3, "Guest exec hook called. File %s", filename);
228  tcmi_try_npm_on_exec(filename, argv, envp, regs, TCMI_MAN(pen_man), 1);
229  } else {
230  // On first exceve, we notify about possible vfork done
231  // TODO: We may optimize this, and notify only in case it is required.. this would require some flag in tcmi_task, that would be caried here during emigration
232  struct tcmi_msg *m;
233  // We do not aquire reference to t_task as it cannot be release at this moment
234  struct tcmi_task* t_task = current->tcmi.tcmi_task;
235 
236  if ( !t_task) {
237  mdbg(ERR3, "No tcmi task associated!");
238  goto exit0;
239  }
240 
241  if (!(m = tcmi_vfork_done_procmsg_new_tx(t_task->remote_pid))) {
242  mdbg(ERR3, "Can't create vfork done message");
243  // TODO: What to do in this case? ;) For now we just ignore it.
244  goto exit0;
245  }
246 
247 
248  tcmi_task_send_anonymous_msg(t_task, m);
249  tcmi_msg_put(m);
250 
251  exit0:
252  ;
253  }
254  tcmi_task_inc_execve_count(current->tcmi.tcmi_task);
255  } else {
256  // We are shadow task => ignore this hook, shadow execves only when restarting process that was migrated back
257  tcmi_task_inc_execve_count(current->tcmi.tcmi_task);
258  }
259 
260  return 0;
261 };
