64  goto exit0;

66  if (!(msg = TCMI_RPCRESP_PROCMSG(kmalloc(sizeof(struct tcmi_rpcresp_procmsg),

70  }
71  msg->elem_free = NULL;

74  &rpcresp_procmsg_ops)) {
75  mdbg(ERR3, "Error initializing process message %x", msg_id);
76  goto exit1;
77  }
78 
79  return TCMI_MSG(msg);
80 
81  /* error handling */
82  exit1:
83  kfree(msg);
84  exit0:
85  return NULL;

101  va_list args;
102 
103  if ( !(self_msg = TCMI_RPCRESP_PROCMSG( tcmi_rpcresp_procmsg_new_tx(tcmi_msg_req_id(m),
104  tcmi_task_remote_pid(current->tcmi.tcmi_task) ))) ){
105  mdbg(ERR3, "Error creating RPC response message");

115  return NULL;
116  }

139  mdbg(INFO3, "Adding parametr #%d: at %p size %lu", i, self_msg->elem_base[i], (unsigned long)self_msg->elem_size[i] );
140  }
141  va_end(args);
142  }
143 
144 
145  return TCMI_MSG(self_msg);
146 }

244 /**
245  * <<private>> Receives the message via a specified connection.
246  *
247  * @param *self - this message instance
248  * @param *sock - KKC socket used for receiving message data
249  * @return 0 when successfully received.
250  *
251  * WARNING: Memory for data is allocated at one time, so when freeing free() must be called
252  * on elem_base[0]. It's done automatically in last mdg_put();
253  */
254 static int tcmi_rpcresp_procmsg_recv(struct tcmi_procmsg *self, struct kkc_sock *sock)
255 {
256  /* struct tcmi_rpcresp_procmsg *self_msg = TCMI_RPCRESP_PROCMSG(self); */
257  int err, data_size, total, i;
258  struct tcmi_rpcresp_procmsg *self_msg = TCMI_RPCRESP_PROCMSG(self);
259 
260  err = kkc_sock_recv(sock, &self_msg->rpcnum,
261  sizeof(self_msg->rpcnum) + sizeof(self_msg->nmemb) +
262  sizeof(self_msg->rtn_code), KKC_SOCK_BLOCK);
263  if( err < 0 ) {
264  mdbg(ERR3, "Error receiving RPC message header. Error code: %d", err);
265  return err;
266  }
267  total = err;
268 
269  if( self_msg->nmemb > 0 ){
270  /* Allocate memory for array of elements size and array of pointers to elements */
271  if( (self_msg->elem_free = kmalloc(SIZE_OF_ELEM_SIZE +
272  SIZE_OF_ELEM_BASE + SIZE_OF_ELEM_FREE, GFP_KERNEL)) == NULL){
273  mdbg(ERR3, "Can't create rpc message");
274  }
275  self_msg->elem_size = (void*)(self_msg->elem_free) + SIZE_OF_ELEM_FREE;
276  self_msg->elem_base = (void*)(self_msg->elem_size) + SIZE_OF_ELEM_SIZE;
277  self_msg->elem_base[0] = NULL;
278 
279  err = kkc_sock_recv(sock, self_msg->elem_size, SIZE_OF_ELEM_SIZE, KKC_SOCK_BLOCK);
280  if( err < 0 ) {
281  mdbg(ERR3, "Error receiving RPC#%d message elem size. Error code: %d", self_msg->rpcnum, err);
282  return err;
283  }
284  total += err;
285 
286  /* Allocate memory for data */
287  for(data_size = 0, i = 0; i < self_msg->nmemb; data_size += self_msg->elem_size[i++]);
288  if( (self_msg->elem_base[0] = kmalloc( data_size, GFP_KERNEL ) ) == NULL){
289  mdbg(ERR3, "Can't create rpc message2");
290  }
291  self_msg->elem_free[0] = TCMI_RPCRESP_PROCMSG_KFREE; /* We allocated data, so we have to free it */
292 
293  err = kkc_sock_recv(sock, self_msg->elem_base[0], data_size, KKC_SOCK_BLOCK);
294  if( err < 0 ) {
295  mdbg(ERR3, "Error receiving RPC#%d message elements. Error code: %d", self_msg->rpcnum, err);
296  return err;
297  }
298  total += err;
299 
300  for( i = 1; i < self_msg->nmemb; i++ ){
301  self_msg->elem_base[i] = self_msg->elem_base[i-1] + self_msg->elem_size[i-1];
302  self_msg->elem_free[i] = TCMI_RPCRESP_PROCMSG_DONTFREE;
303  }
304  }
305  mdbg(INFO3, "Received RPC#%d response message (%d Bytes)", self_msg->rpcnum, total);
306 
307  return 0;
308 }

311  * <<private>> Receives the message via a specified connection.

313  * @param *self - this message instance

315  * @return 0 when successfully sent.

317 static int tcmi_rpcresp_procmsg_send(struct tcmi_procmsg *self, struct kkc_sock *sock)
318 {
319  int err, total, i;
320  struct tcmi_rpcresp_procmsg *self_msg = TCMI_RPCRESP_PROCMSG(self);
321 
322  err = kkc_sock_send(sock, &self_msg->rpcnum,
323  sizeof(self_msg->rpcnum) + sizeof(self_msg->nmemb) +
324  sizeof(self_msg->rtn_code), KKC_SOCK_BLOCK);
325  if( err < 0 ) {
326  mdbg(ERR3, "Error sending RPC#%d message header. Error code: %d", self_msg->rpcnum, err);
327  return err;
328  }
329  total = err;
330 
331  if( self_msg->nmemb > 0 ){
332  err = kkc_sock_send(sock, self_msg->elem_size, SIZE_OF_ELEM_SIZE, KKC_SOCK_BLOCK);
333  if( err < 0 ) {
334  mdbg(ERR3, "Error sending RPC#%d message elem size. Error code: %d", self_msg->rpcnum, err);
335  return err;
336  }
337  total += err;
338 
339  for( i = 0; i < self_msg->nmemb; i++ ){
340  err = kkc_sock_send(sock, self_msg->elem_base[i],
341  self_msg->elem_size[i], KKC_SOCK_BLOCK);
342 
343  if( err < 0 ) {
344  mdbg(ERR3, "Error sending RPC#%d message element[%d]. Error code: %d", self_msg->rpcnum, i, err);
345  return err;
346  }
347  total += err;
348  }
349  }
350  mdbg(INFO3, "Sent RPC#%d message (%d Bytes)", self_msg->rpcnum, total);
351 
352  return 0;
353 }

355 /**
356  * <<private>> Frees RPC message resources
357  *
358  * @param *self - this message instance
359  */
360 static void tcmi_rpcresp_procmsg_free(struct tcmi_procmsg *self)
361 {
362  struct tcmi_rpcresp_procmsg *self_msg = TCMI_RPCRESP_PROCMSG(self);
363  int i;
364  if( self_msg->nmemb ){
365  for( i = 0; i < self_msg->nmemb; i++ ){
366  switch( self_msg->elem_free[i] ){
367  case TCMI_RPCRESP_PROCMSG_DONTFREE:
368  mdbg(INFO3, "Skipping unallocate of #%d data at %p size %lu",
369  i, self_msg->elem_base[i], (unsigned long)self_msg->elem_size[i]);
370  break;
371  case TCMI_RPCRESP_PROCMSG_KFREE:
372  mdbg(INFO3, "Freeing (kfree) data #%d at %p size %lu",
373  i, self_msg->elem_base[i], (unsigned long)self_msg->elem_size[i]);
374  kfree(self_msg->elem_base[i]);
375  break;
376  case TCMI_RPCRESP_PROCMSG_VFREE:
377  mdbg(INFO3, "Freeing (vfree) data #%d at %p size %lu",
378  i, self_msg->elem_base[i], (unsigned long)self_msg->elem_size[i]);
379  vfree(self_msg->elem_base[i]);
380  break;
381  default:
382  mdbg(ERR3, "Unknown action in %p->elem_free[%d]", self_msg, i);
383  }
384  }
385  kfree(self_msg->elem_free);
386  }
387 }
