47 /** @defgroup tcmi_msg_class tcmi_msg class
48  *
49  * @ingroup tcmi_comm_group
50  *
51  * This class represents a generic message in TCMI communication
52  * protocol. A message can be part of a \link tcmi_transaction_class
53  * transaction\endlink. For that reason it carries two extra ID's
54  * -# a transaction request ID - identifies a transaction started
55  * by the sending party. The sending party expects a message with response
56  * transaction ID set to this value. This ID is assigned automatically
57  * upon message creation depending on a particular messagere type.
58  * -# a transaction response ID - this value comes from the
59  * transaction request ID of the original request message that the new
60  * message is replying to. This ID has to be specified explicitely
61  * when building the response message.
62  *
63  * A message is instantiated always twice during its lifetime:
64  *
65  * - the sender builds the message based on parameters specified by
66  * the message instantiator.
67  * - the receiver performs message materialization from the data stream.
68  *
69  * These two cases are substantially different. In the first case the
70  * instantiator knows what message is to be built. While the receiver
71  * is left to the very last moment upon discovery of the message ID to build
72  * the message. A suitable solution to this problem is provided by a
73  * a <<factory>> design pattern. The message instantiation on the receiver
74  * side is carried out by \link tcmi_msg_factory_class message factory \endlink.
75  *
76  * A generic message consists of following parts:
77  * - message id - unique identifier that describes a particular message type.
78  * This is specified by particular subclasses.
79  * - transaction request ID - explained above
80  * - transaction response ID - explained above
81  *
82  * Generic message methods:
83  * - init_rx - initializes a new message so that it is ready to be
84  * received.
85  * - init_tx - initializes a new message for sending based on
86  * parameters specified by the user.
87  * - recv - receives a message via a communication channel
88  * - send - sends out a message via a communication channel
89  * - deliver - delivers a request message to a queue or delivers a response
90  * message to a transaction and completes the transaction.
91  *
92  * A particular message of the protocol then provides following interface:
93  * - new_rx - this method is registered with the factory class, so that
94  * the build process is fully automated.
95  * - new_rx_err - also registered with the factory class, provides an
96  * alternate(error) version of the message.
97  * - new_tx - builds a message based on parameters specified by the user
98  *
99  * @{
100  */
101

189 /**
190  * <<public>> Instance accessor, increments the reference count.

192  * @param *self - pointer to this message instance

194 static inline struct tcmi_msg* tcmi_msg_get(struct tcmi_msg *self)
195 {
196  if (self) {
197  atomic_inc(&self->ref_count);
198  }
199  return self;
200 }

202 /**
203  * <<public>> Decrements reference counter, if it reaches 0 the
204  * custom free method is called if defined. Eventually, the instance
205  * is destroyed.
206  *
207  * The user is responsible for taking the message out of the message
208  * queue or detaching it from the transaction when needed.
209  *
210  * @param *self - pointer to this message instance
211  */
212 static inline void tcmi_msg_put(struct tcmi_msg *self)
213 {
214  if (!self)
215  return;
216  if (atomic_dec_and_test(&self->ref_count)) {
217  mdbg(INFO4, "Destroying message ID %x, %p", self->msg_id, self);
218  if(self->msg_ops && self->msg_ops->free)
219  self->msg_ops->free(self);
220  tcmi_transaction_put(self->transaction);
221  kfree(self);
222  }
223 }
