42 /**
43  * <<public>> This method is used when initializing a new entry.
44  * It follows these steps:
45  *
46  * - converts the nameformat string and the args into a regular string
47  * This will be the name assigned to the dentry
48  * - checks whether we have a valid parent directory
49  * - check whether an entry with the same name already exists
50  * in the parent directory entry(if so, an error is returned)
51  * - creates a new inode with the specified super block and operations
52  * - creates a new dentry associated with the inode.
53  * - sets a entry_operations specific to this particular entry type
54  * This supports polymorphism.
55  *
56  * @param *self - pointer to this instance
57  * @param *parent - pointer to the parent entry - has to be a directory
58  * @param *mode - file type and access rights for the inode
59  * @param *entry_ops - custom operations of the child class
60  * @param *i_ops - inode operations
61  * @param *f_ops - file operations(also stored in the new inode)
62  * @param namefmt - nameformat string (printf style)
63  * @param args - arguments to the format string
64  * @return 0 upon success
65  */
66 int tcmi_ctlfs_entry_init(struct tcmi_ctlfs_entry *self,
67  struct tcmi_ctlfs_entry *parent,
68  mode_t mode,
69  struct tcmi_ctlfs_ops *entry_ops,
70  const struct inode_operations *i_ops,
71  const struct file_operations *f_ops,
72  const char namefmt[], va_list args)
73 {
74  struct inode *dir;
75  struct inode *inode;
76 
77  vsnprintf(self->name, sizeof(self->name), namefmt, args);
78  mdbg(INFO4, "Initializing a new tcmi_ctlfs_entry, '%s'(%p)", self->name, self);
79  if (!parent) {
80  mdbg(ERR4, "Can't initialized new entry '%s', no parent specified! ", self->name);
81  goto exit0;
82  }
83  mdbg(INFO4, "Parent entry='%s'", tcmi_ctlfs_entry_name(parent));
84  /* */
85  dir = parent->dentry->d_inode;
86  if (!S_ISDIR(dir->i_mode)) {
87  mdbg(ERR3, "Can't initialize new entry '%s', parent entry is not a directory!", self->name);
88  goto exit0;
89  }
90 
91 
92  /* check parent for an entry with 'name' */
93  if (tcmi_ctlfs_entry_exists(parent, self->name)) {
94  mdbg(ERR3,"Can't initialize new entry '%s', parent entry contains an entry of the same name!",
95  self->name);
96  goto exit0;
97  }
98 
99  /* allocate a new inode */
100  if (!(inode = tcmi_ctlfs_get_inode(dir, dir->i_sb, mode, i_ops, f_ops))) {
101  mdbg(ERR3, "Failed to allocate an inode for '%s'", self->name);
102  goto exit0;
103  }
104  /* allocate a new dentry, associated with the inode */
105  if (tcmi_ctlfs_entry_alloc_dentry(self, parent, inode)) {
106  mdbg(ERR3, "Failed to allocate a new dentry for %s", self->name);
107  goto exit1;
108  }
109  /* Custom operations of the child class */
110  self->entry_ops = entry_ops;
111  return 0;
112  /* error handling */
113  exit1:
114  iput(inode);
115  exit0:
116  return -ENOSPC;
117 }

121  *
122  * @param *self - pointer to this instance
123  * @param *sb - pointer to the super block of the filesystem
124  * @param *mode - file type and access rights for the inode
125  * @param *entry_ops - custom operations of the child class
126  * @param *i_ops - inode operations
127  * @param *f_ops - file operations(also stored in the new inode)
128  * @return 0 upon success
129  */
130 int tcmi_ctlfs_rootentry_init(struct tcmi_ctlfs_entry *self,
131  struct super_block *sb,
132  mode_t mode,
133  struct tcmi_ctlfs_ops *entry_ops,
134  const struct inode_operations *i_ops,
135  const struct file_operations *f_ops)
136 {
137  struct dentry *root;
138  struct inode *inode;
139

157  return 0;

159  exit1:
160  iput(inode);
161  exit0:
162  return -ENOSPC;
163 }

167 /**
168  * <<public>> Iterates through all entry predecessors and calls
169  * the specified method for each entry (including self). The root
170  * entry is omitted. The callback function is supplied with the
171  * pointer to data that the users specified as parameter. This
172  * provides a safe way of iterating through all predecessors. The
173  * reference counters are used to control the access from multiple
174  * threads.
175  *
176  * @param *self - pointer to this instance
177  * @param *func - pointer to the callback function
178  * @param *data - pointer to the data that is passed the callback.
179  */
180 void tcmi_ctlfs_entry_traverse_pred(struct tcmi_ctlfs_entry *self, callback *func, void *data)
181 {
182  struct tcmi_ctlfs_entry *parent;
183  struct tcmi_ctlfs_entry *entry = self;
184  /* increment the ref counter and get the instance */
185  parent = tcmi_ctlfs_entry_get(entry);
186 
187  /* mdbg(INFO4, "getting %s", tcmi_ctlfs_entry_name(entry)); */
188  while (!tcmi_ctlfs_entry_isroot(parent)) {
189  entry = parent;
190  parent = tcmi_ctlfs_entry_get_parent(entry);
191  /* mdbg(INFO4, "getting %s", tcmi_ctlfs_entry_name(parent)); */
192  (*func)(entry, data); /* notify the next entry */
193  tcmi_ctlfs_entry_put(entry);
194  /* mdbg(INFO4, "putting %s", tcmi_ctlfs_entry_name(entry)); */
195  }
196  /* decrement the root's reference counter too */
197  tcmi_ctlfs_entry_put(parent);
198  /* mdbg(INFO4, "putting %s", tcmi_ctlfs_entry_name(parent)); */
199 }

231  * <<public>> Fills in the entry's pathname into the specified
232  * buffer. The pathname is being filled in reverse order starting
233  * from the end of the buffer.
234  *
235  * @param *self - pointer to this symlink instance
236  * @param *buffer - buffer where store the pathname
237  * @param length - size of the buffer allocated for the pathname - this
238  * is can be > pathname length, as the user might want to add
239  * some prefix to the pathname (e.g. ../'s)
240  */
241 void tcmi_ctlfs_entry_fill_path(struct tcmi_ctlfs_entry *self, char *buffer, int length)
242 {
243  struct {
244  char *buffer;
245  int length;
246  } buf_data;
247 
248  buf_data.buffer = buffer;
249  /* account for terminating \0 */
250  buf_data.length = --length;
251 
252  tcmi_ctlfs_entry_traverse_pred(self, tcmi_ctlfs_add_path, &buf_data);
253 }

287 /**
288  * <<private>> Helper callback method when filling the entry's
289  * pathname The entry fills in its name and the '/' as prefix to the
290  * current end of the buffer.
291  *
292  * @param *self - pointer to this instance
293  * @param *buf_data - buffer data that contain pointer to the buffer and
294  * its current length.
295  */
296 static void tcmi_ctlfs_add_path(struct tcmi_ctlfs_entry *self, void *buf_data)
297 {
298  struct {
299  char *buffer;
300  int length;
301  } *buf = buf_data;
302 
303  int cur = strlen(tcmi_ctlfs_entry_name(self));
304  mdbg(INFO4, "Adding.. %s, buffer length=%d", tcmi_ctlfs_entry_name(self), buf->length);
305 
306  /* back up enough to print entry name with '/' prefix */
307  buf->length -= cur;
308  strncpy(buf->buffer + buf->length, tcmi_ctlfs_entry_name(self), cur);
309  *(buf->buffer + --(buf->length)) = '/';
310 
311  mdbg(INFO4, "New buffer length=%d", buf->length);
312 }

337 /**
338  * <<private>> A method for inode allocation. Currently the uid
339  * and gid is assigned based on the current's fsuid and fsgid. Block
340  * size is not relevant, set to page size. If the user doesn't
341  * specify any inode or file operations, the default(empty operations,
342  * but not NULL) assigned by the VFS will be kept.
343  *
344  * If the user specified a parent directory inode, it's access rights
345  * are investigated and inherited by the new inode as follows:
346  * - if the parent has the SGID bit set, parent's gid is assigned to
347  * the new inode.
348  * - a parent SGID bit is set on the new inode, if the new inode is a
349  * directory
350  *
351  *
352  * @param *dir - inode of the parent directory for the new inode
353  * @param *sb - super block of the filesystem where the inode is
354  * to be allocated
355  * @param *mode - file type and access rights for the inode
356  * @param *i_ops - inode operations
357  * @param *f_ops - file operations (also stored in the new inode)
358  * @return a pointer to the new inode or NULL
359  */
360 static struct inode* tcmi_ctlfs_get_inode(struct inode *dir,
361  struct super_block *sb,
362  mode_t mode,
363  const struct inode_operations *i_ops,
364  const struct file_operations *f_ops)
365 {
366  struct inode *inode = new_inode(sb);
367 
368  if (inode) {
369  inode->i_mode = mode;
370  inode->i_uid = current_fsuid();
371  inode->i_gid = current_fsgid();
372  inode->i_blocks = 0;
373  inode->i_op = (i_ops ? i_ops : inode->i_op);
374  inode->i_fop = (f_ops ? f_ops : inode->i_fop);
375  inode->i_atime = inode->i_mtime = inode->i_ctime = CURRENT_TIME;
376  mdbg(INFO4, "New inode allocated, inode number = %ld", inode->i_ino);
377  /* check parent inode's(if any) access rights */
378  if (dir && (dir->i_mode & S_ISGID)) {
379  inode->i_gid = dir->i_gid;
380  if (S_ISDIR(mode))
381  inode->i_mode |= S_ISGID;
382  }
383 
384  }
385  else
386  minfo(ERR3, "Failed to allocate a new inode");
387 
388  return inode;
389 }

391 /**
392  * <<private>> The new dentry is
393  * allocated based on the parent's dentry and the current name of the
394  * ctlfs entry. The dentry is associated with the specified inode. The
395  * dentry is assigned our custom operations vector, so that we are
396  * informed about dentry destruction.
397  *
398  * @param *self - pointer to this instance
399  * @param *parent - pointer to the parent entry - has to be a directory
400  * @param *inode - pointer to the inode, that is to be associated with
401  * the new dentry.
402  * @return 0 upon success.
403  */
404 static int tcmi_ctlfs_entry_alloc_dentry(struct tcmi_ctlfs_entry *self,
405  struct tcmi_ctlfs_entry *parent,
406  struct inode *inode)
407 {
408  struct dentry *dentry;
409  int error = 1;
410  /* try allocating the dentry */
411  if ((dentry = d_alloc_name(parent->dentry, self->name))) {
412  d_add(dentry, inode); /* enters the dentry into hash queues */
413  dentry->d_fsdata = (void *)self;
414  dentry->d_op = &tcmi_ctlfs_dentry_ops;
415  self->dentry = dentry;
416  /* dget(dentry); Extra count - pin the dentry in core - not needed */
417  error = 0;
418  }
419  else
420  mdbg(ERR3, "Failed to allocate a new dentry for %s", self->name);
421  return error;
422 }

424 /**
425  * <<private>> Checks whether a directory entry contains entry of
426  * the specified name
427  *
428  * @param *self - pointer to this instance
429  * @param *name - pointer to the name to be checked
430  * @return 1 - if the entry of the specified name exists
431  */
432 static int tcmi_ctlfs_entry_exists(struct tcmi_ctlfs_entry *self, const char *name)
433 {
434  struct dentry *dentry;
435  struct qstr q;
436  int exists = 0;
437 
438  q.name = name;
439  q.len = strlen(name);
440  q.hash = full_name_hash(q.name, q.len);
441  dentry = d_lookup(self->dentry, &q);
442  /* dentry already exists, decrement its reference counter since lookup
443  incremented it */
444  if (dentry) {
445  dput(dentry);
446  exists = 1;
447  }
448  return exists;
449 }

464  return 1;

467 /**
468  * <<private>> This class method is a custom implementation of a
469  * VFS release_dentry operation. The release_dentry is called by VFS
470  * when releasing the dentry associated with a particular
471  * tcmi_ctlfs_entry object. This happens when the dentry's reference
472  * counter reaches 0.
473  *
474  * Unlike previous VFS method, we are allowed to sleep as the dcache
475  * lock and dentry lock are not held anymore (see fs/dcache.c)
476  *
477  * @param *dentry - pointer to the dentry that is to be deleted
478  * @return - always 1 since we don't want the VFS to put the dentry
479  * into the dentry_unused. This way the dentry will be unhashed and
480  * removed right away
481  */
482 static void tcmi_ctlfs_dentry_release(struct dentry *dentry)
483 {
484  /* get the instance first */
485  struct tcmi_ctlfs_entry *self = TCMI_CTLFS_DENTRY_TO_ENTRY(dentry);
486  if (self)
487  tcmi_ctlfs_entry_release(self);
488  else
489  minfo(ERR2, "VFS issued dentry_release on a dentry without tcmi_ctlfs_instance!!");
490 }
