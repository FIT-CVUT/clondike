43 /** @defgroup tcmi_ctlfs_entry_class tcmi_ctlfs_entry class
44  *
45  * @ingroup tcmi_ctlfs_class
46  *
47  * This class represents a generic entry in the tcmi control file system.
48  * No instances of this class can be made. The goal is to provide generic
49  * functionality that can be used by specific file system entries:
50  * - Initialization/deletion/freeing of instances.
51  * - Instance reference counting - this is needed as we expect
52  * multiple threads of execution to access one entry at a time. This
53  * ensures no memory leaks will occur. The reference counting is not
54  * implemented from scratch. Each entry is associated with a VFS
55  * dentry. These are VFS objects that already have a reference
56  * counter. Each instance registers with VFS and receives a
57  * notification when the reference counter reaches 0
58  * (tcmi_ctlfs_dentry_release()) and the associated dentry is to be
59  * freed. This approach provides a safe access to the entry instances
60  * from multiple threads of execution without writing an extra
61  * reference couting code.
62  * - entry location methods - a generic method that iterates through
63  * all predecessors of an entry is provided. This is used when
64  * determining the depth of the entry in the filesystem tree.
65  *
66  *@{
67  */

157 /**
158  * <<public>> Checks whether a given entry is the root of the TCMI
159  * ctlfs tree. This is when the d_parent of the dentry of this
160  * tcmi_ctlfs_entry points back to the dentry
161  *
162  * @param *self - pointer to this instance
163  * @return - 1 if it is root
164  */
165 static inline int tcmi_ctlfs_entry_isroot(struct tcmi_ctlfs_entry *self)
166 {
167  struct dentry *dentry = self->dentry;
168  return (dentry == dentry->d_parent);
169 }

172 /**
173  * <<public>> Instance accessor, Calls instance custom get method
174  * and increments the reference counter. The user is now guaranteed,
175  * that the instance stays in memory while he is using it.
176  *
177  * @param *self - pointer to this instance
178  * @return tcmi_ctlfs_entry instance
179  */
180 static inline struct tcmi_ctlfs_entry* tcmi_ctlfs_entry_get(struct tcmi_ctlfs_entry *self)
181 {
182  if (self) {
183  mdbg(INFO4, "Incrementing ref. count of '%s'(%p)",
184  tcmi_ctlfs_entry_name(self), self);
185  dget(self->dentry);
186  }
187  return self;
188 }

202 /**
203  * <<public>> Calls the instance custom put method(if any) and
204  * Decrements the reference counter of the dentry that the instance
205  * uses to carry it through out the file system. If the reference
206  * count reaches 0, the dentry will be released from the dentry cache
207  * and the instance of tcmi_ctlfs_entry will be freed. (VFS calls the
208  * d_delete() method of the dentry that we have customized @see
209  * tcmi_ctlfs_dentry_delete and d_release() that is also customized in
210  * tcmi_ctlfs_dentry_release())
211  *
212  *@param *self - pointer to this instance
213  */
214 static inline void tcmi_ctlfs_entry_put(struct tcmi_ctlfs_entry *self)
215 {
216  if (self) {
217  mdbg(INFO4, "Decrementing ref. count of '%s'(%p) (dentry: %p). Count before dec: %d",
218  tcmi_ctlfs_entry_name(self), self, self->dentry, atomic_read(&self->dentry->d_count));
219  dput(self->dentry);
220  }
221 }
