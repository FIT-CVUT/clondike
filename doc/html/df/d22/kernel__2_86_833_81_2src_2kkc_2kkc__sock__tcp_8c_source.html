51 /**
52  * <<private>> Creates an empty TCP socket. The BSD socket is
53  * created for a stream connection using TCP protocol. In addition,
54  * socket reuse option is set. Eventhough a regular TCP socket
55  * instance is created it is casted to kkc_arch_obj_t, which is a
56  * generic object supported by kkc_arch. The user of the architecture
57  * class instance always knows, what instance to expect, so he/she can
58  * cast it to kkc_sock without problems. The reason, why we cannot
59  * return directly kkc_sock, is that it would create a circular
60  * dependency between \link kkc_arch_class kkc_arch \endlink and \link
61  * kkc_sock_class kkc_sock \endlink.
62  *
63  *
64  * @return architecture object instance or NULL
65  */
66 static kkc_arch_obj_t kkc_sock_tcp_new(void)
67 {
68 
69  struct kkc_sock_tcp *sock;
70  mdbg(INFO4, "Creating a new TCP socket..");
71 
72  /* Allocate the instance */
73  if (!(sock = KKC_SOCK_TCP(kmalloc(sizeof(struct kkc_sock_tcp), GFP_KERNEL)))) {
74  mdbg(ERR4, "Can't allocate memory for TCP socket");
75  goto exit0;
76  }
77  /* Initialize the generic socket */
78  if (kkc_sock_init(KKC_SOCK(sock), kkc_arch_get(&KKC_ARCH(tcp)), &tcp_sock_ops)) {
79  mdbg(ERR4, "Error initializing TCP socket");
80  goto exit1;
81  }
82 
83  mdbg(INFO4, "Creating BSD socket");
84  if (sock_create_kern(PF_INET, SOCK_STREAM, IPPROTO_TCP, &sock->sock)) {
85  mdbg(ERR4, "Cannot create socket");
86  goto exit1;
87  }
88 
89  // TODO: At the moment, we set this on all TCP connections, but in a future we may want to use this selectively
90  kkc_sock_tcp_disable_nagle(KKC_SOCK(sock));
91  kkc_sock_tcp_enable_quickack(KKC_SOCK(sock));
92  kkc_sock_enable_keepalive(KKC_SOCK(sock));
93 
94  /* setup reuse option - this is how it was in original KKC, why? */
95  sock->sock->sk->sk_reuse = 1;
96  sock->read_callback = NULL;
97  sock->sock->sk->sk_user_data = sock; // Back reference to kkc socket
98  return KKC_ARCH_OBJ(sock);
99 
100  /* error handling */
101  exit1:
102  kfree(sock);
103  exit0:
104  return NULL;
105 
106 }

178 /**
179  * <<private>> Creates a new socket by accepting the incoming connection.
180  * After instantiating a new socket, a TCP socket specific method is
181  * called to accept a connection. This will setup the rest of the newly
182  * created TCP socket.
183  *
184  * @param *self - pointer to this socket instance
185  * @param **new_kkc_sock - storage for the new socket created for the incoming
186  * connection.
187  * @param flags - specify blocking/nonblocking mode
188  * @return 0 upon successful connection
189  * @todo - this needs an optimization, so that a full socket doesn't get
190  * manufactured everytime. This is a performance problem when using nonblock mode
191  */
192 static int kkc_sock_tcp_accept(struct kkc_sock *self, struct kkc_sock **new_kkc_sock,
193  kkc_sock_flags_t flags)
194 {
195  int err = 0;
196  int tmp_flags;
197  struct socket *sock, *new_sock;
198  /* Create a new TCP socket that will handle the incoming connection */
199  if (!(*new_kkc_sock = KKC_SOCK(kkc_sock_tcp_new()))) {
200  mdbg(ERR4, "Failed to create a new KKC TCP socket");
201  err = -ENOMEM;
202  goto exit0;
203  }
204  /* BSD socket where we check for incoming connections */
205  sock = KKC_SOCK_TCP(self)->sock;
206  /* newly created BSD socket that handles the incoming connection */
207  new_sock = KKC_SOCK_TCP(*new_kkc_sock)->sock;
208 
209  /* compose flags for accepting */
210  tmp_flags = O_RDWR | ((flags & KKC_SOCK_NONBLOCK) != 0 ? O_NONBLOCK : 0);
211 
212  mdbg(INFO4, "Accepting incoming connection");
213  if ((err = sock->ops->accept(sock, new_sock, tmp_flags)) < 0) {
214  if ( err != -EAGAIN ) {
215  mdbg(ERR4, "Cannot accept new connection %d", err);
216  } else {
217  mdbg(INFO4, "Accept timeouted.");
218  }

220  }
221 
222  mdbg(INFO4, "Accepted a TCP connection..");
223 
224  return 0;
225  /* error handling */
226  exit1:

228  new_kkc_sock = NULL; /* user will get back NULL */

255  sock = KKC_SOCK_TCP(self)->sock;
256 
257  /* Setup message header */

259  msg.msg_namelen = 0;

263  * this, it should be default for
264  * sending??*/

266  iov.iov_base = buf;

273

278 /**
279  * <<private>> Receives requested number of bytes of data.
280  * What needs to be done:
281  * - setup a message header
282  * - setup iovec and store the buffer base address and length into it
283  * - asks the kernel to receive the message
284  *
285  * @param *self - pointer to this socket instance
286  * @param *buf - buffer where to store received data.
287  * @param buflen - length of the buffer to be sent
288  * @param flags - specify blocking or non-blocking mode.
289  * @return number of bytes actually received or error when < 0
290  */
291 static int kkc_sock_tcp_recv(struct kkc_sock *self, void *buf, int buflen,
292  kkc_sock_flags_t flags)
293 {
294  struct socket *sock = KKC_SOCK_TCP(self)->sock;
295  struct msghdr msg;
296  struct iovec iov;
297  int tmp_flags;
298  int result = 0;
299 
300  /* Setup message header */
301  msg.msg_name = NULL;
302  msg.msg_namelen = 0;
303  msg.msg_control = NULL;
304  msg.msg_controllen = 0;
305  tmp_flags = (flags & KKC_SOCK_NONBLOCK ? MSG_DONTWAIT : 0);
306 
307  iov.iov_base = buf;
308  iov.iov_len = buflen;
309 
310  result = kernel_recvmsg(sock, &msg, (struct kvec*)&iov, 1, buflen, tmp_flags);
311 
312  mdbg(INFO4, "TCP receiving buffer %p length: %d bytes, received %d", buf, buflen, result);
313 
314 
315  /* connection reset is marked as 0, so if the user requested
316  * more than 0bytes to be sent, we have to check for this and
317  * convert the error */
318  return ((result == 0) && (buflen > 0)) ? -ECONNRESET : result;
319 }

506 /**
507  * <<private>> Helper class method for extracting ip address and port
508  * number from string argument.
509  *
510  * @param *addr - string with the address in the form a.b.c.d:p
511  * @param *sin - output parameter that will have ip address and
512  * port filled out upon success
513  * @return 0 if a valid IP address and port have been extracted
514  */
515 static int kkc_sock_tcp_extract_addr(char *addr, struct sockaddr_in *sin)
516 {
517  char *colon_ptr;
518  int colon_pos;
519  char tmp_ip[16];
520  unsigned long int tmp_port;
521  char *dummy;
522  int err = 0;
523 
524  /* find colon in address */
525  if ((colon_ptr = strchr(addr, ':')) == NULL) {
526  mdbg(ERR4, "Invalid IP address syntax");
527  err = -EINVAL;
528  goto leave;
529  }
530 
531  /*
532  * Extract ip address
533  */
534  colon_pos = colon_ptr - addr;
535  if (colon_pos > 15) {
536  mdbg(ERR4, "Invalid IP address syntax");
537  err = -EINVAL;
538  goto leave;
539  }
540  strncpy(tmp_ip, addr, colon_pos);
541  tmp_ip[colon_pos] = '\0';
542  mdbg(INFO4, "Extracted ip address is '%s'", tmp_ip);
543 
544  /* extract port */
545  tmp_port = simple_strtoul(colon_ptr + 1, &dummy, 10);
546  mdbg(INFO4, "Extracted port number is '%lu'", tmp_port);
547 
548  /* fill results */
549 
550  sin->sin_addr.s_addr = in_aton(tmp_ip);
551 
552  sin->sin_port = htons((unsigned short int)tmp_port);
553  sin->sin_family = AF_INET;
554 
555 leave:
556  return err;
557 }

581 /** Method used for enabling of keep alive on socket and setting of aggressive timing of timeout detection */
582 int kkc_sock_enable_keepalive(struct kkc_sock* self) {
583  struct kkc_sock_tcp* kkc_tcp = KKC_SOCK_TCP(self);
584  struct socket* socket = kkc_tcp->sock;
585  int val = 1;
586  int ret;
587 
588  // Note: must be done via those calls, direct setting does not update timer values!
589 
590  val = 10; // Start tracking timeout after 10 seconds
591  ret = kernel_setsockopt(socket, SOL_TCP, TCP_KEEPIDLE, (char __user *) &val, sizeof(val));
592  if ( ret ) minfo(ERR4, "Failed to set keep idle flag");
593 
594  val = 1; // Check tracking message every second after first timeout
595  ret = kernel_setsockopt(socket, SOL_TCP, TCP_KEEPINTVL, (char __user *) &val, sizeof(val));
596  if ( ret ) minfo(ERR4, "Failed to set keep intl flag");
597 
598  val = 10; // After 10 missed packets, mark the connection as dead
599  ret = kernel_setsockopt(socket, SOL_TCP, TCP_KEEPCNT, (char __user *) &val, sizeof(val));
600  if ( ret ) minfo(ERR4, "Failed to set keep cnt flag");
601 
602  ret = kernel_setsockopt(socket, SOL_SOCKET, SO_KEEPALIVE,
603  (char *)&val, sizeof(val));
604  if (ret < 0) minfo(ERR4, "Failed to set keep-alive");
605 
606 
607  return ret;
608 }
