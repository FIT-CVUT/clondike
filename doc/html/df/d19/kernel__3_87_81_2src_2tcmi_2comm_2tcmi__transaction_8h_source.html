43 /** @defgroup tcmi_transaction_class tcmi_transaction class
44  *
45  * @ingroup tcmi_comm_group
46  *
47  * A transaction in communication is setup when a request message
48  * expects a certain response from the receiving party. This class
49  * provides the functionality to create, start and complete such
50  * transaction.
51  *
52  * An example: A PEN node wants to register with a CCN. It will send
53  * a registration request towards the CCN. A new transaction is
54  * started on the PEN side by the thread creating the initial
55  * message. The transaction is assigned following attributes:
56  * - transaction ID
57  * - response message ID
58  * - response timeout
59  * - flags -
60  * - anonymous - setting an anonymous flag means that the user doesn't
61  * wait for the response synchronously. This flag is checked by
62  * tcmi_msg.c::tcmi_msg_deliver() - see for details.
63  * - expired - the timer upon timout marks the transaction
64  * expired and sets it to aborted state. This flag is used
65  * when checking whether the timer went off or not (needed for
66  * transaction destruction).
67  *
68  * The CCN receives the registration request performs some
69  * authentication tasks and sends a reply back. The response message
70  * will carry the transaction ID. On PEN, the message is identified
71  * based on the transaction ID and successfully received. The message
72  * then completes the transaction, which in turn wakes up the thread
73  * waiting for the message. Message delivery is closely coupled to
74  * transactions. See tcmi_msg.c::tcmi_msg_deliver().
75  *
76  * There are few stages in transaction lifetime:
77  * - INIT - this is the phase when the transaction is instantiated
78  * and inserted into a transaction slot vector, but not active yet
79  * - RUNNING - transaction is activated and the timout timer
80  * is running.
81  * - COMPLETE - transaction has been successfully finished, the context
82  * data is ready to be picked up by the transaction owner
83  * - ABORTED - if the transaction times out, the timer handler marks
84  * it aborted. The transaction owner is supposed to discard such transaction
85  *
86  * @{
87  */

303 /**
304  * <<public>> Waits on a transaction till it expires or is
305  * completed.
306  *
307  * @param *self - pointer to this transaction instance
308  * @return transaction status
309  */
310 static inline int tcmi_transaction_wait(struct tcmi_transaction *self)
311 {
312  int ret;
313  /* Sleep until the transaction times out or completes (response arrival) */
314  wait_event(self->wq, (tcmi_transaction_is_aborted(self) ||
315  tcmi_transaction_is_complete(self)));
316  ret = tcmi_transaction_state(self);
317 
318  return ret;
319 }

321 /**
322  * <<public>> Waits on a transaction till it expires or is
323  * completed, the task can be prematurely woken up by a signal.
324  *
325  * @param *self - pointer to this transaction instance
326  * @return transaction status or -ERESTARTSYS on signal arrival
327  */
328 static inline int tcmi_transaction_wait_interruptible(struct tcmi_transaction *self)
329 {
330  int ret = 0;
331  /* Sleep until the transaction times out or completes (response arrival) */
332  ret = wait_event_interruptible(self->wq, (tcmi_transaction_is_aborted(self) ||
333  tcmi_transaction_is_complete(self)));
334  if (ret < 0)
335  mdbg(INFO3, "Signal arrived");
336  else
337  ret = tcmi_transaction_state(self);
338  return ret;
339 }

375 /**
376  * <<private>> Computes the hash value of the transaction ID.
377  *
378  * The hash is computed as: ((trans_id >> 16) ^ trans_id) & hashmask
379  * @param trans_id - transaction ID to be hashed
380  * @param hashmask - hashmask limiting the maximum hash value.
381  */
382 static inline u_int tcmi_transaction_hash(u_int32_t trans_id, u_int hashmask)
383 {
384  return (((trans_id >> 16) ^ trans_id) & hashmask);
385 }
