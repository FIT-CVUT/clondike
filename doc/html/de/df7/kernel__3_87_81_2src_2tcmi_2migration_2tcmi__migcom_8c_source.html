131 /**
132  * <<public>> Migrates a task from a CCN to a PEN.
133  * This requires:
134  * - instantiating a new shadow task - requires migration manager ID,
135  * communication socket, migproc directory(where the shadow will
136  * present its information) and root directory of the migration
137  * manager(for a symbolic link)
138  * - submitting emigration method
139  * - attaching the shadow task to the target process that is
140  * to be migrated
141  * - forcing the target process into migration mode
142  * - checking if the process has properly picked up the shadow task
143  *
144  * This method exclusively uses preemptive process migration using
145  * physical checkpoint image.
146  *
147  * @param pid - task that is to be migrated
148  * @param *migman - migration manager that will provide the communication
149  * channel for the migrated task.
150  * @return 0 upon success;
151  */
152 int tcmi_migcom_emigrate_ccn_ppm_p(pid_t pid, struct tcmi_migman *migman)
153 {
154  int err = 0;
155  struct tcmi_task *shadow;
156 
157  /* create a new PPM shadow task for physical ckpt image */
158  if (!(shadow =
159  tcmi_shadowtask_new(pid, migman,
160  tcmi_migman_sock(migman),
161  tcmi_migman_migproc_dir(migman),
162  tcmi_migman_root(migman)))) {
163  minfo(ERR3, "Error creating a shadow task");
164  goto exit0;
165  }
166 
167  /* submit the emigrate method */
168  mdbg(INFO3, "Submitting emigrate_ppm_p");
169  tcmi_task_submit_method(shadow, tcmi_task_emigrate_ppm_p, NULL, 0);
170 
171  /* attaches the shadow to its thread. */
172  if (tcmi_taskhelper_attach(shadow, tcmi_migcom_mig_mode_handler) < 0) {
173  mdbg(ERR3, "Failed attaching shadow PID=%d to its thread",
174  tcmi_task_local_pid(shadow));
175  goto exit1;
176  }
177 
178  /* switch the task to migration mode */
179  tcmi_taskhelper_enter_mig_mode(shadow);
180 
181  /* wait for pickup */
182  if (tcmi_taskhelper_wait_for_pick_up_timeout(shadow, 2*HZ) < 0) {
183  mdbg(INFO1, "Shadow not picked up: %p", shadow);
184  director_emigration_failed(pid);
185  } else {
186  mdbg(INFO1, "Shadow successfully picked up: %p", shadow);
187  }
188 
189  /* release the shadow instance reference (Reference is now held by the associated task_struct - it got it in attach method)*/
190  tcmi_task_put(shadow);
191 
192  return 0;
193 
194  /* error handling */
195  exit1:
196  tcmi_task_put(shadow);
197  exit0:
198  return err;
199

271 /**
272  * <<public>> Immigrates a task - handles accepting the task on
273  * the PEN side. This method immigrates a new task based on an
274  * emigration message that a PEN has received as follows:
275  *
276  * - creates a new kernel thread that will wait until the new guest
277  * process is instantiated (waits with timeout on guest_ready
278  * completion)
279  * - instantiates a new guest task.
280  * - delivers the migration message to it and makes a necessary setup,
281  * so that the guest task processes the message as soon as it becomes
282  * a regular migrated process
283  * - attaches the guest to its new process(thread)
284  * - announces that the guest instance is ready to the thread via completion
285  * - waits for the thread to pickup the guest and become a migrated
286  * process
287  *
288  * @param *m - a message based on which we are immigrating the task
289  * @param *migman - migration manager that will provide the communication
290  * channel for the migrated task.
291  * migration information
292  * @return 0 upon success
293  */
294 int tcmi_migcom_immigrate(struct tcmi_msg *m, struct tcmi_migman *migman)
295 {
296  int err = -EINVAL;
297  int pid;
298  int wait_for_msg = 0;
299 
300  struct fs_mounter* mounter;
301 
302  struct immigrate_startup_params startup_params;
303  struct tcmi_task *guest;
304  struct tcmi_penmigman* penmigman = TCMI_PENMIGMAN(migman);
305  struct tcmi_p_emigrate_msg* msg = TCMI_P_EMIGRATE_MSG(m);
306  int accept = 0;
307 
308 
309  int call_res = director_immigration_request(tcmi_migman_slot_index(migman), tcmi_p_emigrate_msg_euid(msg), tcmi_p_emigrate_msg_exec_name(msg), &accept);
310  if ( call_res == 0 ) {
311  if ( !accept ) {
312  mdbg(INFO2, "Immigration rejected by director.");
313  // Rejected immigration
314  return -EINVAL;
315  }
316  }
317 
318 
319  init_completion(&startup_params.guest_ready);
320  init_completion(&startup_params.fs_ready);
321  startup_params.mount_params = tcmi_penmigman_get_mount_params(penmigman);
322  startup_params.sock = tcmi_migman_sock(migman);
323  startup_params.fsuid = tcmi_p_emigrate_msg_fsuid(msg);
324  startup_params.fsgid = tcmi_p_emigrate_msg_fsgid(msg);
325 
326  // Global part of mount
327  mounter = get_new_mounter(startup_params.mount_params);
328  if ( mounter && mounter->global_mount ) {
329  /* Some file system mounter is registered => perform the mount */
330  mdbg(INFO2, "Doing global mount");
331  err = mounter->global_mount(mounter, startup_params.fsuid, startup_params.fsgid);
332  if ( err ) {
333  free_mounter(mounter);
334  goto exit0;
335  }
336  }
337  free_mounter(mounter);
338 
339  /* new thread is started with a private namespace so that the mounts in that thread are not global */
340  if ((pid = kernel_thread(tcmi_migcom_migrated_task, &startup_params, CLONE_NEWNS)) < 0) {
341  mdbg(ERR3, "Cannot start kernel thread for guest process %d", pid);
342  goto exit0;
343  }
344 
345  /* before we proceed with restarting, we have to wait till the file system mount is ready (if it is performed) */
346  if ((err = wait_for_completion_interruptible(&startup_params.fs_ready))) {
347  minfo(ERR3, "Received signal fs not ready");
348  goto exit0;
349  }
350 
351  if (!(guest = tcmi_guesttask_new(pid, migman,
352  tcmi_migman_sock(migman),
353  tcmi_migman_migproc_dir(migman),
354  tcmi_migman_root(migman)))) {
355  mdbg(ERR3, "Error creating a guest task");
356  goto exit0;
357  }
358 
359  /* have the guest task deliver the message to itself, extra message reference is for the guest.*/
360  if (tcmi_task_deliver_msg(guest, tcmi_msg_get(m)) < 0) {
361  mdbg(ERR3, "Error delivering the message to the guest process");
362  tcmi_msg_put(m);
363  }
364  /* The message will be processed first. Submitted version does not wait for other messages,
365  so that it does not block potential processing of other methods later. */
366  tcmi_task_submit_method(guest, tcmi_task_process_msg, &wait_for_msg, sizeof(int));
367 
368  /* attaches the guest to its thread. */
369  if (tcmi_taskhelper_attach(guest, tcmi_migcom_mig_mode_handler) < 0) {
370  minfo(ERR3, "Failed attaching guest PID=%d to its thread",
371  tcmi_task_local_pid(guest));
372  goto exit1;
373  }
374 
375  /* signal the new thread that the guest is ready. */
376  complete(&startup_params.guest_ready);
377 
378  /* wait for pickup */
379  if (tcmi_taskhelper_wait_for_pick_up_timeout(guest, 2*HZ) < 0) {
380  mdbg(ERR1, "Guest not picked up: %p!!!!", guest);
381  goto exit1;
382  }
383  mdbg(INFO2, "Guest successfully picked up: %p", guest);
384  director_immigration_confirmed(tcmi_migman_slot_index(migman), tcmi_p_emigrate_msg_euid(msg), tcmi_p_emigrate_msg_exec_name(msg), pid, tcmi_p_emigrate_msg_reply_pid(msg));
385  tcmi_task_put(guest);
386  return 0;
387 
388  /* error handling */
389  exit1:
390  tcmi_task_put(guest);
391  exit0:
392  return err;
393 
394 }

424 /**
425  * <<private>> This method represents a kernel thread that will
426  * eventually become the 'immigrated' task. It does following:
427  *
428  * - tries to mount a distributed filesystem, if required (note, that this must be done by this thread as we are mounting into its private namespace)
429  * - mounts a proxyfs into a private namespace
430  * - waits until signalled, that the guest is ready
431  * - runs its migration mode handler (this picks up the task).
432  *
433  * @param *data - pointer to startup parameters
434  * @return shouldn't return as the execve will overlay current
435  * execution image.
436  * @todo process identity should be set according to the
437  * the user configuration (tcmi_confman)
438  */
439 static int tcmi_migcom_migrated_task(void *data)
440 {
441  int err = -EINVAL;
442  struct immigrate_startup_params* startup_params = (struct immigrate_startup_params*) data;
443  struct completion *guest_ready = &startup_params->guest_ready;
444  struct completion *fs_ready = &startup_params->fs_ready;
445  struct fs_mounter* mounter;
446  struct cred* new_cred;
447 
448  mdbg(INFO2, "New potential guest thread started. Pid: %d", current->pid);
449 
450  new_cred = prepare_creds();
451  if ( !new_cred )
452  return -ENOMEM;
453 
454  /* set process credentials */
455  new_cred->uid = new_cred->euid = new_cred->suid = new_cred->fsuid =
456  new_cred->gid = new_cred->egid = new_cred->sgid = new_cred->fsgid = 999;
457 
458  commit_creds(new_cred);
459 
460 
461 // current_uid() = current_euid() = current_suid() = current_fsuid() =
462 // current_gid() = current_egid() = current_sgid() =
463 // current_fsgid() = 999;
464  /* This part simulates when the process didn't pickup
465  the task on time
466  set_current_state(TASK_INTERRUPTIBLE);
467  schedule_timeout(5*HZ);
468  minfo(INFO1, "Task has woken up %p", current);
469  */
470 
471  // TODO: Close ALL fs.. are there some others? ;)
472  sys_close(0);
473  sys_close(1);
474  sys_close(2);
475 
476  mdbg(INFO2, "Creating mounter: %s", startup_params->mount_params->mount_type);
477  minfo(ERR1, "Creating mounter: %s", startup_params->mount_params->mount_type);
478  mounter = get_new_mounter(startup_params->mount_params);
479  mdbg(INFO2, "Created mounter: %p", mounter);
480  minfo(ERR1, "Created mounter: %p", mounter);
481 
482  if ( mounter ) {
483  /* Some file system mounter is registered => perform the mount */
484  err = mounter->private_mount(mounter, startup_params->fsuid, startup_params->fsgid);
485  if ( !err )
486  mdbg(INFO2, "Private Mount succeeded");
487  free_mounter(mounter);
488  mdbg(INFO2, "Freed mounter");
489  minfo(ERR1, "Freed mounter");
490 
491  if ( err ) {
492  /* We must notify anyway to prevent blocking */
493  complete(fs_ready);
494  goto exit0;
495  }
496  }
497  minfo(ERR1, "BEFORE mount_proxyfs");
498  err = mount_proxyfs(startup_params->sock);
499  if ( !err )
500  err = mount_local_procfs(startup_params->sock);
501 
502  if ( err ) {
503  // We must notify anyway to prevent blocking
504  complete(fs_ready);
505  goto exit0;
506  }
507 
508  /* Notify about file system mount finished */
509  complete(fs_ready);
510  minfo(ERR1, "BEFORE Starting new thread for migrating task..");
511  mdbg(INFO2, "Starting new thread for migrating task..");
512  if ((err = wait_for_completion_interruptible(guest_ready))) {
513  minfo(INFO1, "Received signal guest not ready");
514  goto exit0;
515  }
516  mdbg(INFO3, "Guest task is ready by now..");
517 
518  /* migration mode handler picks up the task */
519  if (tcmi_taskhelper_run_mig_mode_handler() < 0) {
520  mdbg(ERR3, "Failed to run migration mode handler");
521  goto exit0;
522  }
523 
524  return 0;
525 
526  /* error handling */
527  exit0:
528 
529  return err;
530 }

575 
576  mdbg(INFO3, "Forking shadow process => Making a new child shadow as well.");
577 
578  if (!(shadow =
579  tcmi_shadowtask_new(child->pid, migman,
580  tcmi_migman_sock(migman),
581  tcmi_migman_migproc_dir(migman),
582  tcmi_migman_root(migman)))) {

585  }

587  // Submit process message so that the task is started in a main waiting loop
588  // First expected message to get is either about successful start, or about start failure
589  tcmi_task_submit_method(shadow, tcmi_task_process_msg, NULL, 0);

592  if (tcmi_taskhelper_attach_exclusive(child,shadow, tcmi_migcom_mig_mode_handler) < 0) {
593  mdbg(ERR3, "Failed attaching shadow PID=%d to its thread",
594  tcmi_task_local_pid(shadow));
595  goto exit1;
596  }
597 
598  /* release the the instance reference (Reference is now held by the associated task_struct - it got it in attach method)*/
599  tcmi_task_put(shadow);
600 
601 
602  /* switch the task to migration mode */
603  tcmi_taskhelper_enter_mig_mode_exclusive(child, shadow);
604 
605  return 0;
606 exit1:
607  tcmi_task_put(shadow);
608 exit0:
609  return -EINVAL;

613 /**
614  * <<private>> This handler is invoked when the sig_unused hook
615  * finds a non-NULL entry in the clondike record in the task_struct.
616  *
617  * What needs to be done:
618  * - check whether the process in migration mode has a valid TCMI task
619  * - launch the method pump
620  * - based on its return status:
621  * -# let's the task return to user mode
622  * -# detaches the TCMI task from the process and resumes user mode
623  * -# detaches the TCMI task from the process and kills the process
624  */
625 static void tcmi_migcom_mig_mode_handler(void)
626 {
627  struct tcmi_task *tmp, *task;
628  int res;
629  long exit_code;
630 
631  mdbg(INFO3, "Migration mode handler for task '%s' stack rp(%p) executing",
632  current->comm, &res);
633 
634 
635  if (!(task = tcmi_taskhelper_sanity_check())) {
636  mdbg(ERR2, "Process %p PID=%d - invalid entry to migration mode",
637  current, current->pid);
638  goto exit0;
639  }
640 
641  /* keep moving the task between migration managers */
642  while ((res = tcmi_task_process_methods(task)) ==
643  TCMI_TASK_MOVE_ME);
644  /* check the result of the last method processed */
645  switch (res) {
646  case TCMI_TASK_KEEP_PUMPING:
647  mdbg(INFO2, "KEEP PUMPING - but no methods left %d", res);
648  break;
649  case TCMI_TASK_LET_ME_GO:
650  mdbg(INFO2, "LET ME GO - request %d", res);
651  break;
652  /* no need for special handling a fall through causes
653  * do_exit which invokes the migration handler again. */
654  case TCMI_TASK_EXECVE_FAILED_KILL_ME:
655  mdbg(INFO2, "EXECVE_FAILED KILL ME - request %d", res);
656  break;
657  case TCMI_TASK_KILL_ME:
658  mdbg(INFO2, "KILL ME - request %d", res);
659  tmp = tcmi_taskhelper_detach();
660  /* get the exit code prior terminating. */
661  exit_code = tcmi_task_exit_code(tmp);
662  tcmi_task_put(tmp);
663  mdbg(INFO2, "Put done, going to exit %ld", exit_code);
664  complete_and_exit(NULL, exit_code);
665  break;
666  case TCMI_TASK_REMOVE_AND_LET_ME_GO:
667  mdbg(INFO2, "REMOVE AND LET ME GO - request %d", res);
668  tcmi_task_put(tcmi_taskhelper_detach());
669  break;
670  default:
671  mdbg(ERR2, "Unexpected result %d.", res);
672  break;
673  }
674  /* error handling */
675  exit0:
676  return;
677 }
