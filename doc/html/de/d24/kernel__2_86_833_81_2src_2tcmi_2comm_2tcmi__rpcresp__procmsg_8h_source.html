69 /** <<public>> Get pointer do data

97 /** <<public>> Set if and how data will be freed on last put
98  * @param *self - this message instance
99  * @param index - index of data which put action we are going to set
100  * @param action - one of #TCMI_RPCRESP_PROCMSG_DONTFREE, #TCMI_RPCRESP_PROCMSG_KFREE or #TCMI_RPCRESP_PROCMSG_VFREE
101  */
102 static inline void tcmi_rpcresp_procmsg_free_on_put(struct tcmi_rpcresp_procmsg *self, unsigned int index, int action)
103 {
104  if(index < self->nmemb)
105  self->elem_free[index] = action;
106  else
107  mdbg(ERR3, "Trying to acces beyond array boundary");
108 }
