167 static int msg_transaction_request(int msg_code, struct genl_tx* tx, struct msg_transaction_ops* ops, void* params, int interruptible) {
168  int ret;
169  void *msg_head;
170  struct sk_buff *skb;
171  int seq, director_pid;
172 
173  director_pid = is_director_connected();
174  if ( !director_pid )
175  return -EFAULT;
176 
177  skb = nlmsg_new(63000, GFP_KERNEL);
178  if (skb == NULL)
179  return -1;
180 
181  seq = get_unique_seq();
182 
183  msg_head = genlmsg_put(skb, director_pid, seq, &director_gnl_family, 0, msg_code);
184  if (msg_head == NULL) {
185  ret = -ENOMEM;
186  goto failure;
187  }
188 
189  ret = ops->create_request(skb, params);
190  if (ret < 0)
191  goto failure;
192 
193  ret = genlmsg_end(skb, msg_head);
194  if (ret < 0)
195  goto failure;
196 
197  ret = genlmsg_unicast_tx(skb, director_pid, tx, interruptible);
198  if (ret != 0)
199  goto failure;
200 
201  return ret;
202 
203 failure:
204  genlmsg_cancel(skb, msg_head);
205 
206  return ret;
207 }
