39 /** @defgroup tcmi_ctlfs_class control filesystem
40  *
41  * @ingroup tcmi_module_class
42  *
43  * This <<singleton>> class represents TCMI control filesystem.
44  * The idea behind the filesystem is very similar to what we have in
45  * sysfs. The reason why a separate filesystem has been created is
46  * that TCMI doesn't fit among regular device drivers which is the
47  * primary use of sysfs.
48  *
49  * Similarly to sysfs, we provide a mechanism that allows any object
50  * in TCMI to create an interface presented to user space. The user
51  * is able to notify the object via reading/writing a file.
52  *
53  * An entity in TCMI ctlfs is called an 'entry'. Currently there are
54  * entries that represent \link tcmi_ctlfs_file a file \endlink, \link
55  * tcmi_ctlfs_dir a directory \endlink and \link tcmi_ctlfs_symlink a
56  * symbolic link \endlink.
57  *
58  *@{
59  */
60 
61 /**
62  * A compound structure that contains all internal data of TCMI ctlfs.
63  * This is for internal use only.
64  */
65 struct tcmi_ctlfs {
66  /** forward declaration of the root directory of the TCMI control filesystem */
67  struct tcmi_ctlfs_entry *root_dir;
68  /** a semaphore that protects the vfs_mount variable against
69  * concurrent accesses. This is important when issuing
70  * tcmi_ctlfs_get_root/tcmi_ctlfs_put_root
71  */
72  struct semaphore vfs_mnt_sem;
73  /** forward declaration of the mount object of TCMI control filesystem */
74  struct vfsmount *vfs_mnt;
75  /** Filesystem super block object */
76  struct super_block *sb;
77  /** A forward declaration describing this filesystem type */
78  struct file_system_type fs_type;
79 };
