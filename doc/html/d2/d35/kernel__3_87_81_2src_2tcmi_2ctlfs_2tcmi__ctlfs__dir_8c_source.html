60 /**
61  * <<public>> Creates a new directory entry in the tcmi_ctlfs tree.
62  * The instance is:
63  * - created
64  * - initialized (delegated to the parent class)
65  * - link counts of parent and new directory need to be adjusted
66  *
67  * @param *parent - pointer to the parent entry - has to be a directory
68  * @param *mode - access rights for the inode
69  * @param namefmt - nameformat string (printf style)
70  * @param args - arguments to the format string
71  * @return pointer to the new directory or NULL
72  */
73 struct tcmi_ctlfs_entry* tcmi_ctlfs_dir_vnew(struct tcmi_ctlfs_entry *parent,
74  mode_t mode, const char namefmt[],
75  va_list args)
76 {
77  struct tcmi_ctlfs_dir *dir;
78 
79  mdbg(INFO4, "Creating new directory");
80  if (!(dir = kmalloc(sizeof(struct tcmi_ctlfs_dir), GFP_ATOMIC))) {
81  mdbg(ERR3, "Failed to allocate memory for directory");
82  goto exit0;
83  }
84  /* directory file operations are taken from fs/libfs.c */
85  if (tcmi_ctlfs_entry_init(TCMI_CTLFS_ENTRY(dir),
86  TCMI_CTLFS_ENTRY(parent),
87  S_IFDIR | mode,
88  NULL,
89  &simple_dir_inode_operations,
90  &simple_dir_operations, namefmt, args)) {
91  mdbg(ERR3, "Failed to create the directory");
92  goto exit1;
93  }
94  /* success, append the directory to its parent (account for '..') and
95  increment the link count for '.' */
96  tcmi_ctlfs_entry_inc_links(TCMI_CTLFS_ENTRY(parent));
97  tcmi_ctlfs_entry_inc_links(TCMI_CTLFS_ENTRY(dir));
98  return TCMI_CTLFS_ENTRY(dir);
99 
100  /* error handling */
101  exit1:
102  kfree(dir);
103  exit0:
104  return NULL;
105 }
