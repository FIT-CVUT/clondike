40 /**
41  * <<public>> Writes a specified memory area into the checkpoint
42  * file. Checks for the requested type - when the light version is
43  * required, it will do so only if the area is non-writable and maps a
44  * file. The actual work is then delegated to the light or heavy
45  * version of this method.
46  *
47  * @param *ckpt - checkpoint file where the area is to be stored
48  * @param *vma - the actual VM area that is being processed
49  * @param type - requested type of the checkpoint
50  * @return 0 upon success.
51  */
52 int tcmi_ckpt_vm_area_write(struct tcmi_ckpt *ckpt, struct vm_area_struct *vma,
53  tcmi_ckpt_vm_area_t type)
54 {
55  int err;
56  struct tcmi_ckpt_vm_area_hdr hdr;
57  hdr.vm_start = vma->vm_start;
58  hdr.vm_end = vma->vm_end;
59  hdr.vm_flags = vma->vm_flags;
60  hdr.vm_pgoff = vma->vm_pgoff;
61  /* light version only when it's non-writable and maps a file */
62  if (type == TCMI_CKPT_VM_AREA_LIGHT &&
63  !(vma->vm_flags & VM_WRITE) && vma->vm_file)
64  err = tcmi_ckpt_vm_area_write_l(ckpt, vma, &hdr);
65  else
66  err = tcmi_ckpt_vm_area_write_h(ckpt, vma, &hdr);
67  return err;
68 }

78  * @return 0 upon success.

90  (unsigned long long)hdr.vm_start, (unsigned long long)hdr.vm_end, (unsigned long long)hdr.vm_flags, (unsigned long long)hdr.vm_pgoff);

113 /**
114  * <<private>> Writes a specified memory area into the checkpoint
115  * file - light version. It fills out the rest of the header,
116  * extracts the mapped file pathname and stores the header, pathname
117  * size and the actual pathname into the checkpoint.
118  *
119  * @param *ckpt - checkpoint file where the area is to be stored
120  * @param *vma - the actual VM area that is being processed
121  * @param *hdr - partially filled VM area header
122  * @return 0 upon success.
123  */
124 static int tcmi_ckpt_vm_area_write_l(struct tcmi_ckpt *ckpt,
125  struct vm_area_struct *vma,
126  struct tcmi_ckpt_vm_area_hdr *hdr)
127 {
128  /* page for the filepathname */
129  unsigned long page;
130  char *pathname;
131 
132  /* finish the header */
133  hdr->type = TCMI_CKPT_VM_AREA_LIGHT;
134 
135  /* resolve the path name. */
136  if (!(page = __get_free_page(GFP_KERNEL))) {
137  mdbg(ERR3, "Can't allocate page for file pathname!");
138  goto exit0;
139  }
140  if (IS_ERR(pathname = d_path(&vma->vm_file->f_path,
141  (char*)page, PAGE_SIZE))) {
142  mdbg(ERR3, "Can't resolve pathname for '%s'",
143  vma->vm_file->f_dentry->d_name.name);
144  goto exit1;
145  }
146  hdr->pathname_size = strlen(pathname) + 1;
147  /* write the header and the pathname into the checkpoint */
148  if (tcmi_ckpt_write(ckpt, hdr, sizeof(*hdr)) < 0) {
149  mdbg(ERR3, "Error writing VM_area header chunk");
150  goto exit1;
151  }
152  /* write the header and the pathname into the checkpoint */
153  if (tcmi_ckpt_write(ckpt, pathname, hdr->pathname_size) < 0) {
154  mdbg(ERR3, "Error writing pathname chunk");
155  goto exit1;
156  }
157 
158  free_page(page);
159  return 0;
160 
161  /* error handling */
162  exit1:
163  free_page(page);
164  exit0:
165  return -EINVAL;
166 }

168 /**
169  * <<private>> Writes a specified memory area into the checkpoint
170  * file - heavy version. It fills out the rest of the header, sets
171  * pathname size to 0 (not used). and writes the header into the
172  * checkpoint file.
173  *
174  * It is necessary to align the file to page size. This is required
175  * otherwise we wouldn't be able to mmap pages from the file upon
176  * restoring the checkpoint.
177  *
178  * Following algorithm dumps the pages, the idea comes from a regular
179  * core dump handler (e.g. in fs/binfmt_elf.c)
180  *
181  \verbatim
182  For each page in the region do:
183  - get the page - might trigger pagefaults to load the pages
184  - map the page descriptor to a valid linear address (user pages might
185  be in high memory)
186  - if the page is zero or marked untouched, advance the offset in the
187  checkpoint only - there is no need to write 0's to disk, just create
188  the gap for them.
189  - otherwise write the entire page into the checkpoint file.
190  \endverbatim
191  *
192  * @param *ckpt - checkpoint file where the area is to be stored
193  * @param *vma - the actual VM area that is being processed
194  * @param *hdr - partially filled VM area header
195  * @return 0 upon success.
196  */
197 static int tcmi_ckpt_vm_area_write_h(struct tcmi_ckpt *ckpt,
198  struct vm_area_struct *vma,
199  struct tcmi_ckpt_vm_area_hdr *hdr)
200 {
201  /* page for the filepathname */
202  unsigned long addr;
203 
204  /* finish the header */
205  hdr->type = TCMI_CKPT_VM_AREA_HEAVY;
206  hdr->pathname_size = 0;
207  /* write the header into the checkpoint */
208  if (tcmi_ckpt_write(ckpt, hdr, sizeof(*hdr)) < 0) {
209  mdbg(ERR3, "Error writing VM area header chunk");
210  goto exit0;
211  }
212  /* align the current file position to page size boundary */
213  tcmi_ckpt_page_align(ckpt);
214  for (addr = vma->vm_start; addr < vma->vm_end; addr += PAGE_SIZE) {
215  struct page *page;
216  struct vm_area_struct *vma;
217  void *kaddr;
218  if (get_user_pages(current, current->mm, addr, 1, 0, 1,
219  &page, &vma) <= 0) {
220  mdbg(INFO4, "Skipping untouched page at %08lx", addr);
221  tcmi_ckpt_seek(ckpt, PAGE_SIZE, 1);
222  continue;
223  }
224  /*if (page == ZERO_PAGE(addr)) {
225  mdbg(INFO4, "Skipping zero page at %08lx", addr);
226  tcmi_ckpt_seek(ckpt, PAGE_SIZE, 1);
227  continue;
228  }
229  */
230  /* actual page, that needs to be written. */
231  flush_cache_page(vma, addr, page_to_pfn(page)); /* TODO: check this fix is correct */
232  kaddr = tcmi_ckpt_vm_area_kmap(page);
233  /* write the page into the checkpoint */
234  if (tcmi_ckpt_write(ckpt, kaddr, PAGE_SIZE) < 0) {
235  mdbg(ERR3, "Error writing page at %08lx", addr);
236  goto exit0;
237  }
238  tcmi_ckpt_vm_area_kunmap(page);
239  }
240  return 0;
241 
242  /* error handling */
243  exit0:
244  return -EINVAL;
245 }

249 /**
250  * <<private>> Reads a memory area from the checkpoint file -
251  * light version. Reads the pathname of the file that is to be mapped
252  * into the area.
253  * - opens the file
254  * - performs the memory mapping,
255  * - releases the file - the memory mapping retains its own reference.
256  *
257  * @param *ckpt - checkpoint file where the area is to be stored
258  * @param *hdr - VM area header
259  * @return 0 upon success.
260  */
261 
262 static int tcmi_ckpt_vm_area_read_l(struct tcmi_ckpt *ckpt,
263  struct tcmi_ckpt_vm_area_hdr *hdr)
264 {
265  /* page for the filepathname */
266  unsigned long page;
267  char *pathname = NULL;
268  struct file *file; /* file object for the vm_file */
269  unsigned long mmap_flags;
270  unsigned long addr;
271 
272  /* finish the header */
273  hdr->type = TCMI_CKPT_VM_AREA_LIGHT;
274 
275  /* resolve the path name. */
276  if (!(page = __get_free_page(GFP_KERNEL))) {
277  mdbg(ERR3, "Can't allocate page for file pathname!");
278  goto exit0;
279  }
280  pathname = (char*) page;
281  /* read the header and the pathname from the checkpoint */
282  if (tcmi_ckpt_read(ckpt, pathname, hdr->pathname_size) < 0) {
283  mdbg(ERR3, "Error reading pathname");
284  goto exit1;
285  }
286  /* convert the VM flags to MMAP flags */
287  mmap_flags = tcmi_ckpt_vm_area_to_mmap_flags(hdr->vm_flags);
288 
289  file = filp_open(pathname, O_RDONLY, 0);
290  if (IS_ERR(file)) {
291  mdbg(ERR3, "Error opening file '%s'", pathname);
292  goto exit1;
293  }
294  /* prot flags are just the lower bits extracted from vm_flags - see mman.h, mm.h */
295  addr = do_mmap_pgoff(file, hdr->vm_start, hdr->vm_end - hdr->vm_start,
296  hdr->vm_flags & (PROT_READ | PROT_EXEC| PROT_WRITE),
297  mmap_flags, hdr->vm_pgoff);
298 
299  if (addr != hdr->vm_start) {
300  mdbg(ERR3, "Error mapping file '%s' at: %lx", pathname, addr);
301  goto exit2;
302  }
303 
304  /* or shall we perform filp_close??? */
305  fput(file);
306  free_page(page);
307 
308  mdbg(INFO4, "Mapped at: 08%lx, VM_flags = %08llx, mmap flags = %08lx",
309  addr, (unsigned long long)hdr->vm_flags, mmap_flags);
310  return 0;
311 
312  /* error handling */
313  exit2:
314  fput(file);
315  exit1:
316  free_page(page);
317  exit0:
318  return -EINVAL;
319 
320 }

322 /**
323  * <<private>> Reads a memory area from the checkpoint file -
324  * heavy version.
325  * - aligns the position in the file to page boundary
326  * - computes memory mapping flags.
327  * - performs the memory mapping.
328  * There is special case that needs to be handled - if the area
329  * grows down which indicates a stack region stack setup
330  * work is delegated to tcmi_ckpt_vm_area_stack_fixup().
331  *
332  *
333  * @param *ckpt - checkpoint file where the area is to be stored
334  * @param *hdr - VM area header
335  * @return 0 upon success.
336  */
337 static int tcmi_ckpt_vm_area_read_h(struct tcmi_ckpt *ckpt,
338  struct tcmi_ckpt_vm_area_hdr *hdr)
339 {
340  unsigned long mmap_flags;
341  unsigned long addr;
342  unsigned long length;
343 
344  tcmi_ckpt_page_align(ckpt);
345 
346  length = hdr->vm_end - hdr->vm_start;
347 
348  if ( length < PAGE_SIZE ) {
349  mdbg(ERR3, "Length of the area shorter than one page. Start: %16llx End: %16llx", (unsigned long long)hdr->vm_start, (unsigned long long)hdr->vm_end);
350  goto exit0;
351  }
352 
353  /* check if stack fixup is needed */
354  if ((hdr->vm_flags & VM_GROWSDOWN) ) {
355  if (tcmi_ckpt_vm_area_stack_fixup(ckpt, hdr) < 0) {
356  mdbg(ERR3, "Failed setting up stack area");
357  goto exit0;
358  }
359 
360  length -= PAGE_SIZE;
361 
362  /* The are was fully maped by the stack fixup */
363  if ( length == 0 )
364  return 0;
365 
366  }
367 
368  /* convert the VM flags to MMAP flags */
369  mmap_flags = tcmi_ckpt_vm_area_to_mmap_flags(hdr->vm_flags);
370  /* prot flags are just the lower bits extracted from vm_flags - see mman.h, mm.h */
371  addr = tcmi_ckpt_do_mmap(ckpt, hdr->vm_start, hdr->vm_end - hdr->vm_start,
372  hdr->vm_flags & (PROT_READ | PROT_EXEC| PROT_WRITE),
373  mmap_flags);
374  if (addr != hdr->vm_start) {
375  mdbg(ERR3, "Error mapping checkpoint file at: %08lx (hdr->vm_start: %16llx", addr, (unsigned long long)hdr->vm_start);
376  goto exit0;
377  }
378  mdbg(INFO4, "Mapped checkpoint at: %08lx, VM_flags = %16llx, mmap flags = %08lx",
379  addr, (unsigned long long)hdr->vm_flags, mmap_flags);
380  return 0;
381 
382  /* error handling */
383  exit0:
384  return -EINVAL;
385 }

388 /**
389  * <<private>> This method is responsible for fixing a the stack
390  * in the area just read from the checkpoint file. The idea is the
391  * split the supplied VM area into two:
392  *
393  * - one containing the top of the stack
394  * - the other will be created by the standard mechanism.
395  *
396  * What needs to be done is to:
397  *
398  * - create an anonymous mapping with the MAP_GROWSDOWN flag, with start
399  * address the same as the VM area in the checkpoint and size of one page.
400  * - read the first page of the stack area from the checkpoint file
401  * and copy it into the newly created mapping. The current top of the
402  * stack is somewhere in this page.
403  * - adjust the VM area header read from the checkpoint file:
404  * increment vm_start address and reset the VM_GROWSDOWN flag.
405  *
406  * Why is it so complex? Stack expansion is possible only for areas
407  * marked by VM_GROWSDOWN flag. However, it is not possible to create
408  * a file memory mapping using MAP_GROWSDOWN. The memory mapping
409  * framework simply denies this as a regular user doesn't need it. To
410  * change the size of a memory mapping, all processes are encouraged
411  * to use mremap(). Further, we would have a problem when the current
412  * stack pointer leaves the memory region by the allowed limit(max
413  * 32bytes). Where would the expansion get the page? It might end up
414  * asking the file to supply the missing page. This is obviously not
415  * possible.
416  *
417  *
418  * @param *ckpt - checkpoint file where the current stack area resides.
419  * @param *hdr - VM area header
420  * @return 0 upon success.
421  */
422 static int tcmi_ckpt_vm_area_stack_fixup(struct tcmi_ckpt *ckpt,
423  struct tcmi_ckpt_vm_area_hdr *hdr)
424 {
425  /* temporary page allocated for the top of the stack */
426  unsigned long page;
427  /* address of the target mapping - must match hdr->vm_start */
428  unsigned long addr;
429  unsigned long mmap_flags;
430 
431  /* convert the VM flags to MMAP flags */
432  mmap_flags = tcmi_ckpt_vm_area_to_mmap_flags(hdr->vm_flags);
433 
434  /* prot flags are just the lower bits extracted from vm_flags - see mman.h, mm.h */
435  down_write(&current->mm->mmap_sem);
436  addr = do_mmap(NULL, hdr->vm_start, PAGE_SIZE,
437  hdr->vm_flags & (PROT_READ | PROT_EXEC| PROT_WRITE),
438  mmap_flags, 0);
439  up_write(&current->mm->mmap_sem);
440 
441  if (addr != hdr->vm_start) {
442  mdbg(ERR3, "Error creating an anonymous mapping for stack page %d", (int)addr);
443  goto exit0;
444  }
445 
446  if (!(page = __get_free_page(GFP_KERNEL))) {
447  mdbg(ERR3, "Can't allocate page for stack!");
448  goto exit0;
449  }
450  /* read the first page of the stack */
451  if (tcmi_ckpt_read(ckpt, (void*)page, PAGE_SIZE) < 0) {
452  mdbg(ERR3, "Error stack page!");
453  goto exit1;
454  }
455  mdbg(INFO4, "Copying stack page to %08lx from %08lx", addr, page);
456  if (tcmi_ckpt_vm_area_copy_page(addr, page) < 0) {
457  mdbg(ERR3, "Failed to copy stack page!!!");
458  goto exit1;
459  }
460  /* memcpy((void*)addr, (void*)page, PAGE_SIZE); */
461  mdbg(INFO4, "Copying done");
462  /* adjust the attributes */
463  hdr->vm_start += PAGE_SIZE;
464  hdr->vm_flags &= ~VM_GROWSDOWN; /* FIXME: make this unified for all archs!!!*/
465 
466  free_page(page);
467 
468  mdbg(INFO4, "Created anonymous mapping for stack at: %08lx", addr);
469  return 0;
470 
471  /* error handling */
472  exit1:
473  free_page(page);
474  exit0:
475  return -EINVAL;
476 }
