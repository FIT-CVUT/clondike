38 /**
39  * <<public>> Creates new instance.
40  * - Vector of slots is allocated and all slots are initialized.
41  * Since we want the slot to do the management of unused/used lists
42  * on their own, we pass the vector lock as well as both list heads
43  * to the init method.
44  * - reference counter is initialized to 1 - the instantiator
45  * owns this slot vector now
46  *
47  * @param size - number of slots to be allocated in the vector.
48  * @return a new slot vector or NULL
49  */
50 struct tcmi_slotvec* tcmi_slotvec_new(u_int size)
51 {
52  struct tcmi_slotvec *vec;
53  struct tcmi_slot *slots;
54  int i;
55  if (!(vec = kmalloc(sizeof(struct tcmi_slotvec), GFP_ATOMIC))) {
56  mdbg(ERR3, "Failed to allocate a slotvector");
57  goto exit0;
58  }
59  if (!(slots = kmalloc(sizeof(struct tcmi_slot) * size, GFP_ATOMIC))) {
60  mdbg(ERR3, "Failed to allocate a %d slots", size);
61  goto exit1;
62  }
63  vec->slots = slots;
64  vec->slot_count = size;
65  INIT_LIST_HEAD(&vec->unused_slots);
66  INIT_LIST_HEAD(&vec->used_slots);
67  atomic_set(&vec->ref_count, 1);
68  spin_lock_init(&vec->s_lock);
69  /* initialize all slots */
70  for (i = 0; i < size; i++) {
71  tcmi_slot_init(slots, i, &vec->unused_slots,
72  &vec->used_slots, &vec->s_lock);
73  slots++;
74  }
75  mdbg(INFO4, "Allocated %d slots (%lu bytes), memory=%p", size,
76  (unsigned long)(sizeof(struct tcmi_slot) * size), vec);
77  return vec;
78 
79  /* error handling */
80  exit1:
81  kfree(vec);
82  exit0:
83  return NULL;
84 }

86 /**
87  * <<public>> This method creates a slot vector of the size
88  * \$2^{order}\$. It is intended when using the slot vector as a
89  * hashtable with \$2^{order}\$ buckets. A hashmask is calculated as
90  * the vector size - 1.
91  *
92  * @param order - \$log_2(buckets)\$
93  * @return a new slot vector or NULL
94  */
95 struct tcmi_slotvec* tcmi_slotvec_hnew(u_int order)
96 {
97  struct tcmi_slotvec *vec;
98  if (!(vec = tcmi_slotvec_new(1 << order))) {
99  mdbg(ERR3, "Error creating a slot vector hashtable with 2^%d buckets", order);
100  goto exit0;
101  }
102  /* precompute the hashmask */
103  vec->hashmsk = vec->slot_count - 1;
104  return vec;
105  /* error handling */
106  exit0:
107  return NULL;
108 }

110 /**
111  * <<public>> Decrements the reference counter, when the count
112  * reaches 0, the instance is destroyed. This requires releasing all
113  * slots and eventually releasing the instance itself.
114  *
115  * @param *self - pointer to this vector instance
116  */
117 void tcmi_slotvec_put(struct tcmi_slotvec *self)
118 {
119  struct tcmi_slot *slots;
120  int i;
121  if (!self)
122  return;
123  mdbg(INFO3, "Putting ref. count=%d (%p)",
124  atomic_read(&self->ref_count), self);
125  if (atomic_dec_and_test(&self->ref_count)) {
126  slots = self->slots;
127  for (i = 0; i < self->slot_count; i++, slots++) {
128  if (!(tcmi_slot_empty(slots)))
129  mdbg(WARN3, "Warning - non empty slot %d", i);
130  }
131  mdbg(INFO4, "Released %d slots (%lu bytes), memory=%p",
132  self->slot_count,
133  (unsigned long)(sizeof(struct tcmi_slot) * self->slot_count), self);
134  kfree(self->slots);
135  kfree(self);
136  }

157 /**
158  * <<public>> Finds the first empty slot and reserves it. The
159  * slot is removed from the unused list, so that other threads won't
160  * consider it when searching for empty slots. The user of this method
161  * is responsible for either filling the slot with some data or
162  * releasing the slot back to unused list by means of
163  * tcmi_slot.h::tcmi_slot_move_unused()
164  *
165  * @param *self - pointer to this vector instance
166  * @return pointer to an empty slot or NULL
167  */
168 struct tcmi_slot* tcmi_slotvec_reserve_empty(struct tcmi_slotvec *self)
169 {
170  struct tcmi_slot *slot = NULL;
171  struct list_head *list;
172 
173  tcmi_slotvec_lock(self);
174  if (!list_empty(&self->unused_slots)) {
175  /* get the first slot */
176  list_for_each(list, &self->unused_slots)
177  break;
178  slot = list_entry(list, struct tcmi_slot, slot_list);
179  /* remove the slot, from the unused list */
180  list_del_init(list);
181  mdbg(INFO4, "Found empty slot index = %d", slot->index);
182  }
183  tcmi_slotvec_unlock(self);
184 
185  return slot;
186 }
