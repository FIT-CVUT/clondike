43 /** <<public>> Do ioctl
44  * @param *self - pointer to this file instance
45  * @param cmd - ioctl cmd
46  * @param arg - ioctl arg
47  *
48  * @return - return code of ioctl
49  * */
50 long proxyfs_real_file_do_ioctl(struct proxyfs_real_file *self, unsigned int cmd, unsigned long arg)
51 {
52  int error = -ENOTTY;
53  mm_segment_t old_fs;
54  old_fs = get_fs();
55 
56  if (S_ISREG(self->file->f_path.dentry->d_inode->i_mode))
57  goto out;
58 
59  if (!self->file->f_op)
60  goto out;
61 
62  set_fs(KERNEL_DS);
63 
64  if (self->file->f_op->unlocked_ioctl) {
65  error = self->file->f_op->unlocked_ioctl(self->file, cmd, arg);
66  if (error == -ENOIOCTLCMD) {
67  error = -EINVAL;
68  mdbg(ERR2, "No ioctl cmd %d for file %lu", cmd, proxyfs_file_get_file_ident(PROXYFS_FILE(self)));
69  }
70  goto out;
71  }
72  /* Commented by Jiru Rakosnik because ioctl fileoperation is not exist in kernel 3.x
73  else if (self->file->f_op->ioctl) {
74  lock_kernel();
75  error = self->file->f_op->ioctl(self->file->f_path.dentry->d_inode,
76  self->file, cmd, arg);
77  unlock_kernel();
78  }
79  */
80 out:
81  set_fs(old_fs);
82  return error;
83 }

157 {
158  struct proxyfs_msg *msg;
159  int rtn = 0;
160 
161  if( PROXYFS_FILE(self)->peer == peer ){
162  msg = proxyfs_msg_new(MSG_CLOSE_RESP, proxyfs_file_get_file_ident(PROXYFS_FILE(self)), 0, NULL );
163  if ( !msg )
164  return -ENOMEM;

167  proxyfs_file_set_peer(PROXYFS_FILE(self),NULL);
168  PROXYFS_FILE(self)->write_buf_unconfirmed = 0;

170  else{
171  mdbg(ERR3, "Close message come from peer which hasn't opened the file");
172  rtn = -1;
173  }
174 
175  return rtn;
176 }
