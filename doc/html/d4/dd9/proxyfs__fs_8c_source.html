62 /** <<private>> Fill proxyfs superblock
63  *
64  * @param sb - pointer to superblock
65  * @param data - mount options etc...
66  * @param silent -
67  *
68  * @return Zero on success
69  */
70 static int proxyfs_fs_fill_super(struct super_block *sb, void *data, int silent)
71 {
72  struct inode *rootinode;
73 
74  sb->s_blocksize = 1024;
75  sb->s_blocksize_bits = 10;
76  sb->s_magic = 0x28021985;
77  sb->s_op = &proxyfs_sops; // super block operations
78  sb->s_type = &proxyfs; // file_system_type
79  sb->s_fs_info = data; // proxyfs_client task
80 
81  rootinode = iget_locked(sb, 0xFFFFFFFF); // allocate an inode
82  rootinode->i_op = &rootinode_iops; // set the inode ops
83  //rootinode->i_fop = &rootinode_fops;
84  rootinode->i_mode = S_IFDIR | S_IRUSR | S_IXUSR;
85  rootinode->i_mtime = rootinode->i_atime = rootinode->i_ctime = CURRENT_TIME;
86 
87  if( !(sb->s_root = d_alloc_root(rootinode)) ){
88  iput(rootinode);
89  return -ENOMEM;
90  }
91 
92  unlock_new_inode(rootinode);
93 
94  return 0;
95 }

142 /** <<private>> root inode lookup
143  *
144  * @param parent_inode -
145  * @param dentry -
146  * @param nameidata - file name (Proxyfile identification number)
147  *
148  * @return pointer to dentry with filled inode or NULL on error
149  */
150 static struct dentry *proxyfs_fs_rootinode_lookup(
151  struct inode *parent_inode, struct dentry *dentry, struct nameidata *nameidata)
152 {
153  struct inode *fileinode;
154  unsigned long inode_num;
155  uint32_t file_number;
156  uint32_t owner_pid;
157 
158  memory_sanity_check("Prescan");
159 
160  mdbg(INFO3, "filename %s", dentry->d_name.name);
161 
162 
163  if ( sscanf(dentry->d_name.name, "%u-%u", &owner_pid, &file_number) != 2 )
164  goto not_found; // Invalid format
165 
166  memory_sanity_check("Postscan");
167 
168  inode_num = (owner_pid << 16) + file_number;
169 
170  mdbg(INFO3, "rootinode lookup %lx (pid: %d, file_num: %d)\n", inode_num, owner_pid, file_number);
171  fileinode = iget_locked(parent_inode->i_sb, inode_num);
172  //fileinode = iget(parent_inode->i_sb, inode_num);
173 
174  if(!fileinode){
175  mdbg(INFO3, "Cannot access proxyfs inode");
176  goto not_found; // Number isn't pid of tcmi_task
177  }
178 
179  //fileinode->i_mode = S_IFREG | S_IRUSR | S_IWUSR | S_IRGRP;
180  // TODO: Was reg, changed to char.. maybe we should decide based on original file`s mode?
181  fileinode->i_mode = S_IFCHR | S_IRUSR | S_IWUSR | S_IRGRP;
182  fileinode->i_fop = &fdinode_fops;
183  fileinode->i_op = &fdinode_iops;
184 
185  d_add(dentry, fileinode);
186 
187  if (fileinode->i_state & I_NEW) {
188  unlock_new_inode(fileinode);
189  }
190 
191  return NULL;
192 
193 not_found:
194  d_add(dentry, NULL);
195  return ERR_PTR(-EACCES);
196 }

264 /** <<private>> Write function for VFS
265  * @param *file - pointer to file we wanted to write to
266  * @param *buf - pointer to buffer in userspace with data
267  * @param count - size of data
268  * @param *offset - pointer to file offset
269  *
270  * @return Amount of bytes written or error
271  */
272 static ssize_t proxyfs_fs_fdinode_write(struct file *file, const char *buf, size_t count, loff_t *offset)
273 {
274  int length, total = 0;
275  struct proxyfs_proxy_file_t *proxyfile;
276  struct proxyfs_task *task = PROXYFS_TASK( file->f_dentry->d_inode->i_sb->s_fs_info );
277 
278  proxyfile = (struct proxyfs_proxy_file_t *) file->private_data;
279 
280  if((ssize_t) count < 0) return -EINVAL;
281 
282  mdbg(INFO2, "file %p, buf %p, count %lu, offset %llu", proxyfile, buf, (unsigned long)count, (unsigned long long)*offset);
283  mdbg(INFO2, "Content %.*s", (int)count, buf);
284 
285 write: length = proxyfs_proxy_file_write_user(proxyfile, buf + total, count - total);
286 
287  mdbg(INFO2, "file %p, written, %d", proxyfile, (int)length);
288 
289  if( length > 0 ){
290  *offset += length;
291  //proxyfs_task_wakeup( task );
292  proxyfs_task_notify_data_ready(task);
293  total += length;
294  if( total == count || file->f_flags & O_NONBLOCK )
295  return total;
296  else
297  goto write;
298  }
299  else if( length == 0 ){ // Buffer is full
300  if( file->f_flags & O_NONBLOCK ) {
301  mdbg(INFO2, "Buffer full in non-block mode -> EAGAIN");
302  return -EAGAIN;
303  }
304  else { // Block on file
305  mdbg(INFO2, "Buffer full in block mode -> wait");
306  switch( proxyfs_proxy_file_wait_interruptible(proxyfile,
307  PROXYFS_FILE_CAN_WRITE | PROXYFS_FILE_EPIPE | PROXYFS_FILE_CLOSED)){
308  case PROXYFS_FILE_EPIPE:
309  return -EPIPE; // This is seen by reading process only if it catches
310  // or ignore SIGPIPE
311  case PROXYFS_FILE_CLOSED:
312  return -EBADF;
313  case PROXYFS_FILE_CAN_WRITE:
314  goto write;
315  case 0:
316  return -ERESTARTSYS;
317  }
318  }
319  }
320  else if( length < -4 ){ // Writing to pipe, needs -length bytes.
321  if( file->f_flags & O_NONBLOCK ) {
322  mdbg(INFO2, "Returning AGAIN");
323  return -EAGAIN;
324  } else { // Block on file
325  int file_state = proxyfs_proxy_file_wait_for_space_interruptible(proxyfile, -length);
326  if ( file_state & PROXYFS_FILE_CAN_WRITE ) {
327  mdbg(INFO2, "Restarting write");
328  goto write;
329  }
330 
331  switch( file_state){
332 
333  case PROXYFS_FILE_EPIPE:
334  mdbg(INFO2, "Returning EPIPE");
335  return -EPIPE; // This is seen by reading process only if it catches
336  // or ignore SIGPIPE
337  case 0:
338  mdbg(INFO2, "Returning ERESTARTSYS");
339  return -ERESTARTSYS;
340  default:
341  mdbg(INFO2, "Unknown file state: %d", file_state);
342  }
343 
344  }
345  }
346  return -1;
347 }

349 /** <<private>> Read function for VFS
350  * @param *file - pointer to file we wanted to read from
351  * @param *buf - pointer to buffer in userspace witch will be filled with data
352  * @param count - size of buffer
353  * @param *offset - pointer to file offset
354  *
355  * @return Amount of bytes read or error
356  */
357 static ssize_t proxyfs_fs_fdinode_read(struct file *file, char *buf, size_t count, loff_t *offset)
358 {
359  size_t length;
360  struct proxyfs_proxy_file_t *proxyfile;
361 
362  proxyfile = (struct proxyfs_proxy_file_t *) file->private_data;
363 
364  if((ssize_t) count < 0) return -EINVAL;
365 
366  mdbg(INFO2, "file %p, buf %p, count %lu, offset %llu", proxyfile, buf, (unsigned long)count, (unsigned long long)*offset);
367 
368 read: length = proxyfs_proxy_file_read_user(proxyfile, buf, count);
369 
370  mdbg(INFO2, "Read content %.*s", (int)length, buf);
371 
372  mdbg(INFO2, "file %p, read, %d", proxyfile, (int)length);
373 
374  if( length > 0 ){
375  *offset += length;
376  return length;
377  }
378  else if( length == 0 ){ // Buffer is empty
379  if( proxyfs_file_get_status( PROXYFS_FILE( proxyfile ), PROXYFS_FILE_READ_EOF ) ){
380  mdbg(INFO4, "EOF");
381  // The following condition is check for a situation, where we got data + eof after the read was processed
382  // In that case, old read may return 0, but now there are data and in addition EOF is set
383  if ( proxyfs_file_get_status( PROXYFS_FILE( proxyfile ), PROXYFS_FILE_CAN_READ) ) {
384  mdbg(INFO4, "Still can read -> read buffer before eof");
385  goto read;
386  }
387  return 0;
388  }
389  if( file->f_flags & O_NONBLOCK )
390  return -EAGAIN;
391  else { // Block on file
392  mdbg(INFO2, "file %p, waiting for can read", proxyfile);
393  switch( proxyfs_proxy_file_wait_interruptible(proxyfile,
394  PROXYFS_FILE_CAN_READ | PROXYFS_FILE_READ_EOF )){
395  // Can read shall be checked before EOF. It is possible to have both can_read and EOF up. In that case,
396  // we first read and eof after the buffer is empty
397  case PROXYFS_FILE_CAN_READ:
398  goto read;
399  case PROXYFS_FILE_READ_EOF:
400  return 0; // End of file
401  case 0:
402  return -EINTR;
403  }
404  }
405  }
406  return length;
407 }
