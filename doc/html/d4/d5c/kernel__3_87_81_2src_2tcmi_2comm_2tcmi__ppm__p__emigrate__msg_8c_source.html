45 /**
46  * <<public>> PPM_P message rx constructor.
47  * This method is called by the factory class.
48  *
49  * @param msg_id - message ID that will be used for this error message
50  * instance.
51  * @return a new error message or NULL.
52  */
53 struct tcmi_msg* tcmi_ppm_p_emigrate_msg_new_rx(u_int32_t msg_id)
54 {
55  struct tcmi_ppm_p_emigrate_msg *msg;
56 
57  /* Check if the factory is building what it really thinks. */
58  if (msg_id != TCMI_PPM_P_EMIGRATE_MSG_ID) {
59  mdbg(ERR3, "Factory specified message ID(%x) doesn't match real ID(%x)",
60  msg_id, TCMI_PPM_P_EMIGRATE_MSG_ID);
61  goto exit0;
62  }
63  /* Allocate the instance */
64  if (!(msg = TCMI_PPM_P_EMIGRATE_MSG(kmalloc(sizeof(struct tcmi_ppm_p_emigrate_msg), GFP_KERNEL)))) {
65  mdbg(ERR3, "Can't allocate test request message");
66  goto exit0;
67  }
68  /* Initialized the message for receiving. */
69  if (tcmi_msg_init_rx(TCMI_MSG(msg), TCMI_PPM_P_EMIGRATE_MSG_ID, &ppm_p_emigrate_msg_ops)) {
70  mdbg(ERR3, "Error initializing test request message %x", msg_id);
71  goto exit1;
72  }
73 
74  return TCMI_MSG(msg);
75 
76  /* error handling */
77  exit1:
78  kfree(msg);
79  exit0:
80  return NULL;
81 }

83 /**
84  * <<public>> PPM_P emigration message tx constructor.
85  *
86  * An emigration message expects a response that will be delivered back
87  * to the originating task (usually a shadow process on CCN). Therefore,
88  * there has to be a transaction associated with it.
89  *
90  * Response message ID is TCMI_GUEST_STARTED_PROCMSG_ID.
91  *
92  *
93  * @param *transactions - storage for the new transaction
94  *
95  * @param reply_pid - denotes the pid that the reply to this message
96  * should be directed to. It is a key information for the receiving
97  * party in order to compose a valid response
98  * @param ckpt_name - checkpoint pathname - this has to be a valid location
99  * in the network filesystem, so that the receiving party can restart the process
100  * @return a new error ready for the transfer or NULL.
101  */
102 struct tcmi_msg* tcmi_ppm_p_emigrate_msg_new_tx(struct tcmi_slotvec *transactions,
103  pid_t reply_pid, char *ckpt_name)
104 {
105  struct tcmi_ppm_p_emigrate_msg *msg;
106 
107  if (!(msg = TCMI_PPM_P_EMIGRATE_MSG(kmalloc(sizeof(struct tcmi_ppm_p_emigrate_msg), GFP_KERNEL)))) {
108  mdbg(ERR3, "Can't allocate test request message");
109  goto exit0;
110  }
111  msg->pid_and_size.reply_pid = reply_pid;
112  msg->pid_and_size.size = strlen(ckpt_name) + 1;
113 
114  if (!(msg->ckpt_name = (char*)kmalloc(msg->pid_and_size.size, GFP_KERNEL))) {
115  mdbg(ERR3, "Can't allocate memory for checkpoint name");
116  goto exit1;
117  }
118  strcpy(msg->ckpt_name, ckpt_name);
119  /* Initialize the message for transfer */
120  if (tcmi_msg_init_tx(TCMI_MSG(msg), TCMI_PPM_P_EMIGRATE_MSG_ID, &ppm_p_emigrate_msg_ops,
121  transactions, TCMI_GUEST_STARTED_PROCMSG_ID,
122  TCMI_PPM_P_EMIGRATE_MSGTIMEOUT, TCMI_TRANSACTION_INVAL_ID)) {
123  mdbg(ERR3, "Error initializing test request message message");
124  goto exit2;
125  }
126  return TCMI_MSG(msg);
127 
128  /* error handling */
129  exit2:
130  kfree(msg->ckpt_name);
131  exit1:
132  kfree(msg);
133  exit0:
134  return NULL;
135 
136 }

144 /**
145  * <<private>> Receives the message via a specified connection.
146  * Receiving the message requires reading the remote PID and
147  * checkpoint size. Based on the size, allocate space for the
148  * checkpoint name string and read it to from the specified connection
149  *
150  * @param *self - this message instance
151  * @param *sock - KKC socket used for receiving message data
152  * @return 0 when successfully received.
153  */
154 static int tcmi_ppm_p_emigrate_msg_recv(struct tcmi_msg *self, struct kkc_sock *sock)
155 {
156  int err = -EINVAL;
157  struct tcmi_ppm_p_emigrate_msg *self_msg = TCMI_PPM_P_EMIGRATE_MSG(self);
158 
159  /* Receive the remote PID and checkpoint name size */
160  if ((err = kkc_sock_recv(sock, &self_msg->pid_and_size,
161  sizeof(self_msg->pid_and_size), KKC_SOCK_BLOCK)) < 0) {
162  mdbg(ERR3, "Failed to receive pid and size");
163  goto exit0;
164  }
165 
166 
167  if (!(self_msg->ckpt_name = (char*)kmalloc(self_msg->pid_and_size.size, GFP_KERNEL))) {
168  mdbg(ERR3, "Can't allocate memory for checkpoint name");
169  goto exit0;
170  }
171 
172  /* Receive the checkpoint name */
173  if ((err = kkc_sock_recv(sock, self_msg->ckpt_name,
174  self_msg->pid_and_size.size, KKC_SOCK_BLOCK)) < 0) {
175  mdbg(ERR3, "Failed to receive checkpoint name");
176  goto exit0;
177  }
178  mdbg(INFO2, "PPM emigrate message received PID=%d, size=%d, ckptname='%s'",
179  self_msg->pid_and_size.reply_pid, self_msg->pid_and_size.size, self_msg->ckpt_name);
180 
181  return 0;
182  /* error handling*/
183  exit0:
184  return err;
185 }
