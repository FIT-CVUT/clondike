51 struct proxyfs_task *proxyfs_task_run( int(*func)(void*), const char *addr_str )
52 {
53  struct proxyfs_task_start_struct start_struct;
54  start_struct.addr_str = addr_str;
55  init_completion( & start_struct.startup_done );
56 
57  if( kthread_run( func, (void*)&start_struct, "ProxyFS %s", addr_str ) == ERR_PTR(-ENOMEM) )

178 void proxyfs_task_free(struct proxyfs_task* self) {
179  struct list_head *l, *nxt;
180  struct proxyfs_peer_t *peer;
181 
182  // First stop the processing thread
183  int result;
184  if ( current == self->tsk ) { // Special case when we are invoked from the proxyfs task thread directly
185  result = -EINVAL;
186  mdbg(INFO3, "Failed to startup proxyfs task listening => release it");
187  } else {
188  mdbg(INFO3, "Proxy task requesting thread stop");
189  result = kthread_stop(self->tsk);
190  }
191 
192  mdbg(INFO3, "Proxy task kthread stop result: %d Task: %p", result, self->tsk);
193 
194  if ( result )
195  return;
196 
197  // Process custom freeing
198  if ( self->ops && self->ops->free )
199  self->ops->free(self);
200 
201  // TODO: We can do this also in client free for open&open send files, right?
202  proxyfs_task_release_files(&self->files);
203 
204  // Finally free all peers (we do this after custom free as custom free will first stop possible server listening)
205  list_for_each_safe( l, nxt, &self->peers ){
206  peer = list_entry(l, struct proxyfs_peer_t, peers);
207  proxyfs_peer_disconnect(peer);
208  proxyfs_peer_put(peer);
209  }
210 }

217 {

221

223  peer = list_entry(l, struct proxyfs_peer_t, peers);
224 recv_next: recv_res = proxyfs_peer_real_recv(peer);
225  if ( recv_res == 1 ){ // Message received
226  mdbg(INFO3, "Message received");
227  proxyfs_task_handle_msg( self, peer );
228  proxyfs_peer_clear_msg(peer);
229  // Consider adding protection from receiving from only one peer
230  goto recv_next;
231  } else if ( recv_res == -ECONNRESET || recv_res == -ECONNABORTED ) {
232  // TODO: Some nicer general solution for removing (and detecting) lost peers.. but for now we need at least this
233  proxyfs_peer_set_state(peer, PEER_DEAD);
234  mdbg(INFO1, "Peer marked dead");
235  } else if (recv_res == -EAGAIN ) {
236  // Do nothing, repeat
237  } else {
238  mdbg(INFO1, "Unknown return code: %d", recv_res);
239  }
240 
241  }

280 /** <<protected>> Handle receive messagge
281  * @param *self - pointer to self instance
282  * @param *peer - peer instance with message we wanted to handle
283  */
284 void proxyfs_task_handle_msg(struct proxyfs_task *self, struct proxyfs_peer_t *peer)
285 {
286  struct proxyfs_file_t *file;
287  struct proxyfs_msg *msg = proxyfs_peer_get_msg(peer);
288  if ( !msg ) // Case, when we had incomplete message in the buffer
289  return;
290 
291  mdbg(INFO1, "Msg for identifier has arrived: %lu",msg->header.file_ident);
292 
293  switch( msg->header.msg_num ){
294  case MSG_WRITE_RESP:
295  if(( file = proxyfs_task_find_file(self, msg->header.file_ident)) != NULL){
296  if( msg->header.data[0] <= 0 ){
297  mdbg(ERR3, "Read must be > 0 but it is %lu", (unsigned long)msg->header.data[0]);
298  break;
299  }
300  else{
301  mdbg(INFO3, "Peer reads %lu bytes from file %lu buffer",
302  (unsigned long)msg->header.data[0], proxyfs_file_get_file_ident(file));
303  /*if( self->ops && self->ops->file_read )
304  self->ops->file_read(file, msg->header.data[0]);*/
305  proxyfs_file_submit_read_to_peer( file, msg->header.data[0]);
306  break;
307  }
308  }
309  mdbg(ERR3, "WRITE_RESP: Unknown file identificator %lu", msg->header.file_ident);
310  break;
311  case MSG_WRITE:
312  if(( file = proxyfs_task_find_file(self, msg->header.file_ident)) != NULL){
313  mdbg(INFO3, "Peer writes %lu bytes to file %lu", msg->header.data_size, proxyfs_file_get_file_ident(file));
314 
315  if( msg->header.data_size >= 0 ){
316  proxyfs_file_write_from_peer( file, (void*)msg->header.data, msg->header.data_size);
317  }
318  break;
319  }
320  mdbg(ERR3, "WRITE: Unknown file identificator %lu", msg->header.file_ident);
321  break;
322  default:
323  mdbg(INFO3, "Calling task specific msg handler");
324  if( self->ops && self->ops->handle_msg )
325  self->ops->handle_msg(self, peer);
326  }
327 
328  // Do not free message here. Incoming messages are kept only in rcv buffer and the buffer is always reused, so no need to free
329 }
