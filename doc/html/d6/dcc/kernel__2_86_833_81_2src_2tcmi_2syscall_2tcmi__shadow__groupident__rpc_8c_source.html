108 TCMI_SHADOW_RPC_GENERIC_CALL_DEF(getgroups)
109 {
110  struct tcmi_msg **resp, *m;

112  long rtn, numb_of_groups;
113  unsigned int i, count;
114  gid_t *ker_buf;
115 
116  m = (struct tcmi_msg*) params[0];

118 
119  rpc_params = (long*)tcmi_rpc_procmsg_data_base( TCMI_RPC_PROCMSG(m), 0);
120 
121  numb_of_groups = current_cred()->group_info->ngroups;
122  if( rpc_params[0] == 0 )
123  *resp = tcmi_rpcresp_procmsg_create(m, numb_of_groups, 0 );
124  else if( rpc_params[0] < numb_of_groups )
125  *resp = tcmi_rpcresp_procmsg_create(m, -EINVAL, 0 );
126  else{
127  if( (ker_buf = (gid_t*)kmalloc( numb_of_groups * sizeof(gid_t), GFP_KERNEL ) ) == NULL ){
128  mdbg(ERR3, "Buffer allocation failed");
129  return -1;
130  }
131 
132  rtn = count = numb_of_groups;
133  for (i = 0; i < count; i++) {
134  int cp_count = min(NGROUPS_PER_BLOCK, count);
135  int off = i * NGROUPS_PER_BLOCK;
136  int len = cp_count * sizeof(*ker_buf);
137 
138  memcpy(ker_buf+off, current_cred()->group_info->blocks[i], len);
139 
140  count -= cp_count;
141  }
142 
143  *resp = tcmi_rpcresp_procmsg_create(m, rtn, sizeof(gid_t)*numb_of_groups, ker_buf, 0 );
144  tcmi_rpcresp_procmsg_free_on_put( TCMI_RPCRESP_PROCMSG(*resp), 0, TCMI_RPCRESP_PROCMSG_KFREE );
145  }
146 
147  if( *resp == NULL )
148  return -1;
149  return 0;
150 }
