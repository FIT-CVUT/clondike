126 static inline int proxyfs_peer_clear_msg(struct proxyfs_peer_t *self)
127 {
128  if( self->recv_start >= MSG_HDR_SIZE &&

155 /** <<public>> Add message to sending queue
156  * @param self - pointer to proxyfs_peer_t instance
157  * @param msg - pointer to msg, which will be added to queue
158  */
159 static inline void proxyfs_peer_send_msg(struct proxyfs_peer_t *self, struct proxyfs_msg *msg)
160 {
161  mdbg(INFO4, "Adding msg %p to a send queue of peer %p", msg, self);
162  spin_lock(&self->enqueue_msg_lock);
163  list_add_tail( & msg->msg_queue, & self->msg_queue );
164  spin_unlock(&self->enqueue_msg_lock);
165  mdbg(INFO3, "Queued message %lu for file %lu", msg->header.msg_num, msg->header.file_ident);
166 }
