54 /**
55  * <<public>> TCMI migration manager initializer.
56  * The initialization is accomplished exactly in this order:
57  * - set ID's and initial state
58  * - initialize message queue and the transaction vector
59  * - choose a unique ID for the manager.
60  * - store a reference to a mig proc directory - it's required when
61  * immigrating a process.
62  * - start TCMI communication component
63  * - create TCMI ctlfs directories
64  * - create TCMI ctlfs control files
65  * - create a TCMI socket to represent the connection in filesystem
66  * - run the message processing thread.
67  *
68  * @param *self - this migration manager instance
69  * @param *sock - socket where the control connection between CCN/PEN is running
70  * @param ccn_id - CCN identifier - valid when initializing a CCN mig. manager
71  * @param pen_id - PEN identifier - valid when initializing a PEN mig. manager
72  * @param peer_arch_type - Architecture of the peer node
73  * @param *root - directory where the migration manager should create its
74  * files and directories
75  * @param *migproc - directory where the migrated process will have their info
76  * @param *ops - instance specific operations
77  * @param namefmt - nameformat string for the main manager directory name (printf style)
78  * @param args - variable arguments list
79  * @return 0 upon success
80  */

174 /**
175  * <<public>> The manager is shutdown exactly in this order:
176  * - calls the free method specific to a particular migration manager
177  * - stop message receiving thread
178  * - destroy TCMI socket
179  * - unregister all ctlfs files as we don't want anymore requests
180  * from users.
181  * - unregister all ctlfs directories
182  * - release the communication component
183  * - release the transaction slot vector - check for stale transactions
184  * - release the shadows slot vector
185  *
186  * @param *self - pointer to this migration manager instance
187  * @return Returns 1, if the instance was destroyed as a result of this call
188  */
189 int tcmi_migman_put(struct tcmi_migman *self)
190 {
191  if (self && atomic_dec_and_test(&self->ref_count)) {
192  mdbg(INFO3, "Shutting down the TCMI migration manager %p", self);
193  tcmi_migman_free(self);
194  kfree(self);
195 
196  return 1;
197  }
198 
199  return 0;
200 }

218 /**
219  * <<private>> Free method is has two purposes:
220  * -# when the last reference is released, it will free all
221  * instance resources, except for the instance itself.
222  * -# calls the child class free method if defined
223  *
224  * Note: It is important to free communication channels first, i.e. communication socket, processing thread and files. This ensures,
225  * that no new messages that may require reference to migman for processing will arrive. File operations are synchronized with releasing
226  * so it may be possible that some file works with the migman during the free method, but only until the file is ''synchronously'' released.
227  */
228 static void tcmi_migman_free(struct tcmi_migman *self)
229 {
230  /* terminate the processing thread first to prevent any
231  * message processing */
232  tcmi_migman_stop_thread(self);
233  /* also stop receiving all messages */
234  tcmi_comm_put(self->comm);
235 
236  tcmi_migman_stop_ctlfs_files(self);
237  tcmi_migman_stop_ctlfs_dirs(self);
238  tcmi_sock_put(self->sock);
239  tcmi_migman_unhash(self);
240 
241 
242  /* now we are safe and we can start cleaning the rest */
243  if (self->ops->free)
244  self->ops->free(self);
245 
246  /* TODO: check for stale transactions.. */
247  tcmi_slotvec_put(self->transactions);
248  /* Note: Assumes, all tasks are already migrated back. */
249  tcmi_slotvec_put(self->tasks);
250 }

252 /**
253  * Requests asynchronous shutdown of connection with peer.
254  * First, it asynchronously requests emigration of tasks to their home node, in case we are on detached node.
255  * Then it notifies peer about current node being disconnected.
256  * Finally, a self reference is dropped so that the migman is free to be destroyed when all tasks are finished
257  */

263  struct tcmi_msg* msg;
264  int err;
265 
266  if ( !( msg = TCMI_MSG(tcmi_disconnect_msg_new_tx()) ) ) {

276 /**
277  * Actual implementation of stop, see tcmi_migman_stop
278  *
279  * @param remote_requested True, if stop request was initiated as a response to received disconnect message
280  * @Return 1, if the migman instance was destroyed in context of this method
281  */
282 static int tcmi_migman_stop_perform(struct tcmi_migman *self, int remote_requested) {
283  minfo(INFO3, "Requesting stop of migration manager: %d Remote requested: %d", tcmi_migman_slot_index(self), remote_requested);
284 
285  if ( tcmi_migman_change_state(self, TCMI_MIGMAN_CONNECTED, TCMI_MIGMAN_SHUTTING_DOWN) ) {
286  /* We are the first to request stop operation */
287 
288  if (self->ops->stop)
289  self->ops->stop(self, remote_requested);
290 
291  if ( !remote_requested ) {
292  /* Sent notification to peer about disconnection being performed.. even if this fails (peer may be dead already), we proceed further */
293  tcmi_migman_notify_peer_disconnect(self);
294  }
295 
296  /* Finally drop the reference to "self". If this is the last reference, the instance is going to be unhased and deleted.
297  At this phase it is possible some tasks are still running and so they have reference to this instance. Asynchronous emigration requests were already issued */
298  return tcmi_migman_put(self);
299  } else {
300  minfo(INFO3, "Migration manager stop was already requested -> Ignoring this call. Manager state is: %d", tcmi_migman_state(self));
301  }
302 
303  return 0;
304 }

306 /** Sends a kill signal to all tasks managed by this manager */
307 static void kill_all_tasks(struct tcmi_migman *self) {
308  struct tcmi_task* task;
309  struct tcmi_slot *slot;
310  tcmi_slot_node_t* node;
311 
312  tcmi_slotvec_lock(self->tasks);
313  /* Iterate over all managed tasks */
314  tcmi_slotvec_for_each_used_slot(slot, self->tasks) {
315  tcmi_slot_for_each(node, slot) {
316  task = tcmi_slot_entry(node, struct tcmi_task, node);
317  mdbg(INFO3, "Killing task: local_pid=%d", tcmi_task_local_pid(task));
318  tcmi_task_signal_peer_lost(task);
319  };
320  };
321  tcmi_slotvec_unlock(self->tasks);
322 }

324 /**
325  * Kills all tasks associated with this migman and drops reference to the migman (in case it was not already shutting down).
326  * The method is useful in case peer died without cleaning up its own tasks, because we need to clean them up before we can release reference to peer.
327  *
328  * @return 1, if the migman instance was destroyed in context of this method
329  */
330 int tcmi_migman_kill(struct tcmi_migman *self) {
331  minfo(INFO3, "Requesting kill of migration manager: %d", tcmi_migman_slot_index(self));
332 
333  kill_all_tasks(self);
334 
335  if ( tcmi_migman_change_state(self, TCMI_MIGMAN_CONNECTED, TCMI_MIGMAN_SHUTTING_DOWN) ) {
336  /* Stop operation was not yet requested -> request it. */
337  return tcmi_migman_put(self);
338  } else {
339  minfo(INFO3, "Migration manager stop was already requested -> Do not perform release this call. Manager state is: %d", tcmi_migman_state(self));
340  }
341 
342 
343  return 0;
344 }

588 /**
589  * <<private>> Message delivering method that is called back from
590  * the receiving thread of TCMI comm component instance. This method
591  * is reponsible for delivering the message reliably to its
592  * destination which can be:
593  *
594  * - local migration manager message queue
595  * - local migration manager transaction slot vector
596  * - TCMI task message queue
597  * - TCMI task transaction slot vector
598  *
599  * @param *obj - pointer to the migration manager instance.
600  * passed to the TCMI comm component upon creation
601  * @param *m - points to the message that is to be delivered.
602  * @return 0 upon successful delivery
603  */
604 static int tcmi_migman_deliver_msg(void *obj, struct tcmi_msg *m)
605 {
606  int err = 0;
607  struct tcmi_migman *self = TCMI_MIGMAN(obj);
608  struct tcmi_task *task;
609 
610  if (TCMI_MSG_GROUP(tcmi_msg_id(m)) == TCMI_MSG_GROUP_PROC) {
611  /* try finding the task based on PID in the message */
612  pid_t pid = tcmi_procmsg_dst_pid(TCMI_PROCMSG(m));
613  mdbg(INFO3, "Process message %08x arrived as expected for PID=%d",
614  tcmi_msg_id(m), pid);
615  /* find task by PID */
616  if (!(task = tcmi_taskhelper_find_by_pid(pid))) {
617  mdbg(ERR2, "Can't find dest. task for msg ID=%08x",
618  tcmi_msg_id(m));
619  goto exit0;
620  }
621  /* Verify it belongs to us */
622  if (tcmi_task_migman_id(task) != tcmi_migman_id(self)) {
623  mdbg(ERR2, "Task(PID=%d, migman ID=%08x) doesn't belong"
624  "to migration manager ID=%08x", tcmi_task_local_pid(task),
625  tcmi_task_migman_id(task), tcmi_migman_id(self));
626  goto exit1;
627  }
628  err = tcmi_task_deliver_msg(task, m);
629  tcmi_task_put(task);
630  }
631  else {
632  mdbg(INFO3, "Delivering regular migration message %08x",
633  tcmi_msg_id(m));
634  err = tcmi_msg_deliver(m, &self->msg_queue, self->transactions);
635  }
636 
637  return err;
638 
639  /* error handling */
640  exit1:
641  tcmi_task_put(task);
642  exit0:
643  return -EINVAL;
644 }

646 /**
647  * <<private>> Message processing thread.
648  * Sleeps on the message queue and notifies the migration manager
649  * instance to process each incoming message.
650  *
651  * @param *data - migration manager instance
652  * @return 0 upon successful thread termination.
653  */
654 static int tcmi_migman_msg_thread(void *data)
655 {
656  int err = 0;
657  int migman_freed = 0;
658  struct tcmi_msg *m;
659  struct tcmi_migman *self = TCMI_MIGMAN(data);
660  /* saves 2 derefences when calling the process msg method */
661  process_msg_t *process_msg = self->ops->process_msg;
662 
663  if (!process_msg) {
664  mdbg(ERR3, "Missing method to process messages!");
665  goto exit0;
666  }
667 
668 
669  mdbg(INFO3, "Message processing thread up and running %p", current);
670  while (!(kthread_should_stop() || signal_pending(current))) {
671  if ((err = tcmi_queue_wait_on_empty_interruptible(&self->msg_queue)) < 0) {
672  minfo(INFO1, "Signal arrived %d", err);
673  break;
674  }
675  tcmi_queue_remove_entry(&self->msg_queue, m, node);
676  if (m) {
677  mdbg(INFO3, "Processing message..");
678  switch(tcmi_msg_id(m)) {
679  default:
680  process_msg(self, m);
681  break;
682 
683  case TCMI_DISCONNECT_MSG_ID:
684  minfo(INFO3, "Received disconnect message");
685  /*
686  Initiate shutdown upon reception of disconnect message
687  Note: It is important not to touch self after this call and before should_stop is checked again, because self reference may be already invalid!
688  */
689  migman_freed = tcmi_migman_stop_perform(self, 1);
690  break;
691  }
692  }
693  else {
694  mdbg(ERR3, "Processing thread woken up, queue is still empty..");
695  }
696  }
697  exit0:
698  /* In case the migman was freed by this thread, no should_stop sync is performed */
699  if ( !migman_freed ) {
700  /* We have to wait for thread terminating us as it is going to call stop at some tim */
701  wait_on_should_stop();
702  }
703 
704  mdbg(INFO3, "Message processing thread terminating");
705  return 0;
706 }
