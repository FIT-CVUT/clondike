38 /** @defgroup tcmi_procmsg_class tcmi_procmsg class
39  *
40  * @ingroup tcmi_msg_class
41  *
42  * This class represents a generic process message. Process message,
43  * in addition to regular migration message, carries a target
44  * process PID. This PID will be used by the receiving party for message
45  * delivery. The only functionality provided by this class is
46  * to send/receive the process ID and notify the specific procmsg
47  * via send/receive/free operations.
48  *
49  *
50  * @{
51  */
52 
53 /** Compound structure, inherits from tcmi_msg_class */
54 struct tcmi_procmsg {
55  /** parent class instance */
56  struct tcmi_msg super;
57  /** destination process PID */
58  pid_t dst_pid;
59  /** == 1, if the we should switch to migration mode after recieving this message */
60  u_int8_t enforce_migmode;
61  /** proc message operations */
62  struct tcmi_procmsg_ops *msg_ops;
63 };

66 /** Process message operations that support polymorphism. */
67 struct tcmi_procmsg_ops {

70  /** Sends the message via a specified connection. */

72  /** Frees custom message resources. The destruction of the
