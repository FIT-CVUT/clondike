35 /** @defgroup tcmi_messages_id message ID's
36  *
37  * @ingroup tcmi_comm_group
38  *
39  * Message ID has 3 components:
40  *
41  * - bits 31-30 - message flags
42  * - 0 - regular message
43  * - 1 - error message
44  * - 2 - reserved
45  * - 3 - reserved
46  * - bits 29-16 - message group ID. Initially, there are 2 groups:
47  * - 0 - migration control connection messages
48  * - 1 - process control messages
49  *
50  * - bits 15-0 - a unique message index within a particular
51  * group.
52  \verbatim
53  31 30 29 16 15 0
54  +------+-----------+-----------+
55  | flgs | group | index |
56  +------+-----------+-----------+
57  \ /
58  \ /
59  \ /
60  \ /
61  \ /
62  message type - used when associating
63  a response message with a transaction
64  \endverbatim
65  * @{
66  */
67 /** There are 2^index_bits messages */
68 #define TCMI_MSG_INDEX_BITS 16
69 /** There are 2^group_bits message groups */
70 #define TCMI_MSG_GROUP_BITS 14
71 /** There are 2^flg_bits flags available */
72 #define TCMI_MSG_FLG_BITS 2
73 /** Total number of bits reserved for message identification. */
74 #define TCMI_MSG_TYPE_BITS (TCMI_MSG_GROUP_BITS + TCMI_MSG_INDEX_BITS)
75 
76 /** Mask for the message index */
77 #define TCMI_MSG_INDEX_MSK ((1 << TCMI_MSG_INDEX_BITS) - 1)
78 /** Extracts message index */
79 #define TCMI_MSG_INDEX(msg_id) (msg_id & TCMI_MSG_INDEX_MSK)
80 /** Mask for the group */
81 #define TCMI_MSG_GROUP_MSK ((1 << TCMI_MSG_GROUP_BITS) - 1)
82 /** Extracts group number */
83 #define TCMI_MSG_GROUP(msg_id) ((msg_id >> TCMI_MSG_INDEX_BITS) & TCMI_MSG_GROUP_MSK)
84 /** Mask for the message type */
85 #define TCMI_MSG_TYPE_MSK ((1 << TCMI_MSG_TYPE_BITS) - 1)
86 /** Extracts the message type */
87 #define TCMI_MSG_TYPE(msg_id) (msg_id & TCMI_MSG_TYPE_MSK)
88 /** Mask for the flag bits */
89 #define TCMI_MSG_FLG_MSK ((1 << TCMI_MSG_FLG_BITS) - 1)
90 /** Extracts the message flags */
91 #define TCMI_MSG_FLG(msg_id) ((msg_id >> TCMI_MSG_TYPE_BITS) & TCMI_MSG_FLG_MSK)
92 /** Flag shift */
93 #define TCMI_MSG_FLG_SHIFT TCMI_MSG_TYPE_BITS
94 /** Sets a specified group */
95 #define TCMI_MSG_SET_GRP(grp) (grp << TCMI_MSG_INDEX_BITS)
96 
97 /** Regular message flag */
98 #define TCMI_MSG_REGFLGS 0
99 /** Error message flag */
100 #define TCMI_MSG_ERRFLGS 1
101 /** Sets an error flag */
102 #define TCMI_MSG_FLG_SET_ERR(id) (id | (TCMI_MSG_ERRFLGS << TCMI_MSG_FLG_SHIFT))
