52 /** Compound structure describes a process memory descriptor.
53  * Explanation if individual items is e.g. in Understanding
54  * the Linux Kernel(chapter 8)
55  */
56 struct tcmi_ckpt_mm_hdr {
57  /** code section, data section. */
58  u_int64_t start_code, end_code, start_data, end_data;
59  /** heap and stack. */
60  u_int64_t start_brk, brk, start_stack;
61  /** arguments and environment. */
62  u_int64_t arg_start, arg_end, env_start, env_end;
63  /** default access flags of the memory regions. */
64  u_int64_t def_flags;
65 } __attribute__((__packed__));

68 /**
69  * Writes a current process memory descriptor into the checkpoint file.
70  *
71  * @param *ckpt - checkpoint file where the area is to be stored
72  * @return 0 upon success.
73  */
74 static inline int tcmi_ckpt_mm_write(struct tcmi_ckpt *ckpt)
75 {
76  struct tcmi_ckpt_mm_hdr mm_hdr;
77  /* Now we dump the memory mapping */
78  mm_hdr.start_code = current->mm->start_code;
79  mm_hdr.end_code = current->mm->end_code;
80  mm_hdr.start_data = current->mm->start_data;
81  mm_hdr.end_data = current->mm->end_data;
82  mm_hdr.start_brk = current->mm->start_brk;
83  mm_hdr.brk = current->mm->brk;
84  mm_hdr.start_stack = current->mm->start_stack;
85  mm_hdr.arg_start = current->mm->arg_start;
86  mm_hdr.arg_end = current->mm->arg_end;
87  mm_hdr.env_start = current->mm->env_start;
88  mm_hdr.env_end = current->mm->env_end;
89  mm_hdr.def_flags = current->mm->def_flags;
90 
91 
92 
93  /* write the header into the checkpoint */
94  if (tcmi_ckpt_write(ckpt, &mm_hdr, sizeof(mm_hdr)) < 0) {
95  mdbg(ERR3, "Error writing memory descriptor header");
96  goto exit0;
97  }
98  mdbg(INFO4, "Written memory descriptor:");
99  mdbg(INFO4, "START CODE : %08lx", current->mm->start_code);
100  mdbg(INFO4, "END CODE : %08lx", current->mm->end_code);
101  mdbg(INFO4, "START DATA : %08lx", current->mm->start_data);
102  mdbg(INFO4, "END DATA : %08lx", current->mm->end_data);
103  mdbg(INFO4, "START BRK : %08lx", current->mm->start_brk);
104  mdbg(INFO4, "BRK : %08lx", current->mm->brk);
105  mdbg(INFO4, "START STACK: %08lx", current->mm->start_stack);
106  return 0;
107 
108  /* error handling */
109  exit0:
110  return -EINVAL;
111 
112 }

114 /**
115  * Reads a memory area from the checkpoint file and sets the those in
116  * the memory descriptor. This also assumes that the old execution
117  * context has already been flushed by the caller.
118  *
119  * @param *ckpt - checkpoint file where the area is to be stored
120  * @return 0 upon success.
121  */
122 static inline int tcmi_ckpt_mm_read(struct tcmi_ckpt *ckpt)
123 {
124  struct tcmi_ckpt_mm_hdr mm_hdr;
125  /* write the header into the checkpoint */
126  if (tcmi_ckpt_read(ckpt, &mm_hdr, sizeof(mm_hdr)) < 0) {
127  mdbg(ERR3, "Error reading memory descriptor header");
128  goto exit0;
129  }
130  current->mm->start_code = CHECKED_UINT64_TO_ULONG(mm_hdr.start_code);
131  current->mm->end_code = CHECKED_UINT64_TO_ULONG(mm_hdr.end_code);
132  current->mm->start_data = CHECKED_UINT64_TO_ULONG(mm_hdr.start_data);
133  current->mm->end_data = CHECKED_UINT64_TO_ULONG(mm_hdr.end_data);
134  current->mm->start_brk = CHECKED_UINT64_TO_ULONG(mm_hdr.start_brk);
135  current->mm->brk = CHECKED_UINT64_TO_ULONG(mm_hdr.brk);
136  current->mm->start_stack = CHECKED_UINT64_TO_ULONG(mm_hdr.start_stack);
137  current->mm->arg_start = CHECKED_UINT64_TO_ULONG(mm_hdr.arg_start);
138  current->mm->arg_end = CHECKED_UINT64_TO_ULONG(mm_hdr.arg_end);
139  current->mm->env_start = CHECKED_UINT64_TO_ULONG(mm_hdr.env_start);
140  current->mm->env_end = CHECKED_UINT64_TO_ULONG(mm_hdr.env_end);
141  current->mm->def_flags = CHECKED_UINT64_TO_ULONG(mm_hdr.def_flags);
142  /* no regions left as they all have been flushed */
143  current->mm->mmap = NULL;
144  mdbg(INFO4, "Read memory descriptor:");
145  mdbg(INFO4, "START CODE : %08lx", current->mm->start_code);
146  mdbg(INFO4, "END CODE : %08lx", current->mm->end_code);
147  mdbg(INFO4, "START DATA : %08lx", current->mm->start_data);
148  mdbg(INFO4, "END DATA : %08lx", current->mm->end_data);
149  mdbg(INFO4, "START BRK : %08lx", current->mm->start_brk);
150  mdbg(INFO4, "BRK : %08lx", current->mm->brk);
151  mdbg(INFO4, "START STACK: %08lx", current->mm->start_stack);
152  mdbg(INFO4, "ARG START: %08lx", current->mm->arg_start);
153  mdbg(INFO4, "ARG END: %08lx", current->mm->arg_end);
154  mdbg(INFO4, "ENV START: %08lx", current->mm->env_start);
155  mdbg(INFO4, "ENV END: %08lx", current->mm->env_end);
156  mdbg(INFO4, "DEF FLAGS: %08lx", current->mm->def_flags);
157  return 0;
158 
159  /* error handling */
160  exit0:
161  return -EINVAL;
162 }
