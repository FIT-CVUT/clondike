40 /** @defgroup tcmi_ckpt_vm_area_class tcmi_ckpt_vm_area class
41  *
42  * @ingroup tcmi_ckpt_class
43  *
44  * This is a <<singleton>> that takes care of (re)storing a
45  * particular memory segment (VM area) into the checkpoint file.
46  *
47  * There are 2 modes supported:
48  * - heavy checkpoint mode - where the entire segment is dumped
49  * page by page into the checkpoint.
50  * - light checkpoint mode - if the segment maps a file on disk
51  * and the segment is non-writable, only file pathname is dumped
52  * into the checkpoint. This decreases the resulting checkpoint
53  * size dramatically. Also, when restoring such checkpoint parts,
54  * there is a big chance that the file is already mapped in memory.
55  * This typically happens with shared libraries and executable code.
56  *
57  * @{
58  */
59

117 /**
118  * <\private>> Converts the VM area flags to mmap flags.
119  *
120  * Each mapping is marked fixed as that is the only way how to restart
121  * a process from the checkpoint (virtual addresses must match).
122  *
123  * VM_GROWSDOWN, VM_DENYWRITE flags directly match mmap
124  * flags, so only bit extraction needs to be done. VM_GROWSDOWN flags
125  * is special as it marks the stack. Stack fixup is handled separate by
126  * tcmi_ckpt_vm_area_stack_fixup().
127  *
128  * VM_SHARED flag doesn't match the mmap flags directly, so a
129  * conditional check needs to be done: anything not shared is mmaped
130  * as private.
131  *
132  * @param vm_flags - flags to be converted
133  * @return mmap flags
134  */
135 static inline unsigned long tcmi_ckpt_vm_area_to_mmap_flags(unsigned long vm_flags)
136 {
137  return MAP_FIXED |
138  (vm_flags & VM_SHARED ? MAP_SHARED : MAP_PRIVATE) |
139  (vm_flags & VM_GROWSDOWN) |
140  (vm_flags & VM_DENYWRITE);
141  //Remove for kernel 3.7.x by Jiri Rakosnik
142 }

145 /**
146  * <\private>> This is really stupid, but linux kernel doesn't
147  * export kmap/kunmap, so we had to duplicate the highmemory check..
148  *
149  * @param *page - page that is to be mapped into the linear address space
150  * @return linear address of the page
151  */
152 static inline void* tcmi_ckpt_vm_area_kmap(struct page *page)
153 {
154  might_sleep();
155 
156  #if defined(__i386__)
157  if (!PageHighMem(page))
158  return page_address(page);
159  return kmap_high(page);
160  #else
161  return page_address(page);
162  #endif
163 }

170  * @param *page - page that is to be unmapped
171  */
172 static inline void tcmi_ckpt_vm_area_kunmap(struct page *page)
173 {
174  if (in_interrupt())
175  BUG();
176 
177  #if defined(__i386__)
178  if (!PageHighMem(page))
179  return;
180  kunmap_high(page);
181  #endif
182 }

247 static inline int tcmi_ckpt_vm_area_copy_page(unsigned long dst_addr, unsigned long src_addr)
248 {
249  struct page *dst_page;
250  struct vm_area_struct *vma;
251  void *kaddr;
252 
253  if (get_user_pages(current, current->mm, dst_addr, 1, 1, 1,
254  &dst_page, &vma) <= 0) {
255  mdbg(ERR3, "Failed to get a user page at %08lx", dst_addr);
256  goto exit0;
257  }
258  /* actual page, that needs to be written. */
259  kaddr = tcmi_ckpt_vm_area_kmap(dst_page);
260  memcpy(kaddr, (void*)src_addr, PAGE_SIZE);
261  tcmi_ckpt_vm_area_kunmap(dst_page);
262  return 0;
263 
264  /* error handling */
265  exit0:
266  return -EINVAL;
