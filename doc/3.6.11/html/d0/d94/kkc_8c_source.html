79 /**
80  * <<private>> Creates an architecture specific socket and starts
81  * listening or connects to the destination. First an architecture,
82  * based on the prefix in the address, has to be retrieved. The
83  * creation of the socket is delegated to the archictecture. If
84  * everything goes fine, the socket is asked to perform an operation
85  * specified by the user - kkc_sock_method().
86  *
87  * @param **sock - ouput parameter - upon successful creation of
88  * a connection or listening contains a valid socket
89  * @param *where - address string with architecture prefix -
90  * e.g. tcp:192.168.0.1.2:2000
91  * @param *kkc_sock_method - method called on the created socket.
92  * This would yield a new connection or a listening, depending
93  * on what method the user specified
94  * @return 0 upon success
95  */
96 static int kkc_establish(struct kkc_sock **sock, const char *where,
97  int (*kkc_sock_method)(struct kkc_sock *, char *))
98 {
99 
100  struct kkc_arch *arch;
101  int err = 0;
102  char addr[KKC_SOCK_MAX_ADDR_LENGTH];
103  *sock = NULL;
104 
105  mdbg(INFO4, "Requesting architecture structure and network address");
106  if ((err = kkc_get_architecture(where, &arch, addr)) < 0) {
107  mdbg(ERR4, "Cannot get architecture structure");
108  err = -ENOMEM;
109  goto exit0;
110  }
111 
112  /* instantiate the socket */
113  if (!((*sock) = KKC_SOCK(kkc_arch_obj_new(arch)))) {
114  mdbg(ERR4, "Failed to create a new socket");
115  err = -ENOMEM;
116  goto exit1;
117  }
118  if (kkc_sock_method(*sock, addr)) {
119  mdbg(ERR4, "KKC socket method failed");
120  err = -ENOMEM;
121  goto exit2;
122  }
123  /* release the architecture as the socket has created its own
124  * reference */
125  kkc_arch_put(arch);
126  return 0;
127  /* error handling */
128  exit2:
129  kkc_sock_put(*sock);
130  *sock = NULL; /* return no socket to the user */
131  exit1:
132  kkc_arch_put(arch);
133  exit0:
134  return err;
135 }

137 /**
138  * <<private>> Helper function for parsing 'where' string and
139  * requesting for particular architecture structure. As side effect
140  * fills 'addr' string with address part of 'where'.
141  *
142  * @param *where - pointer to the where request string
143  * @param **arch - output parameter - stores the architecture if one found
144  * @param *addr - storage for the address string, that gets extracted from
145  * the where string
146  * @return 0 upon success.
147  */
148 static int kkc_get_architecture(const char *where, struct kkc_arch **arch, char *addr)
149 {
150  char arch_name[KKC_MAX_ARCH_LENGTH];
151  char *colon_ptr;
152  int colon_pos;
153 
154  int err = 0;
155 
156  /* find colon */
157  if ((colon_ptr = strchr(where, ':')) == NULL) {
158  mdbg(ERR4, "Invalid syntax - could not find colon");
159  err = -EINVAL;
160  goto exit0;
161  }
162 
163  /* check position */
164  colon_pos = colon_ptr - where;
165  if (colon_pos > (KKC_MAX_ARCH_LENGTH - 1)) {
166  mdbg(ERR4, "Invalid syntax - too long architecture name length");
167  err = -EINVAL;
168  goto exit0;
169  }
170 
171  /* get architecture name */
172  strncpy(arch_name, where, colon_pos);
173  arch_name[colon_pos] = '\0';
174  mdbg(INFO4, "Extracted architecture name: ""%s""", arch_name);
175 
176  /* get address */
177  strncpy(addr, colon_ptr + 1, KKC_SOCK_MAX_ADDR_LENGTH - 1);
178  addr[KKC_SOCK_MAX_ADDR_LENGTH - 1] = '\0';
179  mdbg(INFO4, "Extracted address: ""%s""", addr);
180 
181 
182  if (!(*arch = kkc_arch_get_by_name(arch_name))) {
183  mdbg(ERR4, "Could not get architecture for '%s'", arch_name);
184  err = -EINVAL;
185  }
186 
187  /* error handling */
188 exit0:
189  return err;
190 }
