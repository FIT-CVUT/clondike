38 /** @defgroup kkc_sock_class kkc_sock class
39  *
40  * @ingroup kkc_class
41  *
42  * This is a generic KKC socket. It handles basic administration of
43  * the generic socket, initialization of common socket parameters, and
44  * releasing the socket instance. It is questionable if a regular BSD
45  * socket should be included into the compound structure or
46  * not. Currently, there is no other communication abstraction in Linux
47  * other than sockets. For now, the BSD socket handling functionality
48  * will be in child classes (e.g. kkc_sock_tcp). For the common part
49  * for all network architectures, we will keep only the address in
50  * string form in the kkc_socket super class.
51  *
52  * KKC socket is meant for full duplex communication. In addition, it
53  * provides locking mechanism that allows separate serialization of
54  * send and receive requests.
55  *
56  * @{
57  */

168 /**
169  * <<public>> Local socket name accessor. Asks the architecture specific socket to
170  * fill in its local name (address). The generic socket operation
171  * getname() is used, passing 0 selects the local name.
172  *
173  * @param *self - pointer to this socket instance
174  * @param *sockname - pointer to the buffer where the sockname is to be stored
175  * @param size - maximum string length that fits into the sockname buffer
176  * @return 0 upon success
177  */
178 static inline int kkc_sock_getsockname(struct kkc_sock *self, char *sockname, int size)
179 {
180  int err = 0;
181  if (self->sock_ops && self->sock_ops->getname)
182  err = self->sock_ops->getname(self, sockname, size, 0);
183  return err;
184 }

261 /**
262  * <<public>> Decrements reference counter, if it reaches 0 the
263  * custom free method is called if defined. Also the architecture is
264  * released.
265  *
266  * The user is responsible for taking the socket out of any lists
267  * where he had stored it.
268  *
269  * @param *self - pointer to this socket instance
270  */
271 static inline void kkc_sock_put(struct kkc_sock *self)
272 {
273  if (!self)
274  return;
275  if (atomic_dec_and_test(&self->ref_count)) {
276  mdbg(INFO4, "Destroying KKC socket local: '%s', remote: '%s', %p",
277  kkc_sock_getsockname2(self),
278  kkc_sock_getpeername2(self), self);
279  if (self->sock_ops && self->sock_ops->free)
280  self->sock_ops->free(self);
281  /* release the architecture */
282  kkc_arch_put(self->arch);
283 
284  kfree(self);
285  }
