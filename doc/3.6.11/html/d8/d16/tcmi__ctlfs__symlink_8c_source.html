44 /**
45  * <<public>> Symbolic link is created as a regular ctlfs
46  * entry. In addition, it retains an extra reference to the target
47  * object.
48  *
49  * @param *parent - pointer to the parent entry - has to be a directory
50  * @param *target - target entry to which link should point to
51  * @param namefmt - nameformat string (printf style)
52  * @return new symlink instance or NULL
53  */
54 struct tcmi_ctlfs_entry* tcmi_ctlfs_symlink_new(struct tcmi_ctlfs_entry *parent,
55  struct tcmi_ctlfs_entry *target,
56  const char namefmt[], ...)
57 {
58  struct tcmi_ctlfs_symlink *symlink;
59  va_list args;
60  va_start(args, namefmt);
61  mdbg(INFO4, "Creating new symlink");
62 
63  if (!(symlink = kmalloc(sizeof(struct tcmi_ctlfs_symlink), GFP_ATOMIC))) {
64  mdbg(ERR3, "Failed to allocate memory for symlink");
65  goto exit0;
66  }
67  if (tcmi_ctlfs_entry_init(TCMI_CTLFS_ENTRY(symlink),
68  TCMI_CTLFS_ENTRY(parent),
69  S_IFLNK|S_IRWXUGO,
70  &tcmi_ctlfs_symlink_ops,
71  &tcmi_ctlfs_symlink_inode_operations,
72  NULL, namefmt, args)) {
73  mdbg(ERR3, "Failed to initialize the symlink");
74  goto exit1;
75  }
76  symlink->target_entry = tcmi_ctlfs_entry_get(target);
77  va_end(args);
78 
79  return TCMI_CTLFS_ENTRY(symlink);
80 
81  /* error handling */
82  exit1:
83  kfree(symlink);
84  exit0:
85  va_end(args);
86  return NULL;
87 }

111 /**
112  * <<private>> The path is built as follows:
113  * - The path needs to be traversed to the root directory first '../', the number
114  * of these elements is determined by the depth of the symlink in the tree.
115  * - the minimum required buffer length is calculated based on:
116  * - symlink depth (see above)
117  * - pathname length of the target entry
118  * - the buffer is then filled with ../'s
119  * - the path of the target entry is then filled in
120  *
121  * Should the target entry be invalid, the function terminates with an error
122  * access to the target entry is serialized by the lock.
123  *
124  * @param *self - pointer to this instance
125  * @param *path - storage buffer for the path
126  * @param length - maximum path length (storage buffer size)
127  * @return 0 upon success
128  */
129 static int tcmi_ctlfs_symlink_get_link(struct tcmi_ctlfs_symlink *self,
130  char *path, int length)
131 {
132  char * s;
133  int depth, size;
134  int err = -EINVAL;
135 
136  depth = tcmi_ctlfs_entry_depth(TCMI_CTLFS_ENTRY(self));
137  /* we are interested in the depth of the parent entry, this is safe since
138  a minimum symlink depth is 1 (when it resides in the root directory) */
139  depth--;
140  size = tcmi_ctlfs_entry_path_length(self->target_entry) + depth * 3 - 1;
141  if (size > length) {
142  err = -ENAMETOOLONG;
143  goto exit0;
144  }
145  mdbg(INFO4,"depth = %d, size = %d", depth, size);
146 
147  for (s = path; depth--; s += 3)
148  strcpy(s,"../");
149 
150  if (self->target_entry)
151  tcmi_ctlfs_entry_fill_path(self->target_entry, path, size);
152 
153  mdbg(INFO3, "path = '%s'\n", path);
154 
155  return 0;
156 
157  /* error handling */
158  exit0:
159 /* tcmi_ctlfs_symlink_unlock(self); */
160  return err;
161 }

163 /**
164  * <<private>> Provides symlink resolution.
165  * Supplies the pathname of the target entry that the symlink refers to.
166  * This code is adapted from sysfs/symlink.c.
167  * - allocates a whole page for the pathname (faster than kmalloc)
168  * - extracts the symlink from the dentry
169  * - asks the symlink for path resolution
170  *
171  * @param *dentry - points to the dentry of the symlink that is to be
172  * resolved
173  * @param *nd - where the pathname is to be stored
174  * @return NULL upon success
175  */
176 static void* tcmi_ctlfs_symlink_follow_link(struct dentry *dentry, struct nameidata *nd)
177 {
178  int error = -ENOMEM;
179  unsigned long page = get_zeroed_page(GFP_ATOMIC);
180  struct tcmi_ctlfs_symlink *symlink;
181  if (page) {
182  symlink = TCMI_CTLFS_SYMLINK(TCMI_CTLFS_DENTRY_TO_ENTRY(dentry));
183  error = tcmi_ctlfs_symlink_get_link(symlink, (char *) page, PAGE_SIZE);
184  }
185  nd_set_link(nd, error ? ERR_PTR(error) : (char *)page);
186  return NULL;
187 }

189 /**
190  * <<private>> This VFS operation is called when the symlink pathname is not needed anymore by
191  * VFS. It allows us to release the memory needed for the pathname.
192  *
193  * @param *dentry - points to the dentry of the symlink
194  * @param *nd - where the pathname resides
195  */
196 static void tcmi_ctlfs_symlink_put_link(struct dentry *dentry, struct nameidata *nd, void * vd)
197 {
198  char *page = nd_get_link(nd);
199  mdbg(INFO4, "Destroying pathname related data");
200  if (!IS_ERR(page)) {
201  mdbg(INFO4, "Freeing pathname buffer");
202  free_page((unsigned long)page);
203  }
204 }
