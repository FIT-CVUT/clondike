43 /**
44  * <<public>> Skeleton message rx constructor.
45  * This method is called by the factory class.
46  *
47  * @param msg_id - message ID that will be used for this error message
48  * instance.
49  * @return a new error message or NULL.
50  */
51 struct tcmi_msg* tcmi_signal_msg_new_rx(u_int32_t msg_id)
52 {
53  struct tcmi_signal_msg *msg;
54 
55  /* Check if the factory is building what it really thinks. */
56  if (msg_id != TCMI_SIGNAL_MSG_ID) {
57  mdbg(ERR3, "Factory specified message ID(%x) doesn't match real ID(%x)",
58  msg_id, TCMI_SIGNAL_MSG_ID);
59  goto exit0;
60  }
61  /* Allocate the instance */
62  if (!(msg = TCMI_SIGNAL_MSG(kmalloc(sizeof(struct tcmi_signal_msg), GFP_KERNEL)))) {
63  mdbg(ERR3, "Can't allocate signal message");
64  goto exit0;
65  }
66  /* Initialized the message for receiving. */
67  if (tcmi_msg_init_rx(TCMI_MSG(msg), TCMI_SIGNAL_MSG_ID, &skel_msg_ops)) {
68  mdbg(ERR3, "Error initializing signal message %x", msg_id);
69  goto exit1;
70  }
71 
72  return TCMI_MSG(msg);
73 
74  /* error handling */
75  exit1:
76  kfree(msg);
77  exit0:
78  return NULL;
79 }

81 /**
82  * <<public>> Skeleton message tx constructor.
83  *
84  * The skeleton message can have a transaction associated with it =
85  * there is a response expected for this message by the
86  * sender. Therefore, the user specifies the transaction slot vector.
87  *
88  * When performing the generic tx init, the response message ID is
89  * specified (TCMI_RESPSIGNAL_MSG_ID) so that it will be associated with the
90  * transaction
91  *
92  * @param *transactions - storage for the new transaction(NULL for one-way messages)
93  * @return a new error ready for the transfer or NULL.
94  */
95 struct tcmi_msg* tcmi_signal_msg_new_tx(struct tcmi_slotvec *transactions)
96 {
97  struct tcmi_signal_msg *msg;
98 
99  if (!(msg = TCMI_SIGNAL_MSG(kmalloc(sizeof(struct tcmi_signal_msg), GFP_KERNEL)))) {
100  mdbg(ERR3, "Can't allocate signal message");
101  goto exit0;
102  }
103 
104  /* Initialize the message for transfer */
105  if (tcmi_msg_init_tx(TCMI_MSG(msg), TCMI_SIGNAL_MSG_ID, &skel_msg_ops,
106  transactions, 0,//TCMI_SIGNALRESP_MSG_ID,
107  TCMI_SIGNAL_MSGTIMEOUT, TCMI_TRANSACTION_INVAL_ID)) {
108  mdbg(ERR3, "Error initializing signal message");

110  }
111  return TCMI_MSG(msg);
112
