163 /**
164  * Polls real file and checks, if there are some data that can be read
165  * @return 0 if there are no data, positive number otherwise (negative on error)
166  */
167 static inline int proxyfs_real_file_poll_read(struct proxyfs_real_file* self) {
168  unsigned long mask;
169 
170  if (self->file->f_op && self->file->f_op->poll ) {
171  mask = self->file->f_op->poll( self->file, NULL );
172  mdbg(INFO3, "TEMPORARY DEBUG.. poll mask %lu", mask);
173  if ( !(mask & (POLLIN|POLLHUP)) ) // We chak as well for POLLHUP so that we get EOF
174  return 0; // Nothing can be read
175  } else {
176  mdbg(ERR1, "Cannot poll real file!");
177  return -1;
178  }
179  return 1;
180 }
181

192  if ( !(mask & (POLLOUT)) )
