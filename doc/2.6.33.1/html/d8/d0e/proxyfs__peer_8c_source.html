144 /** <<public>> Nonblocking recv messages
145  * @param self - pointer to proxyfs_peer_t instance
146  *
147  * @return 1 when complete message is received
148  */
149 int proxyfs_peer_real_recv(struct proxyfs_peer_t *self)
150 {
151  int result = 0;
152  int msg_size;
153 
154  if( self->recv_start < MSG_HDR_SIZE ){
155  result = kkc_sock_recv( self->sock, self->recv_buf + self->recv_start,
156  MSG_HDR_SIZE - self->recv_start, KKC_SOCK_NONBLOCK);
157 
158  if( result == -EAGAIN ){
159  //mdbg(INFO3, "No incoming data from %s", kkc_sock_getpeername2(self->sock));
160  }
161  else if( result > 0 ){
162  mdbg(INFO3, "Received %d bytes from %s", result, kkc_sock_getpeername2(self->sock));
163  self->recv_start += result;
164  }
165  else
166  mdbg(ERR3, "recv(%s) returned %d", kkc_sock_getpeername2(self->sock), result);
167  }
168  if( self->recv_start >= MSG_HDR_SIZE ){
169  msg_size = proxyfs_msg_get_size( self->recv_buf );
170 
171  if( msg_size == self->recv_start )
172  return 1; // Message is complete
173 
174  result = kkc_sock_recv( self->sock, self->recv_buf + self->recv_start,
175  msg_size - self->recv_start, KKC_SOCK_NONBLOCK);
176 
177  if( result == -EAGAIN ){
178  //mdbg(INFO3, "No incoming data from %s", kkc_sock_getpeername2(self->sock));
179  }
180  else if( result > 0 ){
181  mdbg(INFO3, "Received %d bytes from %s", result, kkc_sock_getpeername2(self->sock));
182  self->recv_start += result;
183  if( self->recv_start == proxyfs_msg_get_size(self->recv_buf) )
184  return 1; // Message is complete
185  else
186  return 0;
187  }
188  else
189  mdbg(ERR3, "recv(%s) returned %d", kkc_sock_getpeername2(self->sock), result);
190  }
191  return result;
192 }
