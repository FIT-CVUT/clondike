44  * <<public>> Test response message rx constructor. This method is

52 struct tcmi_msg* tcmi_skelresp_procmsg_new_rx(u_int32_t msg_id)
53 {
54  struct tcmi_skelresp_procmsg *msg;
55 
56  /* Check if the factory is building what it really thinks. */
57  if (msg_id != TCMI_SKELRESP_PROCMSG_ID) {

59  msg_id, TCMI_SKELRESP_PROCMSG_ID);

65  goto exit0;
66  }

71  goto exit1;
72  }
73 
74  return TCMI_MSG(msg);
75 
76  /* error handling */

83 /**
84  * <<public>> Testing response process message tx constructor.
85  *
86  * The testing response process message is a one-way message, no transaction
87  * is needs to be started, the user only needs to specify the
88  * transaction ID that it is the reply to and the destination process PID.
89  *
90  * When performing the generic tx init, the response message ID is
91  * specified (TCMI_SKELRESP_PROCMSG_ID) along with the destination PID.
92  *
93  * @param trans_id - transaction ID that this message is replying to.
94  * @param dst_pid - destination process PID
95  * @return a new test response message for the transfer or NULL.
96  */
97 struct tcmi_msg* tcmi_skelresp_procmsg_new_tx(u_int32_t trans_id, pid_t dst_pid)
98 {
99  struct tcmi_skelresp_procmsg *msg;
100 
101  if (!(msg = TCMI_SKELRESP_PROCMSG(kmalloc(sizeof(struct tcmi_skelresp_procmsg),
102  GFP_KERNEL)))) {
103  mdbg(ERR3, "Can't allocate testing message");
104  goto exit0;
105  }
106 
107  /* Initialize the message for transfer, no transaction
108  * required, no timout, no response ID */
109  if (tcmi_procmsg_init_tx(TCMI_PROCMSG(msg), TCMI_SKELRESP_PROCMSG_ID, &skelresp_procmsg_ops,
110  dst_pid, 0,
111  NULL, 0, 0, trans_id)) {
112  mdbg(ERR3, "Error initializing testing response message");
113  goto exit1;
114  }
115  return TCMI_MSG(msg);
116 
117  /* error handling */
118  exit1:
119  kfree(msg);
120  exit0:
121  return NULL;
122 
123 }
