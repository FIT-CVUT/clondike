52 /** @defgroup tcmi_module_class TCMI
53  *
54  * This <<singleton>> class is the base of the infrastructure and
55  * is responsible for initialization/shutdown of TCMI. The
56  * initializition requires notifying CCN and/or PEN manager resp.
57  *
58  * Initialized TCMI provides control in user space to:
59  * - migrate a process from a CCN to a PEN
60  * - migrate a process home from a PEN to a CCN
61  * - display information about migrated processes
62  * - connect a PEN to CCN
63  * - setup interfaces where CCN listens for incoming connections from
64  * PEN's
65  * - shutdown the whole infrastructure
66  * @{
67  */

110 /**
111  * Module initialization.
112  * Initializes 3 core components:
113  * - tcmi hooks in kernel
114  * - CCN manager - (when enabled in kernel)
115  * - PEN manager - (when enabled in kernel)
116  *
117  * The resulting node setup depends how TCMI has been configured in
118  * kernel.
119  *
120  * @return 0 upon successfull initialization
121  */
122 static int __init tcmi_module_init(void)
123 {
124  struct tcmi_ctlfs_entry *root;
125  minfo(INFO1, "Loading the TCMI framework");
126  tcmi_dbg = 1; // Enable debug temporarily at least for startup time
127 
128  root = tcmi_ctlfs_get_root();
129 
130  tcmi_mighooks_init();
131  tcmi_syscall_hooks_init();
132  tcmi_register_director_user_message_handler();
133 
134  if (tcmi_ccnman_init(root) < 0) {
135  minfo(ERR1, "Failed initializing TCMI CCN manager");
136  goto exit0;
137  }
138  if (tcmi_penman_init(root) < 0) {
139  minfo(ERR1, "Failed initializing TCMI PEN manager");
140  goto exit1;
141  }
142 
143  debug_file = tcmi_ctlfs_intfile_new(root, TCMI_PERMS_FILE_RW,
144  NULL, tcmi_debug_get, tcmi_debug_set,
145  sizeof(int), "debug");
146  if ( !debug_file ) {
147  minfo(ERR1, "Failed to create debug control file");
148  goto exit2;
149  }
150 
151 
152  minfo(INFO1, "TCMI framework loaded");
153  tcmi_dbg = 0; // Startup succeeded, disable debug, it can be reenabled via ctlfs
154  return 0;
155 
156  /* error handling */
157  exit2:
158  tcmi_penman_shutdown();
159  exit1:
160  tcmi_ccnman_shutdown();
161  exit0:
162  return -EINVAL;
163 }

168 static void __exit tcmi_module_exit(void)
169 {
170 
171  minfo(INFO1, "Unloading TCMI framework");
172 
173  /* Managers must be shutdown first, because they may need other parts of infrastructure to proceed with migration back */
174  tcmi_penman_shutdown();
175  tcmi_ccnman_shutdown();
176 
177  tcmi_mighooks_exit();
178  tcmi_syscall_hooks_exit();
179 
180  tcmi_ctlfs_file_unregister(debug_file);
181  tcmi_ctlfs_entry_put(debug_file);
182 
183  tcmi_ctlfs_put_root();
184 
185  minfo(INFO1, "TCMI framework unloaded");
186 }
