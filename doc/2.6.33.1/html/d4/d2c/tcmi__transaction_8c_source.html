42 /**
43  * <<public>> Creates a new transaction.
44  * Instance is created as follows:
45  * - allocate the instance
46  * - generate a transaction ID (random number)
47  * - initilialize wait queue for process(es) that is/are to be woken
48  * up when:
49  * - the transaction times out or
50  * - when it is sucessfully completed
51  * - store timeout that will be activated when the transaction is
52  * started.
53  * - store the transaction in the transaction slot vector specified by
54  * the user
55  * - setup reference counter
56  * - clear transaction flags - the transaction owner will adjust the
57  * flags afterwards
58  * - one instance reference is for the timer and one instance
59  * reference is returned to the instantiator
60  *
61  * @param *transactions - storage for the transactions
62  * @param resp_id - response message ID - required for the transaction so that match
63  * the response upon completion.
64  * @param timeout - transaction time out
65  * @return new transaction or NULL
66  */
67 struct tcmi_transaction* tcmi_transaction_new(struct tcmi_slotvec *transactions,
68  u_int32_t resp_id,
69  unsigned long timeout)
70 {
71  struct tcmi_transaction *trans;
72  /* hash of the transaction ID */
73  u_int hash;
74 
75  if (!(trans = kmalloc(sizeof(struct tcmi_transaction), GFP_KERNEL))) {
76  mdbg(ERR3, "Can't allocate memory for a transaction");
77  goto exit0;
78  }
79  tcmi_transaction_gen_id(trans);
80  hash = tcmi_transaction_hash(trans->id, tcmi_slotvec_hashmask(transactions));
81 
82  trans->resp_id = resp_id;
83  trans->state = TCMI_TRANSACTION_INIT;
84  trans->transactions = transactions;
85  init_waitqueue_head(&trans->wq);
86 
87  /* setup the timer that handles transaction timeout */
88  init_timer(&trans->timer);
89  /* the expiration has to be added the current value of jiffies
90  * when the timer is started, but this is upon transaction start */
91  trans->timer.expires = timeout;
92  trans->timer.data = (unsigned long) trans;
93  trans->timer.function = tcmi_transaction_timer;
94  /* context will be filled in upon response arrival */
95  trans->context = NULL;
96  /* one reference goes the the transaction instantiator */
97  atomic_set(&trans->ref_count, 1);
98  atomic_set(&trans->flags, 0);
99  spin_lock_init(&trans->t_lock);
100 
101  /* now, the transaction is ready to be inserted into the slot vector */
102  if (!(trans->slot = tcmi_slotvec_insert_at(transactions, hash, &trans->node))) {
103  minfo(ERR3, "Can't insert transaction into slot vector!");
104  goto exit1;
105  }
106  mdbg(INFO4, "Allocated transaction ID=%x, memory=%p",
107  tcmi_transaction_id(trans), trans);
108 
109  return trans;
110 
111  /* error handling */
112  exit1:
113  kfree(trans);
114  exit0:
115  return NULL;
116 }

119 /**
120  * <<public>> Starts a transaction. The transaction is started
121  * only if it is in INIT state. The timer is started, so that a
122  * timeout can be handled automatically. The timer is assigned an extra
123  * reference for the transaction to prevent destruction of the transaction
124  * before the timer goes off in case of timeout.
125  *
126  * @param *self - this transaction instance
127  * @param flags - flags to be set on the running transaction
128  * @return 0 upon successful start of the transaction
129  */
130 int tcmi_transaction_start(struct tcmi_transaction *self, int flags)
131 {
132  int err = 0;
133  tcmi_transaction_lock(self);
134  if (self->state != TCMI_TRANSACTION_INIT) {
135  mdbg(ERR3, "Can't start - transaction (ID=%x),"
136  " not in INIT state! state=%d", self->id, self->state);
137  err = -EINVAL;
138  goto exit0;
139  }
140  /* reference counter for the timer */
141  tcmi_transaction_get(self);
142  tcmi_transaction_set_flags(self, flags);
143  self->timer.expires += jiffies;
144  self->state = TCMI_TRANSACTION_RUNNING;
145  mdbg(INFO3, "Starting transaction ID %x, response ID %x",
146  self->id, self->resp_id);
147  add_timer(&self->timer);
148 
149  exit0:
150  tcmi_transaction_unlock(self);
151  return err;
152 }

154 /**
155  * <<public>> Aborting the transaction requires:
156  * - deleting the timer and checking if it hasn't expired
157  * - if the timer has been deleted without expiring, it is necessary
158  * to drop the reference that the timer was retaining
159  * - proceed further if the transaction is in INIT or RUNNING state
160  * and change the state to ABORTED.
161  * - notifying the transaction owner
162  *
163  * @param *self - pointer to this transaction instance
164  * @return if the transaction has already been completed, it
165  * will return a valid context and doesn't change the transaction state.
166  * An aborted transaction has still its context set to NULL.
167  */
168 void* tcmi_transaction_abort(struct tcmi_transaction *self)
169 {
170  int put_needed = 0;
171 
172  del_timer_sync(&self->timer);
173  tcmi_transaction_lock(self);
174  if (!tcmi_transaction_is_expired(self)) {
175  mdbg(INFO3, "Transaction hasn't expired yet, dropping timer reference.");
176  tcmi_transaction_set_expired(self);
177  put_needed = 1;
178  }
179  /* Transaction can be aborted from RUNNING as well as from
180  * INIT state. */
181  if (!tcmi_transaction_is_complete(self)) {
182  mdbg(INFO3, "Aborting..");
183  self->state = TCMI_TRANSACTION_ABORTED;
184  tcmi_transaction_notify_owner(self);
185  }
186  tcmi_transaction_unlock(self);
187 
188  if ( put_needed ) {
189  tcmi_transaction_put(self);
190  return NULL;
191  }
192 
193  return self->context;
194 }

196 /**
197  * <<public>> Completes a transaction and wakes up the processing
198  * thread. The completion requires:
199  * - deleting the running timer and checking if it hasn't expired
200  * - if the timer has been deleted without expiring, it is necessary
201  * to drop the reference that the timer was retaining
202  * - proceed further if the transaction is in running state ONLY
203  * - verifying, that the response ID in the transaction matches
204  * caller's expected response ID. The transaction is aborted if those
205  * two ID's don't match as apparently a wrong response has
206  * arrived.
207  * - upon successful completion, the context specified by the
208  * user, is associated with the transaction. That way, the transaction
209  * owner can read any data related to a particular transaction, when
210  * it picks up the completed transaction.
211  * - notifying the transaction owner
212  *
213  * Completing a transaction is performed under transaction lock being
214  * held.
215  *
216  * @param *self - this transaction instance
217  * @param *context - context that is to be assigned to
218  * the transaction upon successful completion.
219  * @param resp_id - response message ID that must match the ID set
220  * initially in the transaction.
221  * @return 0 when sucessfully completed
222  */
223 int tcmi_transaction_complete(struct tcmi_transaction *self, void *context,
224  u_int32_t resp_id)
225 {
226  /* default is error */
227  int err = -EINVAL;
228  int put_needed = 0;
229 
230  del_timer_sync(&self->timer);
231  tcmi_transaction_lock(self);
232  if (!tcmi_transaction_is_expired(self)) {
233  mdbg(INFO3, "Transaction hasn't expired yet, dropping timer reference.");
234  tcmi_transaction_set_expired(self);
235  put_needed = 1;
236  }
237  mdbg(INFO3, "Transaction state: %d", tcmi_transaction_state(self));
238  if (!tcmi_transaction_is_running(self)) {
239  mdbg(INFO3, "Can't complete-transaction ID %x is not running!", self->id);
240  goto exit0;
241  }
242  if (self->resp_id == resp_id) {
243  mdbg(INFO3, "Successfully completing transaction ID %x", self->id);
244  self->state = TCMI_TRANSACTION_COMPLETE;
245  self->context = context;
246  err = 0;
247  }
248  else {
249  self->state = TCMI_TRANSACTION_ABORTED;
250  mdbg(ERR4, "Aborting transaction ID %x, resp_id(%x)"
251  "doesn't match the expected resp_id(%x)!",
252  self->id, resp_id, self->resp_id);
253  }
254  /* In every case notify the transaction owner*/
255  tcmi_transaction_notify_owner(self);
256 
257  exit0:
258  tcmi_transaction_unlock(self);
259 
260  if ( put_needed )
261  tcmi_transaction_put(self);
262 
263  return err;
264 }

266 /**
267  * <<public>> Tries to find a transaction based on its ID. This
268  * is a class method that tries to find a transaction instance. It
269  * computes a hash of the transaction ID and searches the slot at
270  * index equal to the hash value.
271  *
272  * The access to the transaction is serialized via the slotvector that
273  * is being searched.
274  *
275  * @param *transactions - slot vector where to perform the search
276  * @param trans_id - ID used when searching for the transaction.
277  * @return transaction if found or NULL
278  */
279 struct tcmi_transaction* tcmi_transaction_lookup(struct tcmi_slotvec *transactions,
280  u_int32_t trans_id)
281 {
282  u_int hash;
283  /* slot where the hash of the transaction ID points to */
284  struct tcmi_slot *slot;
285  struct tcmi_transaction *tran = NULL;
286  /* temporary storage for the slot elements, needed by the iterator */
287  tcmi_slot_node_t *nd;
288 
289  hash = tcmi_transaction_hash(trans_id, tcmi_slotvec_hashmask(transactions));
290 
291  tcmi_slotvec_lock(transactions);
292  /* get the slot, the hash is pointing too */
293  if (!(slot = tcmi_slotvec_at(transactions, hash))) {
294  mdbg(ERR3, "Can't locate slot");
295  goto exit0;
296  }
297 
298  /* Search the slot for matching transaction */
299  tcmi_slot_for_each_entry(tran, nd, slot, node) {
300  if (tcmi_transaction_id(tran) == trans_id) {
301  minfo(INFO4,"Found matching transaction ID %x", trans_id);
302  break;
303  }
304 
305  }
306  /* succesfully found a transaction, adjust its reference counter */
307  if (tran)
308  tcmi_transaction_get(tran);
309 
310  tcmi_slotvec_unlock(transactions);
311 
312  return tran;
313 
314 /* error handling */
315  exit0:
316  tcmi_slotvec_unlock(transactions);
317  return NULL;
318 }

321 /**
322  * <<public>> Releases the instance. Decrements reference counter
323  * and if it reaches 0, the transaction is removed from its slot and
324  * freed.
325  *
326  * @param *self - pointer to this instance
327  */
328 void tcmi_transaction_put(struct tcmi_transaction *self)
329 {
330  struct tcmi_slotvec *transactions;
331 
332  if ( !self )
333  return;
334 
335  transactions = self->transactions;
336 
337  // Transactions slotvec lock used to guard against lookup/put races
338  tcmi_slotvec_lock(transactions);
339 
340  if (self && atomic_dec_and_test(&self->ref_count)) {
341  mdbg(INFO4, "Destroying transaction ID=%x, memory=%p",
342  tcmi_transaction_id(self), self);
343  tcmi_slot_remove(self->slot, &self->node);
344  kfree(self);
345  }
346 
347  tcmi_slotvec_unlock(transactions);
348 }

355 /**
356  * <<private>> Generates a transaction ID.
357  *
358  * @param *self - this transaction instance
359  */
360 static void tcmi_transaction_gen_id(struct tcmi_transaction *self)
361 {
362  u_int32_t id = TCMI_TRANSACTION_INVAL_ID;
363 
364  while (id == TCMI_TRANSACTION_INVAL_ID) {
365  get_random_bytes(&id, sizeof(u_int32_t));
366  }
367  self->id = id;
368  mdbg(INFO4, "Generated new transaction ID: %x", id);
369 }

371 /**
372  * <<private>> Callback for the kernel timer that handles
373  * transaction timeouts. This method is called upon timer expiration
374  * and means that no response has arrived so far, the transaction
375  * needs to be aborted. Expired flag is set to indicate that the
376  * timer has executed. This means that the 'complete' or 'abort'
377  * operations won't try to release the timer reference.
378  *
379  * Also, the thread that handles the transaction is woken up.
380  *
381  * Other transaction operations (complete, abort) are serialized with
382  * this method by means of del_timer_sync() - they cannot proceed any
383  * further until this method finishes.
384  *
385  * @param trans - pointer to the transaction instance
386  */
387 static void tcmi_transaction_timer(unsigned long trans)
388 {
389  struct tcmi_transaction *self = (struct tcmi_transaction*) trans;
390  self->state = TCMI_TRANSACTION_ABORTED;
391  mdbg(INFO3, "Transaction ID %x expired, aborting..", self->id);
392  tcmi_transaction_set_expired(self);
393  tcmi_transaction_notify_owner(self);
394 
395  tcmi_transaction_put(self);
396 }
