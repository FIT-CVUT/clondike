52  * <<public>> Initializes the TCMI manager.
53  * This generic init method, initializes basic singleton instance items
54  * common for both TCMI managers (CCN and PEN). The initialization is
55  * done in this order:
56  * - create a slot vector for migration managers
57  * - create TCMI ctlfs directories
58  * - create TCMI ctlfs control files
59  *
60  *
61  * @param *self - a particular singleton instance to be initialized
62  * @param *root - root directory of the filesystem, where all entries
63  * are to be created
64  * @param *ops - instance specific operations
65  * @param *man_dirname - name for the main directory of the manager (e.g. ccn or pen)
66  * also represent connected nodes (e.g. all PEN's from CCN prospective)
67  * @return 0 upon success
68  */

70  struct tcmi_ctlfs_entry *root,
71  struct tcmi_man_ops *ops,
72  const char *man_dirname)
73 {
74  if (!ops) {
75  mdbg(ERR3, "Missing migration manager operations!");
76  goto exit0;
77  }
78  atomic_set(&self->ready, 0);
79 
80  self->ops = ops;
81 
82  if (!(self->mig_mans = tcmi_slotvec_new(MAX_NODES)))
83  goto exit0;
84  if (tcmi_man_init_ctlfs_dirs(self, root, man_dirname) < 0)
85  goto exit1;

97  exit2:

105 /**
106  * Waits for all migration managers to terminate.
107  *
108  * Note: We perform wait by active polling of count of remaining migmas (100ms) interval.
109  * In case there were no remote processes running, there would be no wait as all migmans were terminated synchronously.
110  * In case there were remote processes, the waiting time for their migration back would be likely in order of seconds, so 100ms does not add much latency
111  * In case somebody volunteers, the code could be migrated to notify/wait patter to get rid of polling
112  */
113 static void tcmi_man_wait_for_migmans_termination(struct tcmi_man *self) {
114  int is_empty;
115 
116  tcmi_slotvec_lock(self->mig_mans);
117  is_empty = tcmi_slotvec_empty(self->mig_mans);
118  tcmi_slotvec_unlock(self->mig_mans);
119 
120  while (!is_empty) {
121  set_current_state(TASK_INTERRUPTIBLE);
122  schedule_timeout(msecs_to_jiffies(100));
123 
124  tcmi_slotvec_lock(self->mig_mans);
125  is_empty = tcmi_slotvec_empty(self->mig_mans);
126  tcmi_slotvec_unlock(self->mig_mans);
127  }
128 }

130 /**
131  * <<public>> The manager is shutdown exactly in this order:
132  * - unregister all ctlfs files as we don't want anymore requests
133  * from users.
134  *
135  * - shutdown the migration framework. This requires terminating all
136  * migration managers(results in termination or migration back of all
137  * processes)
138  * - unregister all ctlfs directories
139  *
140  * @param *self - a particular manager singleton instance to be shutdown
141  */
142 void tcmi_man_shutdown(struct tcmi_man *self)
143 {
144  if ( atomic_cmpxchg(&self->ready, 1, 0) == 1 ) {
145  tcmi_man_stop_ctlfs_files(self);
146  tcmi_man_stop_ctlfs_dirs(self);
147  /* Request async mig mans shutdown */
148  tcmi_man_stop_migration(self);
149  /* Wait for migmans to terminate */
150  tcmi_man_wait_for_migmans_termination(self);
151  tcmi_slotvec_put(self->mig_mans);
152  }
153 }

371 /** Finds and returns arbitrary migration manager that is either in INIT or CONNECTED state. */
372 static struct tcmi_migman * tcmi_man_get_non_shutdowning_migman(struct tcmi_man *self) {
373  struct tcmi_slot *slot;

383  migman_state = tcmi_migman_state(migman);

388  migman = NULL;
389  };
390  };
391 
392 found:
393  tcmi_slotvec_unlock(self->mig_mans);
394  return migman;

551  // Not sure what the intention was, but it does not work well.. interfere with getinode on ccfs initialization
552  // (there are actually more interferences and we ideally need to get rid of getinode in this call chain altoghether)
