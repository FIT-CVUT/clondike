1 /** @file dbg.h - a debugging and logging framework, based on mdbg
2  * library by Martin Kacer. The standard message types are divided
3  * into four categories:
4  * -# info messages (tracing only)
5  * -# warnings (something strange happend)
6  * -# errors (it is not possible to do something which should be possible to do)
7  * -# critical errors (serious program failures).
8  *
9  * Other than that, there is a number from 1 (very important) to 4
10  * (least important) assigned to each of the messages - the severity.
11  * Whenever the logging macro mdbg() is called, the given type of the
12  * message is compared to the mask stored in MDBG_MASK. All the bits
13  * that are set in the message type must also be set in the defined
14  * mask. In other words, the condition "(~mask & type) == 0" must be
15  * satisfied. The standard messages with higher severity have always
16  * less bits set than the ones with lower severity. That means
17  * enabling some type of message enables also all the other types in
18  * the same category which have higher severity. E.g. enabling
19  * 'MDBG_INFO2' enables also 'MDBG_INFO1' too.
20  *
21  * There are two macros provided:
22  * - mdbg()
23  * - minfo() - standard info output by the kernel application. The message
24  * is stored into a corresponding log file set by kernel logging framework.
25  * The user still specifies a message level type. This is needed as the message
26  * is still passed onto mdbg().
27  *
28  * The user has to define selected levels for each
29  * category if he/she doesn't want to use the defaults (which is
30  * 4). For example in a Makefile, it can be done as follows:
31  * \verbatim
32  * EXTRA_CFLAGS += -DMDBG_CRIT=4 -DMDBG_ERR=4 -DMDBG_WARN=4 -DMDBG_INFO=4
33  * \endverbatim
34  * This ensures that levels 1-4 will be displayed in the debug log for
35  * each category.
36  *
37  * Date: 05/05/2005
38  *
39  * Author: Jan Capek, Martin Kacer

154 /* creates a mask that has first b bits starting from LSb set and the

194 /**
195  * This function just allocs and frees various size memory blocks. It is intended to collaborate with kernel memory access check functions
196  * that check for memory poisioning. This call should be added to code just temporarily when chasing broken allocations and trying to find out
197  * when the memory is already broken.
198  */
199 #define memory_sanity_check(msg) \
200 do { \
201  int i = 1; \
202  int kmsize = 1; \
203  int test_size = 99; \
204  int res = 0; \
205  void** memtest; \
206  memtest = kmalloc(sizeof(void*)*test_size, GFP_KERNEL); \
207  for ( i = 0; i < test_size; i++ ) { \
208  memtest[i] = kmalloc(kmsize, GFP_KERNEL); \
209  kmsize = (kmsize * 2) % 8500; \
210  res = res + (int)memtest[i]; \
211  } \
212  for ( i = 0; i < test_size; i++ ) { \
213  kfree(memtest[i]); \
214  } \
215  printk(KERN_DEBUG __APP_NAME " %s()[%d]: " msg " -> Memory check -> %p\n", __FUNCTION__, current->pid, memtest); \
216  kfree(memtest); \
217 } while(0)
