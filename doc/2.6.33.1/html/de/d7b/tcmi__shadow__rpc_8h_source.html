43 /** Creates default RPC call for system call
44  *
45  * @param num - number of parameters
46  * @param name - name of system call
47  * @param use_errno - if true return errno when syscall returns -1
48  *
49  * @return system call return value
50  */
51 #define TCMI_SHADOW_RPC_GENERIC_CALL(num, name, use_errno)\
52 long tcmi_shadow_rpc_##name(unsigned rpc_num, long params[TCMI_RPC_MAXPARAMS]){\
53  struct tcmi_msg *m = (struct tcmi_msg*) params[0], **resp = (struct tcmi_msg**) params[1];\
54  long rtn; _TCMI_SHADOW_RPC_GENERIC_CALL##num(name); \
55  if( use_errno && rtn == -1) rtn = errno;\
56  *resp = tcmi_rpcresp_procmsg_create(m, rtn, (size_t)0 );\
57  if( *resp == NULL ) return -1;return 0;\
58 }
