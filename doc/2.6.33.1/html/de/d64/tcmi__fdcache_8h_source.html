37 /** @defgroup tcmi_fdcache_class tcmi_fdcache class
38  *
39  * @ingroup tcmi_ckpt_class
40  *
41  * This is a helper class used by the checkpoint to keep track of
42  * already open file descriptors. It allows easy lookup of a duplicate
43  * file descriptor. This is needed when checkpointing open files of a
44  * process.
45  *
46  * There is no need for locking nor reference counting as fdcache is a
47  * temporary object (as well as \link tcmi_ckpt_class a checkpoint
48  * object \endlink) accessed from a single thread of execution. The
49  * reference counter is just to keep it consistent with current
50  * software design.
51  *
52  * @{
53  */

55 /** Describes an fd cache entry. */
56 struct tcmi_fdcache_entry {
57  /** file descriptor. */
58  int fd;
59  /** file that references. */
60  struct file *file;
61 };

63 /** Compound structure contains all fd cache entries. */
64 struct tcmi_fdcache {
65  /** File descriptor cache - used when indentifying descriptors
66  * referencing the same file. */
67  struct tcmi_fdcache_entry *entries;
68  /** current files in the cache. */
69  int used;
70  /** actual cache size. */
71  int size;
72  /** Instance reference counter. */
73  atomic_t ref_count;
74 };

76 /**
77  * <<public>> Creates a file descriptor cache of the specified
78  * size. This requires:
79  * - creating new instance
80  * - allocating a requested number of entries
81  *
82  * @param size - maximum number of entries - should match the current
83  * number of open files of a process - this handles the worst case
84  * when there are none duplicate descriptors but one.
85  * @return fdcache instance or NULL
86  */
87 static inline struct tcmi_fdcache* tcmi_fdcache_new(int size)
88 {
89  struct tcmi_fdcache *fdcache;
90  if (!(fdcache = kmalloc(sizeof(struct tcmi_fdcache),
91  GFP_KERNEL))) {
92  mdbg(ERR3, "Can't allocate memory for TCMI fdcache");
93  goto exit0;
94  }
95  if (!(fdcache->entries = kmalloc(sizeof(struct tcmi_fdcache_entry) * size,
96  GFP_KERNEL))) {
97  mdbg(ERR3, "Can't get memory for TCMI fdcache %d entries", size);
98  goto exit1;
99  }
100  atomic_set(&fdcache->ref_count, 1);
101  fdcache->size = size;
102  /* no used entries yet */
103  fdcache->used = 0;
104 
105  return fdcache;
106  /* error handling */
107  exit1:
108  kfree(fdcache);
109  exit0:
110  return NULL;
111 }

113 /**
114  * <<public>> Instance accessor, increments the reference counter.
115  *
116  * @param *self - pointer to this fdcache instance
117  * @return fd cache instance
118  */
119 static inline struct tcmi_fdcache* tcmi_fdcache_get(struct tcmi_fdcache *self)
120 {
121  if (self) {
122  atomic_inc(&self->ref_count);
123  }
124  return self;
125 }

127 /**
128  * <<public>> Releases the instance.
129  *
130  * @param *self - pointer to this fd cache instance
131  */
132 static inline void tcmi_fdcache_put(struct tcmi_fdcache *self)
133 {
134  if (self && atomic_dec_and_test(&self->ref_count)) {
135  mdbg(INFO4, "Destroying TCMI fdcache, %p", self);
136  kfree(self->entries);
137  kfree(self);
138  }
139 }

141 /**
142  * <<public>> Adds a file descriptor - file object pair into the
143  * fd cache.
144  *
145  * @param *self - this fdcache instance
146  * @param fd - file descriptor of the file
147  * @param *file - file object - used for duplicate file descriptor lookup
148  * @return -ENOMEM if the cache is full
149  */
150 static inline int tcmi_fdcache_add(struct tcmi_fdcache *self, int fd, struct file *file)
151 {
152  struct tcmi_fdcache_entry *entry;
153  if (!self || (self->used == self->size)) {
154  mdbg(ERR3, "TCMI fdcache full or not properly instantiated!");
155  goto exit0;
156  }
157  entry = &self->entries[self->used++];
158  entry->fd = fd;
159  entry->file = file;
160  return 0;
161 
162  /* error handling */
163  exit0:
164  return -ENOMEM;
165 }

167 /**
168  * <<public>> Searches the file descriptor cache for a file.
169  *
170  * @param *self - this fdcache instance
171  * @param *file - file object to be looked up
172  * @return file descriptor of the file or error (< 0)
173  */
174 static inline int tcmi_fdcache_lookup(struct tcmi_fdcache *self, struct file *file)
175 {
176  int i;
177  int fd = -EINVAL;
178  /* safes dereferences */
179  int used = self->used;
180  struct tcmi_fdcache_entry *entry = self->entries;
181  for (i = 0; i < used; i++, entry++) {
182  if (entry->file == file) {
183  fd = entry->fd;
184  mdbg(INFO4, "Found descriptor %d in the fdcache", fd);
185  break;
186  }
187  }
188 
189  return fd;
190 }
